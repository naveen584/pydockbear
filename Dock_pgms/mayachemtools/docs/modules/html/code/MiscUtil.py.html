  35 def CheckFileExt(FileName, FileExts):
  36     """Check file type based on the specified file extensions delimited by spaces.
  37     
  38     Arguments:
  39         FileName (str): Name of a file.
  40         FileExts (str): Space delimited string containing valid file extensions.
  41 
  42     Returns:
  43         bool : True, FileName contains a valid file extension; Otherwise, False.
  44 
  45     """
  46     
  47     for FileExt in FileExts.split():
  48         if re.search(r"\.%s$" % FileExt, FileName, re.IGNORECASE):
  49             return True
  50     
  51     return False

  53 def CheckTextValue(Value, ValidValues):
  54     """Check text value based on the specified valid values delimited by spaces.
  55 
  56     Arguments:
  57         Value (str): Text value
  58         ValidValues (str): Space delimited string containing valid values.
  59 
  60     Returns:
  61         bool : True, Value is valid; Otherwise, False.
  62 
  63     """
  64     
  65     ValidValues = re.sub(' ', '|', ValidValues)
  66     if re.match("^(%s)$" % ValidValues, Value, re.IGNORECASE):
  67         return True
  68     
  69     return False

  71 def GetTextLinesWords(TextFilePath, Delimiter, QuoteChar, HeaderLinePresent):
  72     """Parse lines in the specified text file into words in a line and return a list containing
  73     list of parsed line words.
  74 
  75     Arguments:
  76         TextFilePath (str): Text file name including file path.
  77         Delimiter (str): Delimiter for parsing text lines.
  78         QuoteChar (str): Quote character for line words.
  79         HeaderLinePresent (bool): A flag indicating presence of a header line.
  80 
  81     Returns:
  82         list : A list of lists containing parsed words for lines.
  83 
  84     Notes:
  85         The lines starting with # or // are considered comment lines and are
  86         ignored during parsing along with any empty lines.
  87 
  88     """
  89     TextFile = open(TextFilePath, "r")
  90     if TextFile is None:
  91         PrintError("Couldn't open text file: %s.\n" % (TextFilePath))
  92 
  93     # Collect text lines...
  94     TextLines = []
  95     HeaderLine = HeaderLinePresent
  96     for Line in TextFile:
  97         Line = Line.strip()
  98         
  99         # Ignore empty lines...
 100         if not len(Line):
 101             continue
 102         
 103         # Ignore comments...
 104         if re.match("^(#|\/\/)", Line, re.I):
 105             continue
 106 
 107         # Ignore header line...
 108         if HeaderLine:
 109             HeaderLine = False
 110             continue
 111         
 112         TextLines.append(Line)
 113         
 114     TextFile.close()
 115 
 116     # Parse text lines...
 117     TextLinesWords = []
 118     
 119     TextLinesReader = csv.reader(TextLines, delimiter = Delimiter, quotechar = QuoteChar)
 120     for LineWords in TextLinesReader:
 121         TextLinesWords.append(LineWords)
 122     
 123     return TextLinesWords

 125 def DoesSMILESFileContainTitleLine(FileName):
 126     """Determine whether the SMILES file contain a title line based on the  presence
 127     of a string SMILES, Name or ID in the first line.
 128 
 129     Arguments:
 130         FileName (str): Name of a file.
 131 
 132     Returns:
 133         bool : True, File contains title line; Otherwise, False.
 134 
 135     """
 136     
 137     Infile = open(FileName, "r")
 138     if Infile is None:
 139         return False
 140 
 141     Line = Infile.readline()
 142     Infile.close()
 143 
 144     if re.search("(SMILES|Name|ID)", Line, re.I):
 145         return True
 146         
 147     return False

 149 def GetExamplesTextFromDocOptText(DocOptText):
 150     """Get script usage example lines from a docopt doc string. The example text
 151     line start from a line containing `Examples:`  keyword at the beginning of the line.
 152     
 153     Arguments:
 154         DocOptText (str): Doc string containing script usage examples lines starting with
 155             a line marked by `Examples:` keyword at the beginning of a line.
 156 
 157     Returns:
 158         str : A string containing text lines retrieved from the examples section of
 159             DocOptText parameter.
 160 
 161     """
 162     
 163     ExamplesStart = re.compile("^Examples:", re.IGNORECASE)
 164     ExamplesEnd = re.compile("^(Author:|See also:|Copyright:)", re.IGNORECASE)
 165     
 166     ExamplesText = 'Examples text is not available';
 167     ExamplesTextFound = False
 168     
 169     for Line in DocOptText.splitlines():
 170         if ExamplesStart.match(Line):
 171             ExamplesText = 'Examples:';
 172             ExamplesTextFound = True
 173             continue
 174         
 175         if ExamplesEnd.match(Line):
 176             break
 177         
 178         if ExamplesTextFound:
 179             ExamplesText += "\n" + Line;
 180     
 181     return ExamplesText

 193 def GetMayaChemToolsLibDataPath():
 194     """Get location of MayaChemTools lib data directory.
 195     
 196     Returns:
 197         str : Location of MayaChemTools lib data directory.
 198 
 199     Notes:
 200         The location of MayaChemTools lib data directory is determined relative to
 201         MayaChemTools python lib directory name available through sys.path.
 202 
 203     """
 204     MayaChemToolsDataPath = ""
 205     
 206     for PathEntry in sys.path:
 207         if re.search("MayaChemTools", PathEntry, re.I) and re.search("Python", PathEntry, re.I):
 208             MayaChemToolsDataPath = os.path.join( PathEntry, "..",  "data")
 209             break
 210         else:
 211             PrintInfo("PathEntry didn't match")
 212     
 213     if not len(MayaChemToolsDataPath):
 214         PrintWarning("MayaChemTools lib directory location doesn't appear to exist in system search path specified by sys.path...")
 215         
 216     return MayaChemToolsDataPath

 222     Arguments:

 226     Returns:
 227         str : Elapsed time formatted as: %d wallclock secs ( %.2f process secs)

 229     """

 238 def ParseFileName(FilePath):
 239     """Parse specified file path and return file dir, file name, and file extension.
 240     
 241     Arguments:
 242         FilePath (str): Name of a file with complete file path.

 245         str : File directory.

 325 def ValidateOptionNumberValues(OptionName, OptionValueString, OptionValueCount, OptionValueDelimiter, OptionValueType, CmpOpValueMap):
 326     """Validate numerical option values using option value string, delimiter, value type,
 327     and a specified map containing comparison operator and value pairs.
 328     
 329     Arguments:
 330         OptionName (str): Command line option name.
 331         OptionValueString (str): Command line option value.
 332         OptionValueCount (int): Number of values in OptionValueString.
 333         OptionValueDelimiter (str): Delimiter used for values in OptionValueString.
 334         OptionValueType (str): Valid number types (integer or float)
 335         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 336             validate values specified in OptionValueString.
 337 
 338     Notes:
 339         The function exits with an error message for invalid option values specified
 340         in OptionValueString
 341 
 342     Examples:
 343 
 344         ValidateOptionNumberValues("-m, --molImageSize",
 345             Options["--molImageSize"], 2, ",", "integer", {">": 0})
 346 
 347     """
 348     if not CheckTextValue(OptionValueType, "integer float"):
 349         PrintError("The option value type specified, %s, for function MiscUtil.ValidateOptionNumberValues  is not valid. Supported value: integer float " % (OptionValueType))
 350         
 351     Values = OptionValueString.split(OptionValueDelimiter)
 352     if len(Values) != OptionValueCount:
 353         PrintError("The value specified, %s, for option \"%s\" is not valid. It must contain %d %s values separated by \"%s\"" % (OptionValueString, OptionName, OptionValueCount, OptionValueType, OptionValueDelimiter))
 354 
 355     IsIntergerValue = True
 356     if re.match("^float$", OptionValueType, re.I):
 357         IsIntergerValue = False
 358     
 359     for Value in Values:
 360         if IsIntergerValue:
 361             Value = int(Value)
 362         else:
 363             Value = float(Value)
 364         ValidateOptionNumberValue(OptionName, Value, CmpOpValueMap)

 366 def ValidateOptionNumberValue(OptionName, OptionValue, CmpOpValueMap):
 367     """Validate option value using comparison operater and value pairs in specified in
 368     a map.
 369     
 370     Arguments:
 371         OptionName (str): Command line option name.
 372         OptionValue (str): Command line option value.
 373         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 374             validate values specified in OptionValue.
 375 
 376     Notes:
 377         The function exits with an error message for an invalid option values specified
 378         in OptionValue.
 379 
 380     Examples:
 381 
 382         ValidateOptionNumberValue("--maxConfs", int(Options["--maxConfs"]),
 383             {">": 0})
 384         ValidateOptionNumberValue("-b, --butinaSimilarityCutoff", 
 385             float(Options["--butinaSimilarityCutoff"]),
 386             {">": 0.0, "<=" : 1.0})
 387 
 388     """
 389     
 390     Status = True
 391     for CmpOp in CmpOpValueMap:
 392         Value = CmpOpValueMap[CmpOp]
 393         if re.match("^>$", CmpOp, re.I):
 394             if OptionValue <= Value:
 395                 Status = False
 396                 break
 397         elif re.match("^>=$", CmpOp, re.I):
 398             if OptionValue < Value:
 399                 Status = False
 400                 break
 401         elif re.match("^<$", CmpOp, re.I):
 402             if OptionValue >= Value:
 403                 Status = False
 404                 break
 405         elif re.match("^<=$", CmpOp, re.I):
 406             if OptionValue > Value:
 407                 Status = False
 408                 break
 409         else:
 410             PrintError("The specified comparison operator, %s, for function MiscUtil.ValidateOptionNumberValue is not supported\n" % (CmpOp))
 411     
 412     if not Status:
 413         FirstValue = True
 414         SupportedValues = ""
 415         for CmpOp in CmpOpValueMap:
 416             Value = CmpOpValueMap[CmpOp]
 417             if FirstValue:
 418                 FirstValue = False
 419                 SupportedValues = "%s %s" % (CmpOp, Value)
 420             else:
 421                 SupportedValues = "%s and %s %s" % (SupportedValues, CmpOp, Value)
 422         
 423         PrintError("The value specified, %s, for option \"%s\" is not valid. Supported value(s): %s " % (OptionValue, OptionName, SupportedValues))

 507 def ProcessOptionOutfileParameters(ParamsOptionName, ParamsOptionValue, InfileName = None, OutfileName = None):
 508     """Process parameters for writing output files and return a map containing
 509     processed parameter names and values.
 510     
 511     Arguments:
 512         ParamsOptionName (str): Command line input parameters option name.
 513         ParamsOptionValues (str): Comma delimited list of parameter name and value pairs.
 514         InfileName (str): Name of input file.
 515         OutfileName (str): Name of output file.
 516 
 517     Returns:
 518         dictionary: Processed parameter name and value pairs.
 519 
 520     Notes:
 521         The parameter name and values specified in ParamsOptionValues are validated before
 522         returning them in a dictionary.
 523 
 524         The default value of some parameters may depend on type of input file. Consequently,
 525         the input file name is also needed.
 526 
 527     """
 528     ParamsInfo = {'Compute2DCoords' : 'auto', 'Kekulize': False, 'SMILESDelimiter': ' ', 'SMILESIsomeric': True,  'SMILESTitleLine': True}
 529     _ProcessInfileAndOutfileParameters('Outfile', ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)

 533 def _ProcessInfileAndOutfileParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName):
 534     """Process specified infile and outfile paramaters.
 535     
 536     """
 537     if re.match("^auto$", ParamsOptionValue, re.I):
 538         # No specific parameters to process except for parameters with possible auto value...
 539         _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)
 540         return
 541     
 542     ParamsOptionValue = re.sub(" ", "", ParamsOptionValue)
 543     if not ParamsOptionValue:
 544         PrintError("No valid parameter name and value pairs specified using \"%s\" option" % ParamsOptionName)
 545     
 546     ParamsOptionValueWords = ParamsOptionValue.split(",")
 547     if len(ParamsOptionValueWords) % 2:
 548         PrintError("The number of comma delimited paramater names and values, %d, specified using \"%s\" option must be an even number." % (len(ParamsOptionValueWords), ParamsOptionName))
 549         
 550     # Setup a canonical paramater names...
 551     ValidParamNames = []
 552     CanonicalParamNamesMap = {}
 553     for ParamName in sorted(ParamsInfo):
 554         ValidParamNames.append(ParamName)
 555         CanonicalParamNamesMap[ParamName.lower()] = ParamName
 556     
 557     # Validate paramater name and value pairs...
 558     for Index in range(0, len(ParamsOptionValueWords), 2):
 559         Name = ParamsOptionValueWords[Index]
 560         Value = ParamsOptionValueWords[Index + 1]
 561 
 562         CanonicalName = Name.lower()
 563         if  not CanonicalName in CanonicalParamNamesMap:
 564             PrintError("The parameter name, %s, specified using \"%s\" is not a valid name. Supported parameter names: %s" % (Name, ParamsOptionName, " ".join(ValidParamNames)))
 565 
 566         ParamName = CanonicalParamNamesMap[CanonicalName]
 567         ParamValue = Value
 568         
 569         if re.match("^(Sanitize|StrictParsing|RemoveHydrogens|Kekulize|SMILESIsomeric)$", ParamName, re.I):
 570             if not re.match("^(Yes|No|True|False)$", Value, re.I):
 571                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 572             ParamValue = True
 573             if re.match("^(No|False)$", Value, re.I):
 574                 ParamValue = False
 575         elif re.match("^SMILESTitleLine$", ParamName, re.I):
 576             if re.match("^Infile$", Mode, re.I):
 577                 if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 578                     PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 579             elif re.match("^Outfile$", Mode, re.I):
 580                 if not re.match("^(Yes|No|True|False)$", Value, re.I):
 581                     PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 582                 ParamValue = True
 583                 if re.match("^(No|False)$", Value, re.I):
 584                     ParamValue = False
 585         elif re.match("^SMILESDelimiter$", ParamName, re.I):
 586             if not re.match("^(space|tab|comma)$", Value, re.I):
 587                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: space tab comma" % (Value, Name, ParamsOptionName))
 588             ParamValue = " "
 589             if re.match("^tab$", Value, re.I):
 590                 ParamValue = "\t"
 591             elif re.match("^comma$", Value, re.I):
 592                 ParamValue = ","
 593         elif re.match("^Compute2DCoords$", ParamName, re.I):
 594             # No need to set the value. It would be processed later to handle "auto" value...
 595             if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 596                 PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 597         else:
 598             ParamValue = int(Value)
 599             if ParamValue <= 0:
 600                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: > 0" % (Value, Name, ParamsOptionName))
 601         
 602         # Set value...
 603         ParamsInfo[ParamName] = ParamValue
 604         
 605     # Handle paramaters with possible auto values...
 606     _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)
 607 
 608 def _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName):
 609     """Process parameters with possible auto values.
 610     
 611     """
 612     if re.match("^Infile$", Mode, re.I):
 613         # SMILESTitleLine parameter...
 614         Value = ParamsInfo["SMILESTitleLine"]
 615         ParamValue = False
 616         if re.match("^auto$", Value, re.I):
 617             if InfileName is not None:
 618                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 619                     ParamValue = DoesSMILESFileContainTitleLine(InfileName)
 620         elif re.match("^(Yes|True)$", Value, re.I):
 621             ParamValue = True
 622         ParamsInfo["SMILESTitleLine"] = ParamValue
 623     elif re.match("^Outfile$", Mode, re.I):
 624         # Compute2DCoords parameter...
 625         Value = ParamsInfo["Compute2DCoords"]
 626         ParamValue = False
 627         if re.match("^auto$", Value, re.I):
 628             if InfileName is not None:
 629                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 630                     ParamValue = True
 631         elif re.match("^(Yes|True)$", Value, re.I):
 632             ParamValue = True
 633         ParamsInfo["Compute2DCoords"] = ParamValue
