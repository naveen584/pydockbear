  41 def GetMolName(Mol, MolNum = None):
  42     """Get molecule name.
  43     
  44     Arguments:
  45         Mol (object): RDKit molecule object.
  46         MolNum (int or None): Molecule number in input file.
  47 
  48     Returns:
  49         str : Molname corresponding to _Name property of a molecule, generated
  50             from specieid MolNum using the format "Mol%d" % MolNum, or an
  51             empty string.
  52 
  53     """
  54     
  55     MolName = ''
  56     if Mol.HasProp("_Name"):
  57         MolName = Mol.GetProp("_Name")
  58 
  59     if not len(MolName):
  60         if MolNum is not None:
  61             MolName = "Mol%d" % MolNum
  62     
  63     return MolName
  64 
  65 def IsMolEmpty(Mol):
  66     """Check for the presence of atoms in a molecule.
  67     
  68     Arguments:
  69         Mol (object): RDKit molecule object.
  70 
  71     Returns:
  72         bool : True - No atoms in molecule; Otherwise, false. 
  73 
  74     """
  75 
  76     Status = False if Mol.GetNumAtoms() else True
  77     
  78     return Status

  80 def ReadAndValidateMolecules(FileName, **KeyWordArgs):
  81     """Read molecules from an input file, validate all molecule objects, and return
  82     a list of valid and non-valid molecule objects along with their counts.
  83     
  84     Arguments:
  85         FileName (str): Name of a file with complete path.
  86         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
  87             processing molecules.
  88 
  89     Returns:
  90         list : List of valid RDKit molecule objects.
  91         int : Number of total molecules in input file. 
  92         int : Number of valid molecules in input file. 
  93 
  94     Notes:
  95         The file extension is used to determine type of the file and set up an appropriate
  96         file reader.
  97 
  98     """
  99 
 100     AllowEmptyMols = True
 101     if "AllowEmptyMols" in KeyWordArgs:
 102         AllowEmptyMols = KeyWordArgs["AllowEmptyMols"]
 103     
 104     Mols = ReadMolecules(FileName, **KeyWordArgs)
 105 
 106     if AllowEmptyMols:
 107         ValidMols = [Mol for Mol in Mols if Mol is not None]
 108     else:
 109         ValidMols = []
 110         MolCount = 0
 111         for Mol in Mols:
 112             MolCount += 1
 113             if Mol is None:
 114                 continue
 115             
 116             if IsMolEmpty(Mol):
 117                 MolName = GetMolName(Mol, MolCount)
 118                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 119                 continue
 120             
 121             ValidMols.append(Mol)
 122             
 123     MolCount = len(Mols)
 124     ValidMolCount = len(ValidMols)
 125 
 126     return (ValidMols, MolCount, ValidMolCount)

 128 def ReadMolecules(FileName, **KeyWordArgs):
 129     """Read molecules from an input file without performing any validation
 130     and creation of molecule objects.
 131     
 132     Arguments:
 133         FileName (str): Name of a file with complete path.
 134         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 135             processing molecules.
 136 
 137     Returns:
 138         list : List of RDKit molecule objects.
 139 
 140     Notes:
 141         The file extension is used to determine type of the file and set up an appropriate
 142         file reader.
 143 
 144     """
 145 
 146     # Set default values for possible arguments...
 147     ReaderArgs = {"Sanitize": True, "RemoveHydrogens": True, "StrictParsing": True,  "SMILESDelimiter" : ' ', "SMILESColumn": 1, "SMILESNameColumn": 2, "SMILESTitleLine": True }
 148 
 149     # Set specified values for possible arguments...
 150     for Arg in ReaderArgs:
 151         if Arg in KeyWordArgs:
 152             ReaderArgs[Arg] = KeyWordArgs[Arg]
 153 
 154     # Modify specific valeus for SMILES...
 155     if MiscUtil.CheckFileExt(FileName, "smi csv tsv txt"):
 156         Args = ["Sanitize", "SMILESTitleLine"]
 157         for Arg in Args:
 158             if ReaderArgs[Arg] is True:
 159                 ReaderArgs[Arg] = 1
 160             else:
 161                 ReaderArgs[Arg] = 0
 162     
 163     Mols = []
 164     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 165         return ReadMoleculesFromSDFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 166     elif MiscUtil.CheckFileExt(FileName, "mol"):
 167         return ReadMoleculesFromMolFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 168     elif MiscUtil.CheckFileExt(FileName, "mol2"):
 169         return ReadMoleculesFromMol2File(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 170     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 171         return ReadMoleculesFromPDBFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 172     elif MiscUtil.CheckFileExt(FileName, "smi txt csv tsv"):
 173         SMILESColumnIndex = ReaderArgs["SMILESColumn"] - 1
 174         SMILESNameColumnIndex = ReaderArgs["SMILESNameColumn"] - 1
 175         return ReadMoleculesFromSMILESFile(FileName, ReaderArgs["SMILESDelimiter"], SMILESColumnIndex, SMILESNameColumnIndex, ReaderArgs["SMILESTitleLine"], ReaderArgs["Sanitize"])
 176     else:
 177         MiscUtil.PrintWarning("RDKitUtil.ReadMolecules: Non supported file type: %s" % FileName)
 178     
 179     return Mols

 181 def ReadMoleculesFromSDFile(FileName, Sanitize = True, RemoveHydrogens = True, StrictParsing = True):
 182     """Read molecules from a SD file.
 183     
 184     Arguments:
 185         FileName (str): Name of a file with complete path.
 186         Sanitize (bool): Sanitize molecules.
 187         RemoveHydrogens (bool): Remove hydrogens from molecules.
 188         StrictParsing (bool): Perform strict parsing.
 189 
 190     Returns:
 191         list : List of RDKit molecule objects.
 192 
 193     """
 194     return  Chem.SDMolSupplier(FileName, sanitize = Sanitize, removeHs = RemoveHydrogens, strictParsing = StrictParsing)

 248 def ReadMoleculesFromSMILESFile(FileName, SMILESDelimiter = ' ', SMILESColIndex = 0, SMILESNameColIndex = 1, SMILESTitleLine = 1, Sanitize = 1):
 249     """Read molecules from a SMILES file.
 250

 253         SMILESColIndex (int): Column index containing SMILES string.
 254         SMILESNameColIndex (int): Column index containing molecule name.

 265 def MoleculesWriter(FileName, **KeyWordArgs):
 266     """Set up a molecule writer.
 267     
 268     Arguments:
 269         FileName (str): Name of a file with complete path.
 270         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 271             processing molecules.
 272 
 273     Returns:
 274         RDKit object : Molecule writer.
 275 
 276     Notes:
 277         The file extension is used to determine type of the file and set up an appropriate
 278         file writer.
 279 
 280     """
 281     
 282     # Set default values for possible arguments...
 283     WriterArgs = {"Compute2DCoords" : False, "Kekulize": False, "SMILESDelimiter" : ' ', "SMILESIsomeric": True, "SMILESTitleLine": True }
 284 
 285     # Set specified values for possible arguments...
 286     for Arg in WriterArgs:
 287         if Arg in KeyWordArgs:
 288             WriterArgs[Arg] = KeyWordArgs[Arg]
 289     
 290     Writer = None
 291     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 292         Writer = Chem.SDWriter(FileName)
 293         if WriterArgs["Kekulize"]:
 294             Writer.SetKekulize(True)
 295     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 296         Writer = Chem.PDBWriter(FileName)
 297     elif MiscUtil.CheckFileExt(FileName, "smi"):
 298         # Text for the name column in the title line. Blank indicates not to include name column
 299         # in the output file...
 300         NameHeader = 'Name'
 301         Writer = Chem.SmilesWriter(FileName, delimiter = WriterArgs["SMILESDelimiter"], nameHeader = NameHeader, includeHeader = WriterArgs["SMILESTitleLine"],  isomericSmiles = WriterArgs["SMILESIsomeric"], kekuleSmiles = WriterArgs["Kekulize"])
 302     else:
 303         MiscUtil.PrintWarning("RDKitUtil.WriteMolecules: Non supported file type: %s" % FileName)
 304     
 305     return Writer

 307 def WriteMolecules(FileName, Mols, **KeyWordArgs):
 308     """Write molecules to an output file.
 309     
 310     Arguments:
 311         FileName (str): Name of a file with complete path.
 312         Mols (list): List of RDKit molecule objects. 
 313         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 314             processing molecules.
 315 
 316     Returns:
 317         int : Number of total molecules.
 318         int : Number of processed molecules written to output file.
 319 
 320     Notes:
 321         The file extension is used to determine type of the file and set up an appropriate
 322         file writer.
 323 
 324     """
 325     
 326     Compute2DCoords = False
 327     if "Compute2DCoords" in KeyWordArgs:
 328         Compute2DCoords = KeyWordArgs["Compute2DCoords"]
 329         
 330     MolCount = len(Mols)
 331     ProcessedMolCount = 0
 332     
 333     Writer = MoleculesWriter(FileName, **KeyWordArgs)
 334     
 335     if Writer is None:
 336         return (MolCount, ProcessedMolCount)
 337     
 338     for Mol in Mols:
 339         if Mol is None:
 340             continue
 341         
 342         ProcessedMolCount += 1
 343         if Compute2DCoords:
 344             AllChem.Compute2DCoords(Mol)
 345         
 346         Writer.write(Mol)
 347     
 348     Writer.close()
 349     
 350     return (MolCount, ProcessedMolCount)
