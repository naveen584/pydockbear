  94 # Generate atom types for a SD file...
  95 #
  96 sub GenerateAtomTypes {
  97   my($FileIndex) = @_;
  98   my($CmpdCount, $IgnoredCmpdCount, $SDFile, $MoleculeFileIO, $Molecule, $AtomTypes);
  99 
 100   $SDFile = $SDFilesList[$FileIndex];
 101 
 102   # Setup output files...
 103   # demo:
 104 
 105   $MoleculeFileIO = new MoleculeFileIO('Name' => $SDFile);
 106   $MoleculeFileIO->Open();
 107 
 108   $CmpdCount = 0;
 109   $IgnoredCmpdCount = 0;
 110 
 111   COMPOUND: while ($Molecule = $MoleculeFileIO->ReadMolecule()) {
 112     $CmpdCount++;
 113 
 114     # Filter compound data before calculating fingerprints...
 115     if ($OptionsInfo{Filter}) {
 116       if (CheckAndFilterCompound($CmpdCount, $Molecule)) {
 117         $IgnoredCmpdCount++;
 118         next COMPOUND;
 119       }
 120     }
 121 
 122     $AtomTypes = AssignAtomTypes($Molecule);
 123     if (!$AtomTypes) {
 124       $IgnoredCmpdCount++;
 125       ProcessIgnoredCompound('AtomTypesAssignmentFailed', $CmpdCount, $Molecule);
 126       next COMPOUND;
 127     }
 128 
 129   }
 130   $MoleculeFileIO->Close();
 131 
 132   WriteAtomTypesAssignmentSummaryStatistics($CmpdCount, $IgnoredCmpdCount);
 133 }

 135 # Process compound being ignored due to problems in fingerprints geneation...
 136 #
 137 sub ProcessIgnoredCompound {
 138   my($Mode, $CmpdCount, $Molecule) = @_;
 139   my($CmpdID, $DataFieldLabelAndValuesRef);
 140 
 141   $DataFieldLabelAndValuesRef = $Molecule->GetDataFieldLabelAndValues();
 142   $CmpdID = SetupCmpdIDForOutputFiles($CmpdCount, $Molecule, $DataFieldLabelAndValuesRef);
 143 
 144   MODE: {
 145     if ($Mode =~ /^ContainsNonElementalData$/i) {
 146       warn "\nWarning: Ignoring compound record number $CmpdCount with ID $CmpdID: Compound contains atom data corresponding to non-elemental atom symbol(s)...\n\n";
 147       next MODE;
 148     }
 149 
 150     if ($Mode =~ /^ContainsNoElementalData$/i) {
 151       warn "\nWarning: Ignoring compound record number $CmpdCount with ID $CmpdID: Compound contains no atom data...\n\n";
 152       next MODE;
 153     }
 154 
 155     if ($Mode =~ /^FingerprintsGenerationFailed$/i) {
 156       warn "\nWarning: Ignoring compound record number $CmpdCount with ID $CmpdID: Fingerprints generation didn't succeed...\n\n";
 157       next MODE;
 158     }
 159     warn "\nWarning: Ignoring compound record number $CmpdCount with ID $CmpdID: Fingerprints generation didn't succeed...\n\n";
 160   }
 161 }

 163 # Check and filter compounds....
 164 #
 165 sub CheckAndFilterCompound {
 166   my($CmpdCount, $Molecule) = @_;
 167   my($ElementCount, $NonElementCount);
 168 
 169   ($ElementCount, $NonElementCount) = $Molecule->GetNumOfElementsAndNonElements();
 170 
 171   if ($NonElementCount) {
 172     ProcessIgnoredCompound('ContainsNonElementalData', $CmpdCount, $Molecule);
 173     return 1;
 174   }
 175 
 176   if (!$ElementCount) {
 177     ProcessIgnoredCompound('ContainsNoElementalData', $CmpdCount, $Molecule);
 178     return 1;
 179   }
 180 
 181   return 0;
 182 }

 235 # Retrieve information about SD files...
 236 #
 237 sub RetrieveSDFilesInfo {
 238   my($SDFile, $Index, $FileDir, $FileExt, $FileName, $OutFileRoot, $TextOutFileExt, $SDOutFileExt, $FPOutFileExt, $NewSDFileName, $NewFPFileName, $NewTextFileName, $CheckDataField, $CollectDataFields, $AllDataFieldsRef, $CommonDataFieldsRef);
 239 
 240   %SDFilesInfo = ();
 241   @{$SDFilesInfo{FileOkay}} = ();
 242   @{$SDFilesInfo{OutFileRoot}} = ();
 243   @{$SDFilesInfo{SDOutFileNames}} = ();
 244   @{$SDFilesInfo{FPOutFileNames}} = ();
 245   @{$SDFilesInfo{TextOutFileNames}} = ();
 246   @{$SDFilesInfo{AllDataFieldsRef}} = ();
 247   @{$SDFilesInfo{CommonDataFieldsRef}} = ();
 248 
 249   $CheckDataField = ($OptionsInfo{TextOutput} && ($OptionsInfo{DataFieldsMode} =~ /^CompoundID$/i) && ($OptionsInfo{CompoundIDMode} =~ /^DataField$/i)) ? 1 : 0;
 250   $CollectDataFields = ($OptionsInfo{TextOutput} && ($OptionsInfo{DataFieldsMode} =~ /^(All|Common)$/i)) ? 1 : 0;
 251 
 252   FILELIST: for $Index (0 .. $#SDFilesList) {
 253     $SDFile = $SDFilesList[$Index];
 254 
 255     $SDFilesInfo{FileOkay}[$Index] = 0;
 256     $SDFilesInfo{OutFileRoot}[$Index] = '';
 257     $SDFilesInfo{SDOutFileNames}[$Index] = '';
 258     $SDFilesInfo{FPOutFileNames}[$Index] = '';
 259     $SDFilesInfo{TextOutFileNames}[$Index] = '';
 260 
 261     $SDFile = $SDFilesList[$Index];
 262     if (!(-e $SDFile)) {
 263       warn "Warning: Ignoring file $SDFile: It doesn't exist\n";
 264       next FILELIST;
 265     }
 266     if (!CheckFileType($SDFile, "sd sdf")) {
 267       warn "Warning: Ignoring file $SDFile: It's not a SD file\n";
 268       next FILELIST;
 269     }
 270 
 271     if ($CheckDataField) {
 272       # Make sure data field exists in SD file..
 273       my($CmpdString, $SpecifiedDataField, @CmpdLines, %DataFieldValues);
 274 
 275       @CmpdLines = ();
 276       open SDFILE, "$SDFile" or die "Error: Couldn't open $SDFile: $! \n";
 277       $CmpdString = ReadCmpdString(\*SDFILE);
 278       close SDFILE;
 279       @CmpdLines = split "\n", $CmpdString;
 280       %DataFieldValues = GetCmpdDataHeaderLabelsAndValues(\@CmpdLines);
 281       $SpecifiedDataField = $OptionsInfo{CompoundID};
 282       if (!exists $DataFieldValues{$SpecifiedDataField}) {
 283         warn "Warning: Ignoring file $SDFile: Data field value, $SpecifiedDataField, using  \"--CompoundID\" option in \"DataField\" \"--CompoundIDMode\" doesn't exist\n";
 284         next FILELIST;
 285       }
 286     }
 287 
 288     $AllDataFieldsRef = '';
 289     $CommonDataFieldsRef = '';
 290     if ($CollectDataFields) {
 291       my($CmpdCount);
 292       open SDFILE, "$SDFile" or die "Error: Couldn't open $SDFile: $! \n";
 293       ($CmpdCount, $AllDataFieldsRef, $CommonDataFieldsRef) = GetAllAndCommonCmpdDataHeaderLabels(\*SDFILE);
 294       close SDFILE;
 295     }
 296 
 297     # Setup output file names...
 298     $FileDir = ""; $FileName = ""; $FileExt = "";
 299     ($FileDir, $FileName, $FileExt) = ParseFileName($SDFile);
 300 
 301     $TextOutFileExt = "csv";
 302     if ($Options{outdelim} =~ /^tab$/i) {
 303       $TextOutFileExt = "tsv";
 304     }
 305     $SDOutFileExt = $FileExt;
 306     $FPOutFileExt = "fpf";
 307 
 308     if ($OptionsInfo{OutFileRoot} && (@SDFilesList == 1)) {
 309       my ($RootFileDir, $RootFileName, $RootFileExt) = ParseFileName($OptionsInfo{OutFileRoot});
 310       if ($RootFileName && $RootFileExt) {
 311         $FileName = $RootFileName;
 312       }
 313       else {
 314         $FileName = $OptionsInfo{OutFileRoot};
 315       }
 316       $OutFileRoot = $FileName;
 317     }
 318     else {
 319       $OutFileRoot = $FileName . 'AtomTypesFP';
 320     }
 321 
 322     $NewSDFileName = "${OutFileRoot}.${SDOutFileExt}";
 323     $NewFPFileName = "${OutFileRoot}.${FPOutFileExt}";
 324     $NewTextFileName = "${OutFileRoot}.${TextOutFileExt}";
 325 
 326     if ($OptionsInfo{SDOutput}) {
 327       if ($SDFile =~ /$NewSDFileName/i) {
 328         warn "Warning: Ignoring input file $SDFile: Same output, $NewSDFileName, and input file names.\n";
 329         print "Specify a different name using \"-r --root\" option or use default name.\n";
 330         next FILELIST;
 331       }
 332     }
 333 
 334     if (!$OptionsInfo{OverwriteFiles}) {
 335       # Check SD and text outout files...
 336       if ($OptionsInfo{SDOutput}) {
 337         if (-e $NewSDFileName) {
 338           warn "Warning: Ignoring file $SDFile: The file $NewSDFileName already exists\n";
 339           next FILELIST;
 340         }
 341       }
 342       if ($OptionsInfo{FPOutput}) {
 343         if (-e $NewFPFileName) {
 344           warn "Warning: Ignoring file $SDFile: The file $NewFPFileName already exists\n";
 345           next FILELIST;
 346         }
 347       }
 348       if ($OptionsInfo{TextOutput}) {
 349         if (-e $NewTextFileName) {
 350           warn "Warning: Ignoring file $SDFile: The file $NewTextFileName already exists\n";
 351           next FILELIST;
 352         }
 353       }
 354     }
 355 
 356     $SDFilesInfo{FileOkay}[$Index] = 1;
 357 
 358     $SDFilesInfo{OutFileRoot}[$Index] = $OutFileRoot;
 359     $SDFilesInfo{SDOutFileNames}[$Index] = $NewSDFileName;
 360     $SDFilesInfo{FPOutFileNames}[$Index] = $NewFPFileName;
 361     $SDFilesInfo{TextOutFileNames}[$Index] = $NewTextFileName;
 362 
 363     $SDFilesInfo{AllDataFieldsRef}[$Index] = $AllDataFieldsRef;
 364     $SDFilesInfo{CommonDataFieldsRef}[$Index] = $CommonDataFieldsRef;
 365   }
 366 }

 368 # Process option values...
 369 sub ProcessOptions {
 370   %OptionsInfo = ();
 371 
 372   $OptionsInfo{Mode} = $Options{mode};
 373   $OptionsInfo{AromaticityModel} = $Options{aromaticitymodel};
 374 
 375   ProcessAtomIdentifierTypeOptions();
 376 
 377 
 378   $OptionsInfo{CompoundIDMode} = $Options{compoundidmode};
 379   $OptionsInfo{CompoundIDLabel} = $Options{compoundidlabel};
 380   $OptionsInfo{DataFieldsMode} = $Options{datafieldsmode};
 381 
 382   my(@SpecifiedDataFields);
 383   @SpecifiedDataFields = ();
 384 
 385   @{$OptionsInfo{SpecifiedDataFields}} = ();
 386   $OptionsInfo{CompoundID} = '';
 387 
 388   if ($Options{datafieldsmode} =~ /^CompoundID$/i) {
 389     if ($Options{compoundidmode} =~ /^DataField$/i) {
 390       if (!$Options{compoundid}) {
 391         die "Error: You must specify a value for \"--CompoundID\" option in \"DataField\" \"--CompoundIDMode\". \n";
 392       }
 393       $OptionsInfo{CompoundID} = $Options{compoundid};
 394     }
 395     elsif ($Options{compoundidmode} =~ /^(LabelPrefix|MolNameOrLabelPrefix)$/i) {
 396       $OptionsInfo{CompoundID} = $Options{compoundid} ? $Options{compoundid} : 'Cmpd';
 397     }
 398   }
 399   elsif ($Options{datafieldsmode} =~ /^Specify$/i) {
 400     if (!$Options{datafields}) {
 401       die "Error: You must specify a value for \"--DataFields\" option in \"Specify\" \"-d, --DataFieldsMode\". \n";
 402     }
 403     @SpecifiedDataFields = split /\,/, $Options{datafields};
 404     push @{$OptionsInfo{SpecifiedDataFields}}, @SpecifiedDataFields;
 405   }
 406 
 407   $OptionsInfo{IgnoreHydrogens} = ($Options{ignorehydrogens} =~ /^Yes$/i) ? 1 : 0;
 408 
 409   $OptionsInfo{FingerprintsLabel} = $Options{fingerprintslabel} ? $Options{fingerprintslabel} : 'AtomTypesFingerprints';
 410 
 411   $OptionsInfo{Filter} = ($Options{filter} =~ /^Yes$/i) ? 1 : 0;
 412 
 413   if ($Options{fingerprintslabelmode} =~ /^FingerprintsLabelWithIDs$/) {
 414     if ($Options{mode} =~ /^(AtomTypesCount)$/i && $Options{atomtypessettouse} =~ /^FixedSize$/i) {
 415       # Append atom types to the fingerprints label...
 416       my($FixedSizeAtomTypesSetRef);
 417       $FixedSizeAtomTypesSetRef = GetFixedSizeAtomTypesSet();
 418 
 419       $OptionsInfo{FingerprintsLabel} .= "; AtomTypes: " . TextUtil::JoinWords($FixedSizeAtomTypesSetRef, " ", 0);
 420     }
 421   }
 422   $OptionsInfo{FingerprintsLabelMode} = $Options{fingerprintslabelmode};
 423 
 424   $OptionsInfo{KeepLargestComponent} = ($Options{keeplargestcomponent} =~ /^Yes$/i) ? 1 : 0;
 425 
 426   $OptionsInfo{Output} = $Options{output};
 427   $OptionsInfo{SDOutput} = ($Options{output} =~ /^(SD|All)$/i) ? 1 : 0;
 428   $OptionsInfo{FPOutput} = ($Options{output} =~ /^(FP|All)$/i) ? 1 : 0;
 429   $OptionsInfo{TextOutput} = ($Options{output} =~ /^(Text|All)$/i) ? 1 : 0;
 430 
 431   $OptionsInfo{OutDelim} = $Options{outdelim};
 432   $OptionsInfo{OutQuote} = ($Options{quote} =~ /^Yes$/i) ? 1 : 0;
 433 
 434   $OptionsInfo{OverwriteFiles} = $Options{overwrite} ? 1 : 0;
 435   $OptionsInfo{OutFileRoot} = $Options{root} ? $Options{root} : 0;
 436 
 437   # Setup default vector string format...
 438   my($VectorStringFormat);
 439   $VectorStringFormat = '';
 440   if ($Options{vectorstringformat}) {
 441     $VectorStringFormat = $Options{vectorstringformat};
 442   }
 443   else {
 444     $VectorStringFormat = ($Options{atomtypessettouse} =~ /^FixedSize$/) ? "ValuesString" : "IDsAndValuesString";
 445   }
 446   $OptionsInfo{VectorStringFormat} = $VectorStringFormat;
 447 }

 449 # Process atom identifier type and related options...
 450 #
 451 sub ProcessAtomIdentifierTypeOptions {
 452 
 453   $OptionsInfo{AtomIdentifierType} = $Options{atomidentifiertype};
 454 
 455   if ($Options{atomidentifiertype} =~ /^AtomicInvariantsAtomTypes$/i) {

 457   }
 458   elsif ($Options{atomidentifiertype} =~ /^FunctionalClassAtomTypes$/i) {
 459     ProcessFunctionalClassesToUse();
 460   }
 461   elsif ($OptionsInfo{AtomIdentifierType} =~ /^(DREIDINGAtomTypes|EStateAtomTypes|MMFF94AtomTypes|SLogPAtomTypes|SYBYLAtomTypes|TPSAAtomTypes|UFFAtomTypes)$/i) {
 462     # Nothing to do for now...
 463   }
 464   else {
 465     die "Error: The value specified, $Options{atomidentifiertype}, for option \"-a, --AtomIdentifierType\" is not valid. Supported atom identifier types in current release of MayaChemTools: AtomicInvariantsAtomTypes, DREIDINGAtomTypes, EStateAtomTypes, FunctionalClassAtomTypes, MMFF94AtomTypes, SLogPAtomTypes, SYBYLAtomTypes, TPSAAtomTypes, UFFAtomTypes\n";
 466   }
 467 }

 469 # Process specified atomic invariants to use...
 470 #
 471 sub ProcessAtomicInvariantsToUseOption {
 472   my($AtomicInvariant, $AtomSymbolSpecified, @AtomicInvariantsWords);
 473 
 474   @{$OptionsInfo{AtomicInvariantsToUse}} = ();
 475   if (IsEmpty($Options{atomicinvariantstouse})) {
 476     die "Error: Atomic invariants value specified using \"--AtomicInvariantsToUse\" option is empty\n";
 477   }
 478   $AtomSymbolSpecified = 0;
 479   @AtomicInvariantsWords = split /\,/, $Options{atomicinvariantstouse};
 480   for $AtomicInvariant (@AtomicInvariantsWords) {
 481     if (!AtomTypes::AtomicInvariantsAtomTypes::IsAtomicInvariantAvailable($AtomicInvariant)) {
 482       die "Error: Atomic invariant specified, $AtomicInvariant, using \"--AtomicInvariantsToUse\" option is not valid...\n ";
 483     }
 484     if ($AtomicInvariant =~ /^(AS|AtomSymbol)$/i) {
 485       $AtomSymbolSpecified = 1;
 486     }
 487     push @{$OptionsInfo{AtomicInvariantsToUse}}, $AtomicInvariant;
 488   }
 489   if (!$AtomSymbolSpecified) {
 490     die "Error: Atomic invariant, AS or AtomSymbol, must be specified as using \"--AtomicInvariantsToUse\" option...\n ";
 491   }
 492 }
