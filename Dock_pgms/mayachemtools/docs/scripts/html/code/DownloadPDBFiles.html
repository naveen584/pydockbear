  72 # Download appropriate PDB fies...
  73 sub DownloadPDBFiles {
  74   my($PDBFilesLocationURL, $PDBIDCount, $PDBIDOkayCount, $PDBIDFailedCount, $PDBIDsIgnoredCount, $PDBID, $PDBRemoteFileURL, $PDBLocalFile, $DownloadEDSMap, $EDSMapFilesLocaltionURL, $EDSMapRemoteFileURL, $EDSMapLocalFile, $Status, $EDSMapOkayCount, $EDSMapFailedCount, $EDSMapPDBID, %PDBIDsMap);
  75 
  76   print "\nDownloading PDB files...\n";
  77 
  78   $PDBFilesLocationURL = $OptionsInfo{DataLocationURL};
  79 
  80   $DownloadEDSMap = $OptionsInfo{EDSMap};
  81   $EDSMapFilesLocaltionURL = $OptionsInfo{EDSMapLocationURL};
  82 
  83   ($PDBIDCount, $PDBIDOkayCount, $PDBIDFailedCount, $PDBIDsIgnoredCount, $EDSMapOkayCount, $EDSMapFailedCount) = (0) x 6;
  84 
  85   # Turn off warnings from File::Fetch
  86   $File::Fetch::WARN = 0;
  87 
  88   %PDBIDsMap = ();
  89 
  90   PDBID: for $PDBID (@PDBIDs) {
  91     $PDBIDCount++;
  92 
  93     print "\nProcessing PDB ID $PDBID...\n";
  94 
  95     if (exists $PDBIDsMap{$PDBID}) {
  96       $PDBIDsIgnoredCount++;
  97       warn "Warning: Ignoring duplicate PDB ID $PDBID\n";
  98       next PDBID;
  99     }
 100     else {
 101       $PDBIDsMap{$PDBID} = $PDBID;
 102     }
 103 
 104     if ($PDBID =~ /\./) {
 105       $PDBIDsIgnoredCount++;
 106       warn "Warning: Ignoring invalid PDB ID $PDBID\n";
 107       next PDBID;
 108     }
 109 
 110     # Download PDB file...
 111     $PDBRemoteFileURL = "${PDBFilesLocationURL}${PDBID}.pdb";
 112     $PDBLocalFile = "${PDBID}.pdb";
 113 
 114     print "Downloading PDB file: $PDBRemoteFileURL\n";
 115     $Status = DownloadFile($PDBRemoteFileURL, $PDBLocalFile);
 116 
 117     if (!$Status) {
 118       $PDBIDFailedCount++;
 119       next PDBID;
 120     }
 121     $PDBIDOkayCount++;
 122 
 123     # Download EDS map file...
 124     if (!$DownloadEDSMap) {
 125       next PDBID;
 126     }
 127 
 128     $EDSMapPDBID = lc $PDBID;
 129     $EDSMapRemoteFileURL = "${EDSMapFilesLocaltionURL}${EDSMapPDBID}.ccp4";
 130     $EDSMapLocalFile = "${EDSMapPDBID}.ccp4";
 131 
 132     print "Downloading PDB EDS map file: $EDSMapRemoteFileURL\n";
 133     $Status = DownloadFile($EDSMapRemoteFileURL, $EDSMapLocalFile );
 134 
 135     if (!$Status) {
 136       $EDSMapFailedCount++;
 137       next PDBID;
 138     }
 139     print "Moving file from ${EDSMapPDBID}.ccp4 to ${PDBID}.ccp4\n";
 140     move "${EDSMapPDBID}.ccp4", "${EDSMapPDBID}Tmp.ccp4" or "Warning: Couldn't move file ${EDSMapPDBID}.ccp4 to ${EDSMapPDBID}Tmp.ccp4\n";
 141     move "${EDSMapPDBID}Tmp.ccp4", "${PDBID}.ccp4" or "Warning: Couldn't move file ${EDSMapPDBID}Tmp.ccp4 to ${PDBID}.ccp4\n";
 142     $EDSMapOkayCount++;
 143 
 144   }
 145 
 146   print "\nTotal number of PDB IDs:  $PDBIDCount\n";
 147   print "Number of duplicate PDB IDs ignored:  $PDBIDsIgnoredCount\n";
 148 
 149   print "\nNumber of successful downloads:  $PDBIDOkayCount\n";
 150   print "Number of failed downloads:  $PDBIDFailedCount\n";
 151 
 152   if ($DownloadEDSMap) {
 153     print "\nNumber of successful EDS map downloads:  $EDSMapOkayCount\n";
 154     print "Number of failed EDS map downloads:  $EDSMapFailedCount\n";
 155   }
 156 }

 158 # Download specified file...
 159 sub DownloadFile {
 160   my($RemoteFileURL, $LocalFileName) = @_;
 161   my($Status, $FileFetch, $FetchedFilePath);
 162 
 163   $Status = 1;
 164 
 165   # Setup a fetch object...
 166   $FileFetch = File::Fetch->new(uri => $RemoteFileURL);
 167 
 168   # Fetch file to the CWD...
 169   $FetchedFilePath = $FileFetch->fetch();
 170 
 171   if (IsEmpty($FetchedFilePath)) {
 172     warn "Warning: Download failed for file $RemoteFileURL: " . $FileFetch->error() . "\n";
 173     if (-e $LocalFileName) {
 174       warn "Warning: Deleting empty file $LocalFileName\n";
 175       unlink $LocalFileName or warn "Warning: Couldn't delete file $LocalFileName\n";
 176     }
 177     $Status = 0;
 178   }
 179   return $Status;
 180 }

 182 # Collect PDB IDs...
 183 #
 184 sub RetrievePDBIDs {
 185   @PDBIDs = ();
 186 
 187   if ($OptionsInfo{Mode} =~ /^IDsOnCmdLine$/i) {
 188     RetriveCommandLinePDBIDs();
 189   }
 190   elsif ($OptionsInfo{Mode} =~ /^IDsInFile$/i) {
 191     RetriveTextFilePDBIDs();
 192   }
 193 }

 195 # Collect PDB IDs specified on the command line...
 196 #
 197 sub RetriveCommandLinePDBIDs {
 198   my($SpecifiedPDBID, @ProcessedPDBIDs );
 199 
 200   print "\nProcessing PDB ID(s) from command line...\n";
 201 
 202   for $SpecifiedPDBID (@{$OptionsInfo{CmdLinePDBIDs}}) {
 203     @ProcessedPDBIDs = ProcessSpecifiedPDBIDs($SpecifiedPDBID);
 204     if (@ProcessedPDBIDs) {
 205       push @PDBIDs, @ProcessedPDBIDs;
 206     }
 207   }
 208 }

 210 # Collect PDB IDs specified in the text file...
 211 #
 212 sub RetriveTextFilePDBIDs {
 213   my($TextFile, $InDelim, $IDsColIndex, $LineCount, $ProcessedLineCount, $IgnoredLineCount, $PDBID, $Line, @ProcessedPDBIDs, @LineWords);
 214 
 215   $TextFile = $PDBIDsFileInfo{Name};
 216 
 217   $IDsColIndex = $PDBIDsFileInfo{IDsColIndex};
 218   $InDelim = $PDBIDsFileInfo{InDelim} ;
 219 
 220   ($LineCount, $ProcessedLineCount, $IgnoredLineCount) = (0) x 3;
 221 
 222   print "\nProcessing PDB ID(s) from PDB IDs  file $TextFile...\n";
 223 
 224   open TEXTFILE, "$TextFile" or die "Couldn't open $TextFile: $! \n";
 225   # Skip label line...
 226   $_ = <TEXTFILE>;
 227 
 228   LINE: while ($Line = GetTextLine(\*TEXTFILE)) {
 229     $LineCount++;
 230     @LineWords = quotewords($InDelim, 0, $Line);
 231 
 232     if ($IDsColIndex >= scalar @LineWords) {
 233       $IgnoredLineCount++;
 234       warn "Warning: Ignoring line number $LineCount: PDB IDs column number, ". ($IDsColIndex + 1) . ", doesn't exist in the line containing, " . (scalar @LineWords) .  ", columns.\nLine: $Line\n";
 235       next LINE;
 236     }
 237     $PDBID = $LineWords[$IDsColIndex];
 238     if (IsEmpty($PDBID )) {
 239       $IgnoredLineCount++;
 240       warn "Warning: Ignoring line number $LineCount: PDB ID value is empty.\nLine: $Line\n";
 241       next LINE;
 242     }
 243     $ProcessedLineCount++;
 244 
 245     @ProcessedPDBIDs = ProcessSpecifiedPDBIDs($PDBID);
 246     if (@ProcessedPDBIDs) {
 247       push @PDBIDs, @ProcessedPDBIDs;
 248     }
 249   }
 250   close TEXTFILE;
 251 
 252   print "\nTotal number of lines in PDB IDs text file: $LineCount\n";
 253   print "Total number of lines processed: $ProcessedLineCount\n";
 254   print "Total number of lines ignored: $IgnoredLineCount\n";
 255 }

 257 # Process specified PDB IDs...
 258 #
 259 # Notes:
 260 #   . Commas and spaces in the specification of PBD IDs are allowed.
 261 #   . All PDB IDs are turned into uppercase letters.
 262 #
 263 sub ProcessSpecifiedPDBIDs {
 264   my($SpecifiedPDBID) = @_;
 265   my($PDBID, @PDBIDWords, @PDBIDs);
 266 
 267   $SpecifiedPDBID = RemoveLeadingAndTrailingWhiteSpaces($SpecifiedPDBID);
 268    if ($SpecifiedPDBID =~ / /) {
 269     @PDBIDWords = split " ",  $SpecifiedPDBID;
 270   }
 271   elsif ($SpecifiedPDBID =~ /,/) {
 272     @PDBIDWords = split ",",  $SpecifiedPDBID;
 273   }
 274   else {
 275     push @PDBIDWords, $SpecifiedPDBID;
 276   }
 277 
 278   @PDBIDs = ();
 279   for $PDBID (@PDBIDWords) {
 280     $PDBID =~ s/( |,)//g;
 281     push @PDBIDs, uc $PDBID;
 282   }
 283   return @PDBIDs;
 284 }

 286 # Process option values...
 287 sub ProcessOptions {
 288 
 289   %OptionsInfo = ();
 290   %PDBIDsFileInfo = ();
 291 
 292   $OptionsInfo{Mode} = $Options{mode};
 293 
 294   $OptionsInfo{DataLocationURL} = $Options{datalocationurl};
 295   if (IsEmpty($OptionsInfo{DataLocationURL} )) {
 296     die "Error: PDB data location URL specified using \"-d, --dataLocationURL\" is empty. Allowed value: Non empty string\n";
 297   }
 298 
 299   $OptionsInfo{EDSMap} = $Options{edsmap} =~ /^Yes$/i ? 1 : 0;
 300   $OptionsInfo{EDSMapLocationURL} = $Options{edsmaplocationurl};
 301 
 302   $OptionsInfo{ColMode} = $Options{colmode};
 303   $OptionsInfo{InDelim} = $Options{indelim};
 304 
 305   $OptionsInfo{PDBIDsCol } = defined $Options{pdbidscol} ? $Options{pdbidscol} : '';
 306 
 307   @{$OptionsInfo{CmdLinePDBIDs}} = ();
 308   $OptionsInfo{PDBIDsFile} = "";
 309 
 310   if ($OptionsInfo{Mode} =~ /^IDsOnCmdLine$/i) {
 311     push @{$OptionsInfo{CmdLinePDBIDs}}, @ARGV;
 312   }
 313   elsif ($OptionsInfo{Mode} =~ /^IDsInFile$/i) {
 314     if (@ARGV != 1) {
 315       die "Error: Invalid number of PDB IDs text files, ". (scalar @ARGV) . ",specified on the command line for \"IDsInFile\" value of  $Options{mode}, for option \"-m --mode\". Allowed value: Only one text file\n";
 316     }
 317     $OptionsInfo{PDBIDsFile} = $ARGV[0];
 318 
 319     RetrievePDBIDsTextFileInfo();
 320   }
 321   else {
 322     die "Error: The value specified, $Options{mode}, for option \"-m --mode\" is not valid. Allowed values: IDsOnCmdLine or IDsInFile\n";
 323   }
 324 }

 326 # Retrieve information for PDB IDs text file...
 327 #
 328 sub RetrievePDBIDsTextFileInfo {
 329   my($TextFile, $FileDir, $FileName, $FileExt, $InDelim, $Line, $ColNum, $ColLabel, $PDBIDsColIndex, $ColMode, $PDBIDsCol , $ColCount, $ColIndex, @ColLabels);
 330 
 331   $TextFile = $OptionsInfo{PDBIDsFile};
 332 
 333   %PDBIDsFileInfo = ();
 334   $PDBIDsFileInfo{Name} = $TextFile;
 335 
 336   $PDBIDsFileInfo{ColCount} = 0;
 337   @{$PDBIDsFileInfo{ColLabels}} = ();
 338   %{$PDBIDsFileInfo{ColLabelToNumMap}} = ();
 339   $PDBIDsFileInfo{InDelim} = "";
 340 
 341   $PDBIDsFileInfo{IDsColIndex} = "";
 342 
 343   if (!-e $TextFile) {
 344     die "Error: PDBIDs text file, $TextFile, doesn't exist\n";
 345   }
 346 
 347   if (!CheckFileType($TextFile, "csv tsv")) {
 348     die "Error: Ignoring file $TextFile: It's not a csv or tsv file\n";
 349   }
 350 
 351   ($FileDir, $FileName, $FileExt) = ParseFileName($TextFile);
 352   if ($FileExt =~ /^tsv$/i) {
 353     $InDelim = "\t";
 354   }
 355   else {
 356     $InDelim = "\,";
 357     if (!($OptionsInfo{InDelim} =~ /^(comma|semicolon)$/i)) {
 358       die "Error: Ignoring file $TextFile: The value specified, $OptionsInfo{InDelim}, for option \"--indelim\" is not valid for textfile\n";
 359     }
 360     if ($OptionsInfo{InDelim} =~ /^semicolon$/i) {
 361       $InDelim = "\;";
 362     }
 363   }
 364 
 365   if (!open TEXTFILE, "$TextFile") {
 366     die "Error: Ignoring file $TextFile: Couldn't open it: $! \n";
 367   }
 368 
 369   $Line = GetTextLine(\*TEXTFILE);
 370   @ColLabels = quotewords($InDelim, 0, $Line);
 371   close TEXTFILE;
 372   $ColCount = scalar @ColLabels;
 373 
 374   push @{$PDBIDsFileInfo{ColLabels}}, @ColLabels;
 375   $PDBIDsFileInfo{ColCount} = $ColCount ;
 376   $PDBIDsFileInfo{InDelim} = $InDelim;
 377 
 378   # Setup collabel to colnum map...
 379   %{$PDBIDsFileInfo{ColLabelToNumMap}} = ();
 380   for $ColNum (0 .. $#ColLabels) {
 381     $ColLabel = $ColLabels[$ColNum];
 382     $PDBIDsFileInfo{ColLabelToNumMap}{lc $ColLabel} = $ColNum;
 383   }
 384 
 385   # Identify column containing PDB IDs...
 386   $PDBIDsColIndex = "";
 387 
 388   $ColMode = $OptionsInfo{ColMode};
 389   $PDBIDsCol = $OptionsInfo{PDBIDsCol };
 390 
 391   if (IsNotEmpty($PDBIDsCol )) {
 392     if ($ColMode =~ /^collabel$/i) {
 393       $ColLabel = lc $PDBIDsCol;
 394       if (!exists $PDBIDsFileInfo{ColLabelToNumMap}{$ColLabel} ) {
 395         die "Error: Ignoring file $TextFile: The column name, $PDBIDsCol, specified for option \"-p, --PDBIDsCol \" is not valid for text file\n";
 396       }
 397       $PDBIDsColIndex = $PDBIDsFileInfo{ColLabelToNumMap}{$ColLabel};
 398     }
 399     else {
 400       $ColNum = $PDBIDsCol;
 401       if ($ColNum <= 0 || $ColNum > $ColCount) {
 402         die "Error: Ignoring file $TextFile: The column number, $PDBIDsCol, specified for option \"-p, --PDBIDsCol \" is not valid for text file. It must be > 0 and <= $ColCount\n";
 403       }
 404       $PDBIDsColIndex = $ColNum - 1;
 405     }
 406   }
 407   else {
 408     # Look for column name containing PDB_ID or PDBID text string...
 409     $PDBIDsCol = "";
 410     $ColIndex = 0;
 411     COLLABEL: for $ColLabel (@ColLabels) {
 412       if ($ColLabel =~ /(PDB_ID|PDBID)/i) {
 413         $PDBIDsCol = $ColLabel;
 414         $PDBIDsColIndex = $ColIndex;
 415         last COLLABEL;
 416       }
 417       $ColIndex++;
 418     }
 419     if (IsEmpty($PDBIDsCol)) {
 420       die "Error: Ignoring file $TextFile: Couldn't find PDB IDs default column containing text string PDB_ID or PDBID in its name\n";
 421     }
 422   }
 423   $PDBIDsFileInfo{IDsColIndex} = $PDBIDsColIndex;
 424 }

 426 # Setup script usage  and retrieve command line arguments specified using various options...
 427 sub SetupScriptUsage {
 428 
 429   # Retrieve all the options...
 430   %Options = ();
 431 
 432   $Options{colmode} = "colnum";
 433   $Options{datalocationurl} = "http://www.rcsb.org/pdb/files/";
 434   $Options{edsmap} = "no";
 435   $Options{edsmaplocationurl} = "http://www.ebi.ac.uk/pdbe/coordinates/files/";
 436   $Options{indelim} = "comma";
 437   $Options{mode} = "IDsOnCmdLine";
 438 
 439   if (!GetOptions(\%Options, "colmode|c=s", "datalocationurl|d=s", "edsmap|e=s", "edsmaplocationurl=s", "help|h",  "indelim=s", "mode|m=s", "pdbidscol|s=s", "workingdir|w=s")) {
 440     die "\nTo get a list of valid options and their values, use \"$ScriptName -h\" or\n\"perl -S $ScriptName -h\" command and try again...\n";
 441   }
 442   if ($Options{workingdir}) {
 443     if (! -d $Options{workingdir}) {
 444       die "Error: The value specified, $Options{workingdir}, for option \"-w --workingdir\" is not a directory name.\n";
 445     }
 446     chdir $Options{workingdir} or die "Error: Couldn't chdir $Options{workingdir}: $! \n";
 447   }
 448   if ($Options{colmode} !~ /^(colnum|collabel)$/i) {
 449     die "Error: The value specified, $Options{colmode}, for option \"-c, --colmode\" is not valid. Allowed values: colnum or collabel\n";
 450   }
 451   if ($Options{edsmap} !~ /^(yes|no)$/i) {
 452     die "Error: The value specified, $Options{edsmap}, for option \"-e --edsmap\" is not valid. Allowed values: yes or no\n";
 453   }
 454   if ($Options{indelim} !~ /^(comma|semicolon)$/i) {
 455     die "Error: The value specified, $Options{indelim}, for option \"--indelim\" is not valid. Allowed values: comma or semicolon\n";
 456   }
 457   if ($Options{mode} !~ /^(IDsOnCmdLine|IDsInFile)$/i) {
 458     die "Error: The value specified, $Options{mode}, for option \"-m --mode\" is not valid. Allowed values: IDsOnCmdLine or IDsInFile\n";
 459   }
 460 }
