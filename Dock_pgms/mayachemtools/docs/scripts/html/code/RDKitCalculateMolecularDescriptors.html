  33     RDKitCalculateMolecularDescriptors.py [--autocorr2DExclude <yes or no>] [--fragmentCount <yes or no>]

 117     --outfileParams <Name,Value,...>  [default: auto]
 118         A comma delimited list of parameter name and value pairs for writing

 121             
 122             SD: compute2DCoords,auto,kekulize,no
 123             
 124         Default value for compute2DCoords: yes for SMILES input file; no for all other

 271 def PerformCalculations():
 272     """Calculate descriptors for a specified list of descriptors."""
 273 
 274     Infile = OptionsInfo["Infile"]
 275     Outfile = OptionsInfo["Outfile"]
 276     
 277     DescriptorNames = OptionsInfo["SpecifiedDescriptorNames"]
 278     DescriptorCount = len(DescriptorNames)
 279     
 280     TextOutFileMode = OptionsInfo["TextOutFileMode"]
 281     TextOutFileDelim = OptionsInfo["TextOutFileDelim"]
 282 
 283     Precision = OptionsInfo["Precision"]
 284     
 285     MiscUtil.PrintInfo("Calculating %d molecular descriptor(s) for each molecule..." % DescriptorCount)
 286     
 287     # Setup a molecule reader...
 288     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 289     Mols = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 290     
 291     # Setup a writer...
 292     Compute2DCoords = False
 293     if TextOutFileMode:
 294         Writer = open(Outfile, "w")
 295     else:
 296         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 297         Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 298         
 299     if Writer is None:
 300         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 301 
 302     MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 303 
 304     # Wite out headers for a text file...
 305     if TextOutFileMode:
 306         LineWords = ["MolID"]
 307         LineWords.extend(DescriptorNames)
 308         Line = TextOutFileDelim.join(LineWords)
 309         Writer.write("%s\n" % Line)
 310         
 311     # Process molecules...
 312     MolCount = 0
 313     ValidMolCount = 0
 314     MatchCount = 0
 315 
 316     for Mol in Mols:
 317         MolCount += 1
 318         if Mol is None:
 319             continue
 320         
 321         if RDKitUtil.IsMolEmpty(Mol):
 322             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 323             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 324             continue
 325         
 326         ValidMolCount += 1
 327 
 328         # Calculate descriptors...
 329         CalculatedValues = []
 330         for Index in range(0, DescriptorCount):
 331             Name = DescriptorNames[Index]
 332             ComputeFunction = DescriptorNamesMap["ComputeFunction"][Name]
 333             Value = FormatCalculatedValue(ComputeFunction(Mol), Precision)
 334             CalculatedValues.append(Value)
 335 
 336         # Write out calculated values...
 337         if TextOutFileMode:
 338             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 339             LineWords = [MolName]
 340             LineWords.extend(CalculatedValues)
 341             Line = TextOutFileDelim.join(LineWords)
 342             Writer.write("%s\n" % Line)
 343         else:
 344             for Index in range(0, DescriptorCount):
 345                 Name = DescriptorNames[Index]
 346                 Value = CalculatedValues[Index]
 347                 Mol.SetProp(Name, Value)
 348                 
 349             if Compute2DCoords:
 350                 AllChem.Compute2DCoords(Mol)
 351                 
 352             Writer.write(Mol)
 353     
 354     if Writer is not None:
 355         Writer.close()
 356         
 357     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 358     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 359     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))

 361 def FormatCalculatedValue(Value, Precision):
 362     """Format calculated value of descriptor based on its type."""
 363 
 364     if (type(Value) is float) or (type(Value) is numpy.float64):
 365         FormattedValue = "%.*f" % (Precision, Value)
 366         if not re.search("[1-9]", FormattedValue):
 367             FormattedValue = "0.0"
 368     elif type(Value) is list:
 369         FormattedValue = "%s" % Value
 370         FormattedValue = re.sub('\[|\]|,', '', FormattedValue)
 371     else:
 372         FormattedValue = "%s" % Value
 373 
 374     return FormattedValue

 376 def ProcessSpecifiedDescriptorNames():
 377     """Process and validate specified decriptor names."""
 378 
 379     OptionsInfo["SpecifiedDescriptorNames"] = []
 380 
 381     if not re.match("^(2D|3D|All|FragmentCountOnly|Specify)$", OptionsInfo["Mode"], re.I):
 382         MiscUtil.PrintError("Mode value, %s, using \"-m, --mode\" option is not a valid value." % OptionsInfo["Mode"])
 383     
 384     if re.match("^2D$", OptionsInfo["Mode"], re.I):
 385         OptionsInfo["SpecifiedDescriptorNames"] = DescriptorNamesMap["2D"]["Names"]
 386         if OptionsInfo["FragmentCount"]:
 387             OptionsInfo["SpecifiedDescriptorNames"].extend(DescriptorNamesMap["FragmentCount"]["Names"])
 388         return
 389     elif re.match("^3D$", OptionsInfo["Mode"], re.I):
 390         OptionsInfo["SpecifiedDescriptorNames"] = DescriptorNamesMap["3D"]["Names"]
 391         return
 392     elif re.match("^All$", OptionsInfo["Mode"], re.I):
 393         OptionsInfo["SpecifiedDescriptorNames"] = DescriptorNamesMap["2D"]["Names"]
 394         if OptionsInfo["FragmentCount"]:
 395             OptionsInfo["SpecifiedDescriptorNames"].extend(DescriptorNamesMap["FragmentCount"]["Names"])
 396         OptionsInfo["SpecifiedDescriptorNames"].extend(DescriptorNamesMap["3D"]["Names"])
 397         return
 398     elif re.match("^FragmentCountOnly$", OptionsInfo["Mode"], re.I):
 399         OptionsInfo["SpecifiedDescriptorNames"] = DescriptorNamesMap["FragmentCount"]["Names"]
 400         return
 401 
 402     # Set up a canonical descriptor names map for checking specified names...
 403     CanonicalNameMap = {}
 404     for Name in  DescriptorNamesMap["ComputeFunction"]:
 405         CanonicalNameMap[Name.lower()] = Name
 406     
 407     # Parse and validate specified names...
 408     DescriptorNames = re.sub(" ", "", OptionsInfo["DescriptorNames"])
 409     if not DescriptorNames:
 410         MiscUtil.PrintError("No descriptor names specified for \"-d, --descriptorNames\" option")
 411 
 412     SMILESInfile = MiscUtil.CheckFileExt(Options["--infile"], "smi")
 413     Canonical3DNameMap = {}
 414     if SMILESInfile:
 415         for Name in DescriptorNamesMap["3D"]["Names"]:
 416             Canonical3DNameMap[Name.lower()] = Name
 417             
 418     SpecifiedDescriptorNames = []
 419     for Name in DescriptorNames.split(","):
 420         CanonicalName = Name.lower()
 421         if CanonicalName in CanonicalNameMap:
 422             SpecifiedDescriptorNames.append(CanonicalNameMap[CanonicalName])
 423         else:
 424             MiscUtil.PrintError("The descriptor name, %s, specified using \"-d, --descriptorNames\" option is not a valid name." % (Name))
 425         if SMILESInfile:
 426             if CanonicalName in Canonical3DNameMap:
 427                 MiscUtil.PrintError("The 3D descriptor name, %s, specified using \"-d, --descriptorNames\" option is not a valid for SMILES input file." % (Name))
 428                 
 429     if not len(SpecifiedDescriptorNames):
 430         MiscUtil.PrintError("No valid descriptor name specified for \"-d, --descriptorNames\" option")
 431     
 432     OptionsInfo["SpecifiedDescriptorNames"] = SpecifiedDescriptorNames
 433 
 434 def RetrieveMolecularDescriptorsInfo():
 435     """Retrieve descriptors information."""
 436 
 437     MiscUtil.PrintInfo("\nRetrieving information for avalible molecular descriptors...")
 438     
 439     # Initialze data for 2D, FragmentCount and 3D descriptors...
 440     DescriptorNamesMap["Types"] = ["2D", "FragmentCount", "3D"]
 441     DescriptorNamesMap["ComputeFunction"] = {}
 442 
 443     Autocorr2DExclude = OptionsInfo["Autocorr2DExclude"]
 444     
 445     for Type in DescriptorNamesMap["Types"]:
 446         DescriptorNamesMap[Type] = {}
 447         DescriptorNamesMap[Type]["Names"] = []
 448     
 449     # Setup data for 2D and FragmentCount...
 450     DescriptorsInfo = Descriptors.descList
 451     for DescriptorInfo in DescriptorsInfo:
 452         Name = DescriptorInfo[0]
 453         ComputeFunction = DescriptorInfo[1]
 454 
 455         Type = "2D"
 456         if re.match("^fr_", Name, re.I):
 457             Type = "FragmentCount"
 458         elif re.match("^Autocorr2D$", Name, re.I):
 459             if Autocorr2DExclude:
 460                 continue
 461         
 462         if Name in DescriptorNamesMap["ComputeFunction"]:
 463             MiscUtil.PrintWarning("Ignoring duplicate descriptor name: %s..." % Name)
 464         else:
 465             DescriptorNamesMap[Type]["Names"].append(Name)
 466             DescriptorNamesMap["ComputeFunction"][Name] = ComputeFunction
 467 
 468     # Add new 2D decriptor name to the list...
 469     Type = "2D"
 470     if not Autocorr2DExclude:
 471         try:
 472             Name = "Autocorr2D"
 473             ComputeFunction =  rdMolDescriptors.CalcAUTOCORR2D
 474             if not Name in DescriptorNamesMap["ComputeFunction"]:
 475                 DescriptorNamesMap[Type]["Names"].append(Name)
 476                 DescriptorNamesMap["ComputeFunction"][Name] = ComputeFunction
 477         except (AttributeError):
 478             MiscUtil.PrintInfo("2D descriptor, %s, is not available in your current version of RDKit." % Name)
 479         
 480     # Set data for 3D descriptors...
 481     Type = "3D"
 482     NameToComputeFunctionMap = {'PMI1' : Descriptors3D.PMI1, 'PMI2' : Descriptors3D.PMI2, 'PMI3' : Descriptors3D.PMI3, 'NPR1' : Descriptors3D.NPR1, 'NPR2' : Descriptors3D.NPR2, 'RadiusOfGyration' : Descriptors3D.RadiusOfGyration, 'InertialShapeFactor' :  Descriptors3D.InertialShapeFactor, 'Eccentricity' : Descriptors3D.Eccentricity, 'Asphericity' : Descriptors3D.Asphericity, 'SpherocityIndex' : Descriptors3D.SpherocityIndex}
 483     
 484     for Name in NameToComputeFunctionMap:
 485         ComputeFunction = NameToComputeFunctionMap[Name]
 486         if Name in DescriptorNamesMap["ComputeFunction"]:
 487             MiscUtil.PrintWarning("Ignoring duplicate descriptor name: %s..." % Name)
 488         else:
 489             DescriptorNamesMap[Type]["Names"].append(Name)
 490             DescriptorNamesMap["ComputeFunction"][Name] = ComputeFunction
 491 
 492     # Check and add new 3D descriptors not directly available through Descriptors3D module...
 493     Type = "3D"
 494     AvailableName3DMap = {}
 495     for Name in ['Autocorr3D', 'RDF', 'MORSE', 'WHIM', 'GETAWAY']:
 496         ComputeFunction = None
 497         try:
 498             if re.match("^Autocorr3D$", Name, re.I):
 499                 ComputeFunction =  rdMolDescriptors.CalcAUTOCORR3D
 500             elif re.match("^RDF$", Name, re.I):
 501                 ComputeFunction =  rdMolDescriptors.CalcRDF
 502             elif re.match("^MORSE$", Name, re.I):
 503                 ComputeFunction =  rdMolDescriptors.CalcMORSE
 504             elif re.match("^WHIM$", Name, re.I):
 505                 ComputeFunction =  rdMolDescriptors.CalcWHIM
 506             elif re.match("^GETAWAY$", Name, re.I):
 507                 ComputeFunction =  rdMolDescriptors.CalcGETAWAY
 508             else:
 509                 ComputeFunction = None
 510         except (AttributeError):
 511             MiscUtil.PrintWarning("3D descriptor, %s, is not available in your current version of RDKit" % Name)
 512         
 513         if ComputeFunction is not None:
 514             AvailableName3DMap[Name] = ComputeFunction
 515 
 516     for Name in AvailableName3DMap:
 517         ComputeFunction = AvailableName3DMap[Name]
 518         if not Name in DescriptorNamesMap["ComputeFunction"]:
 519             DescriptorNamesMap[Type]["Names"].append(Name)
 520             DescriptorNamesMap["ComputeFunction"][Name] = ComputeFunction
 521 
 522     Count = 0
 523     TypesCount = []
 524     for Type in DescriptorNamesMap["Types"]:
 525         TypeCount = len(DescriptorNamesMap[Type]["Names"])
 526         TypesCount.append(TypeCount)
 527         Count += TypeCount
 528 
 529     if not Count:
 530         MiscUtil.PrintError("Failed to retrieve any  molecular descriptors...")
 531 
 532     # Sort descriptor names...
 533     for Type in DescriptorNamesMap["Types"]:
 534         DescriptorNamesMap[Type]["Names"] = sorted(DescriptorNamesMap[Type]["Names"])
 535     
 536     MiscUtil.PrintInfo("\nTotal number of availble molecular descriptors: %d" % Count)
 537     for Index in range(0, len(DescriptorNamesMap["Types"])):
 538         Type = DescriptorNamesMap["Types"][Index]
 539         TypeCount = TypesCount[Index]
 540         MiscUtil.PrintInfo("Number of %s molecular descriptors: %d" % (Type, TypeCount))
 541

 556 def ProcessOptions():
 557     """Process and validate command line arguments and options"""
 558     
 559     MiscUtil.PrintInfo("Processing options...")
 560     
 561     # Validate options...
 562     ValidateOptions()
 563 
 564     OptionsInfo["Autocorr2DExclude"] = True
 565     if not re.match("^Yes$", Options["--autocorr2DExclude"], re.I):
 566         OptionsInfo["Autocorr2DExclude"] = False
 567     
 568     OptionsInfo["FragmentCount"] = True
 569     if not re.match("^Yes$", Options["--fragmentCount"], re.I):
 570         OptionsInfo["FragmentCount"] = False
 571     
 572     OptionsInfo["DescriptorNames"] = Options["--descriptorNames"]
 573     OptionsInfo["Mode"] = Options["--mode"]
 574     
 575     OptionsInfo["Infile"] = Options["--infile"]
 576     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 577     
 578     OptionsInfo["Outfile"] = Options["--outfile"]
 579     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 580     
 581     OptionsInfo["Overwrite"] = Options["--overwrite"]
 582     
 583     TextOutFileMode = False
 584     TextOutFileDelim = ""
 585     if MiscUtil.CheckFileExt(Options["--outfile"], "csv"):
 586         TextOutFileMode = True
 587         TextOutFileDelim = ","
 588     elif MiscUtil.CheckFileExt(Options["--outfile"], "tsv txt"):
 589         TextOutFileMode = True
 590         TextOutFileDelim = "\t"
 591     OptionsInfo["TextOutFileMode"] = TextOutFileMode
 592     OptionsInfo["TextOutFileDelim"] = TextOutFileDelim
 593     
 594     OptionsInfo["Precision"] = int(Options["--precision"])
 595 
 596 def RetrieveOptions():
 597     """Retrieve command line arguments and options"""
 598     
 599     # Get options...
 600     global Options
 601     Options = docopt(__doc__)
 602     
 603     # Set current working directory to the specified directory...
 604     WorkingDir = Options["--workingdir"]
 605     if WorkingDir:
 606         os.chdir(WorkingDir)
 607     
 608     # Handle examples option...
 609     if "--examples" in Options and Options["--examples"]:
 610         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 611         sys.exit(0)
 612         
 613     # Handle listing of descriptor information...
 614     if  Options["--list"]:
 615         ProcessListMolecularDescriptorsOption()
 616         sys.exit(0)

 624 def ValidateOptions():
 625     """Validate option values"""
 626 
 627     MiscUtil.ValidateOptionTextValue("-a, --autocorr2DExclude", Options["--autocorr2DExclude"], "yes no")
 628     MiscUtil.ValidateOptionTextValue("-f, --fragmentCount", Options["--fragmentCount"], "yes no")
 629     
 630     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "2D 3D All FragmentCountOnly Specify")
 631     
 632     if re.match("^Specify$", Options["--mode"], re.I):
 633         if re.match("^none$", Options["--descriptorNames"], re.I):
 634             MiscUtil.PrintError("The name(s) of molecular descriptors must be specified using \"-d, --descriptorNames\" option during \"Specify\" value of \"-m, --mode\" option.")
 635     
 636     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 637     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi csv tsv txt")
 638     
 639     if re.match("^3D|All$", Options["--mode"], re.I):
 640         if MiscUtil.CheckFileExt(Options["--infile"], "smi"):
 641             MiscUtil.PrintError("The input SMILES file, %s, is not valid for  \"3D or All\" value of \"-m, --mode\" option.")
 642     
 643     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd csv tsv txt")
 644     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 645     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 646     
 647     MiscUtil.ValidateOptionNumberValue("-p, --precision", int(Options["--precision"]), {">": 0})
 648         
 649 if __name__ == "__main__":
 650     main()
