  29 """
  30 RDKitCalculateRMSD.py - Calculate RMSD between molecules
  31 
  32 Usage:
  33     RDKitCalculateRMSD.py [--calcRMSD <RMSD, BestRMSD>] [--infileParams <Name,Value,...>]
  34                           [--maxIters <number>] [--mode <OneToOne, AllToAll, FirstToAll>]
  35                           [--overwrite] [-w <dir>] -r <reffile> -p <probefile> -o <outfile> 
  36     RDKitCalculateRMSD.py -h | --help | -e | --examples
  37 
  38 Description:
  39     Calculate Root Mean Square Distance (RMSD) between a set of similar molecules in
  40     reference and probe input files. The RDKit function fails to calculate RMSD values for
  41     dissimilar molecules. Consequently, a text string 'None' is written out as a RMSD value
  42     for dissimilar molecule pairs.
  43 
  44     The supported input file formats are: Mol (.mol), SD (.sdf, .sd)
  45 
  46     The supported output file formats are:  CSV (.csv), TSV (.tsv, .txt)
  47 
  48 Options:
  49     -c, --calcRMSD <RMSD, BestRMSD>  [default: RMSD]
  50         Methodology for calculating RMSD values. Possible values: RMSD, BestRMSD.
  51         During BestRMSMode mode, the RDKit 'function AllChem.GetBestRMS' is used to
  52         align and calculate RMSD. This function calculates optimal RMSD for aligning two
  53         molecules, taking symmetry into account. Otherwise, the RMSD value is calculated
  54         using 'AllChem.AlignMol function' without changing the atom order. A word to the
  55         wise from RDKit documentation: The AllChem.GetBestRMS function will attempt to
  56         align all permutations of matching atom orders in both molecules, for some molecules
  57         it will lead to 'combinatorial explosion'.
  58     --infileParams <Name,Value,...>  [default: auto]
  59         A comma delimited list of parameter name and value pairs for reading
  60         molecules from files. The supported parameter names for different file
  61         formats, along with their default values, are shown below:
  62             
  63             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  64             
  65     --maxIters <number>  [default: 50]
  66         Maximum number of iterations to perform for each molecule pair during minimization
  67         of RMSD values. This option is ignored during BestRMSD mode.
  68     -m, --mode <OneToOne, AllToAll, FirstToAll>  [default: OneToOne]
  69         Specify how molecules are handled in reference and probe input files during
  70         calculation of RMSD between reference and probe molecules.  Possible values:
  71         OneToOne, AllToAll and AllToFirst. For OneToOne mode, the number of molecules
  72         in reference file must be equal to the number of molecules in probe file. The RMSD
  73         is calculated for each pair of molecules in the reference and probe file and written
  74         to the output file. For AllToAll mode, the RMSD is calculated for each reference
  75         molecule against all probe molecules. For FirstToAll mode, however, the RMSD
  76         is only calculated for the first reference molecule against all probe molecules.
  77     -e, --examples
  78         Print examples.
  79     -h, --help
  80         Print this help message.
  81     -p, --probefile <probefile>
  82         Probe input file name.
  83     -r, --reffile <reffile>
  84         Reference input file name.
  85     -o, --outfile <outfile>
  86         Output file name for writing out RMSD values. Supported text file extensions: csv or tsv.
  87     --overwrite
  88         Overwrite existing files.
  89     -w, --workingdir <dir>
  90         Location of working directory which defaults to the current directory.
  91 
  92 Examples:
  93     To calculate RMSD between pair of molecules in reference and probe molecules in
  94     3D SD files and write out a CSV file containing calculated RMSD values along with
  95     appropriate molecule IDs, type:
  96 
  97         % RDKitCalculateRMSD.py  -r Sample3DRef.sdf -p Sample3DProb.sdf
  98           -o SampleOut.csv
  99 
 100     To calculate RMSD between all molecules in reference and probe molecules in
 101     3D SD files and write out a CSV file containing calculated RMSD values along with
 102     appropriate molecule IDs, type:
 103 
 104         % RDKitCalculateRMSD.py  -m AllToAll -r Sample3DRef.sdf -p
 105           Sample3DProb.sdf -o SampleOut.csv
 106 
 107     To calculate best RMSD between first  molecule in reference all probe molecules
 108     in 3D SD files and write out a TSV file containing calculated RMSD values along with
 109     appropriate molecule IDs, type:
 110 
 111         % RDKitCalculateRMSD.py  -m FirstToAll --calcRMSD BestRMSD -r
 112           Sample3DRef.sdf -p Sample3DProb.sdf -o SampleOut.tsv
 113 
 114     To calculate RMSD between all molecules in reference and probe molecules in
 115     3D SD files without removing hydrogens and write out a TSV file containing
 116     calculated RMSD values along with appropriate molecule IDs, type:
 117 
 118         % RDKitCalculateRMSD.py  -m AllToAll --infileParams
 119           "removeHydrogens,no" -r Sample3DRef.sdf  -p Sample3DProb.sdf
 120           -o SampleOut.tsv

 176 def main():
 177     """Start execution of the script"""
 178     
 179     MiscUtil.PrintInfo("\n%s (RDK v%s): Starting...\n" % (ScriptName, rdBase.rdkitVersion))

 195 def CalculateRMSD():
 196     """Calculate RMSD values."""
 197     
 198     if not re.match("^(OneToOne|AllToAll|FirstToAll)$", OptionsInfo["Mode"], re.I):
 199         MiscUtil.PrintError("RMSD couldn't be calculated: Specified mode, %s, is not supported" % OptionsInfo["Mode"])
 200         
 201     RefFile = OptionsInfo["RefFile"]
 202     ProbeFile = OptionsInfo["ProbeFile"]
 203     
 204     Outfile = OptionsInfo["Outfile"]
 205     OutDelim = OptionsInfo["OutDelim"]
 206 
 207     # Read reference and probe molecules...
 208     OptionsInfo["InfileParams"]["AllowEmptyMols"] = False
 209     
 210     MiscUtil.PrintInfo("\nProcessing file %s..." % (RefFile))
 211     ValidRefMols, RefMolCount, ValidRefMolCount  = RDKitUtil.ReadAndValidateMolecules(RefFile, **OptionsInfo["InfileParams"])
 212     
 213     MiscUtil.PrintInfo("Processing file %s..." % (ProbeFile))
 214     ValidProbeMols, ProbeMolCount, ValidProbeMolCount  = RDKitUtil.ReadAndValidateMolecules(ProbeFile, **OptionsInfo["InfileParams"])
 215 
 216     # Set up output file...
 217     MiscUtil.PrintInfo("\nGenerating file %s...\n" % Outfile)
 218     OutFH = open(Outfile, "w")
 219     if OutFH is None:
 220         MiscUtil.PrintError("Couldn't open output file: %s.\n" % (Outfile))
 221     
 222     Line = "RefMolID%sProbeMolID%sRMSD\n" % (OutDelim, OutDelim)
 223     OutFH.write(Line)
 224     
 225     if re.match("^OneToOne$", OptionsInfo["Mode"], re.I):
 226         CalculateOneToOneRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 227     elif re.match("^AllToAll$", OptionsInfo["Mode"], re.I):
 228         CalculateAllToAllRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 229     elif re.match("^FirstToAll$", OptionsInfo["Mode"], re.I):
 230         CalculateFirstToAllRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 231     else:
 232         MiscUtil.PrintError("RMSD couldn't be calculated: Specified mode, %s, is not supported" % OptionsInfo["Mode"])
 233 
 234     OutFH.close()
 235     
 236     MiscUtil.PrintInfo("\nTotal number of molecules: Reference - %d; Probe - %d" % (RefMolCount, ProbeMolCount))
 237     MiscUtil.PrintInfo("Number of valid molecules: Reference - %d; Probe - %d" % (ValidRefMolCount, ValidProbeMolCount))
 238     MiscUtil.PrintInfo("Number of ignored molecules:  Reference - %d; Probe - %d" % ((RefMolCount - ValidRefMolCount), (ProbeMolCount - ValidProbeMolCount)))
 239     
 240 def CalculateOneToOneRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 241     """Calculate pairwise RMSD values."""
 242     
 243     ValidRefMolCount = len(ValidRefMols)
 244     ValidProbeMolCount = len(ValidProbeMols)
 245     
 246     MolCount = ValidRefMolCount
 247     if ValidRefMolCount > ValidProbeMolCount:
 248         MolCount = ValidProbeMolCount
 249     
 250     if ValidRefMolCount != ValidProbeMolCount:
 251         MiscUtil.PrintWarning("Number of valid reference molecules, %d,  is not equal to number of valid probe molecules, %d .\n" % (ValidRefMolCount, ValidProbeMolCount))
 252         MiscUtil.PrintWarning("Pairwise RMSD will be calculated only for first %s molecules.\n" % (MolCount))
 253 
 254     # Process molecules...
 255     for MolIndex in range(0, MolCount):
 256         RefMol = ValidRefMols[MolIndex]
 257         ProbeMol = ValidProbeMols[MolIndex]
 258 
 259         RefMolName = RDKitUtil.GetMolName(RefMol, (MolIndex + 1))
 260         ProbeMolName = RDKitUtil.GetMolName(ProbeMol, (MolIndex + 1))
 261 
 262         RMSD = CalculateRMSDValue(RefMol, ProbeMol)
 263         
 264         Line = "%s%s%s%s%s\n" % (RefMolName, OutDelim, ProbeMolName, OutDelim, RMSD)
 265         OutFH.write(Line)
 266         
 267 def CalculateAllToAllRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 268     """Calculate RMSD values between all pairs of molecules."""
 269     
 270     # Process molecules...
 271     RefMolCount = 0
 272     for RefMol in ValidRefMols:
 273         RefMolCount += 1
 274         RefMolName = RDKitUtil.GetMolName(RefMol, RefMolCount)
 275 
 276         ProbeMolCount = 0
 277         for ProbeMol in ValidProbeMols:
 278             ProbeMolCount += 1
 279             ProbeMolName = RDKitUtil.GetMolName(ProbeMol, ProbeMolCount)
 280             
 281             RMSD = CalculateRMSDValue(RefMol, ProbeMol)
 282 
 283             Line = "%s%s%s%s%s\n" % (RefMolName, OutDelim, ProbeMolName, OutDelim, RMSD)
 284             OutFH.write(Line)
 285         
 286 def CalculateFirstToAllRMSDValues(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 287     """Calculate RMSD values between first reference molecues and all probe molecules. """
 288     
 289     # Process molecules...
 290     RefMol = ValidRefMols[0]
 291     RefMolCount = 1
 292     RefMolName = RDKitUtil.GetMolName(RefMol, RefMolCount)
 293 
 294     ProbeMolCount = 0
 295     for ProbeMol in ValidProbeMols:
 296         ProbeMolCount += 1
 297         ProbeMolName = RDKitUtil.GetMolName(ProbeMol, ProbeMolCount)
 298             
 299         RMSD = CalculateRMSDValue(RefMol, ProbeMol)
 300 
 301         Line = "%s%s%s%s%s\n" % (RefMolName, OutDelim, ProbeMolName, OutDelim, RMSD)
 302         OutFH.write(Line)
 303         
 304 def CalculateRMSDValue(RefMol, ProbeMol):
 305     """Calculate RMSD value for a pair of molecules and return it as a string"""
 306 
 307     try:
 308         if OptionsInfo["UseBestRMSD"]:
 309             RMSD = AllChem.GetBestRMS(RefMol, ProbeMol)
 310         else:
 311             RMSD = rdMolAlign.AlignMol(ProbeMol, RefMol, maxIters = OptionsInfo["MaxIters"])
 312         RMSD = "%.2f" % RMSD
 313     except (RuntimeError, ValueError):
 314         RMSD = "None"
 315         
 316     return RMSD

 318 def ProcessOptions():
 319     """Process and validate command line arguments and options"""
 320     
 321     MiscUtil.PrintInfo("Processing options...")
 322     
 323     # Validate options...
 324     ValidateOptions()
 325     
 326     OptionsInfo["CalcRMSD"] = Options["--calcRMSD"]
 327     OptionsInfo["UseBestRMSD"] = False
 328     if re.match("^BestRMSD$", OptionsInfo["CalcRMSD"], re.I):
 329         OptionsInfo["UseBestRMSD"] = True
 330     
 331     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 332     
 333     OptionsInfo["Mode"] = Options["--mode"]
 334     
 335     OptionsInfo["RefFile"] = Options["--reffile"]
 336     OptionsInfo["ProbeFile"] = Options["--probefile"]
 337 
 338     # No need for any RDKit specific --outfileParams....
 339     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"])
 340     
 341     OptionsInfo["Outfile"] = Options["--outfile"]
 342     
 343     OptionsInfo["Overwrite"] = Options["--overwrite"]
 344     
 345     OptionsInfo["OutDelim"] = " "
 346     if MiscUtil.CheckFileExt(OptionsInfo["Outfile"], "csv"):
 347         OptionsInfo["OutDelim"] = ","
 348     elif MiscUtil.CheckFileExt(OptionsInfo["Outfile"], "tsv txt"):
 349         OptionsInfo["OutDelim"] = "\t"
 350     else:
 351         MiscUtil.PrintError("The file name specified , %s, for option \"--outfile\" is not valid. Supported file formats: csv tsv txt\n" % (OptionsInfo["Outfile"]))
 352 
 353 def RetrieveOptions():
 354     """Retrieve command line arguments and options"""
 355     
 356     # Get options...
 357     global Options
 358     Options = docopt(__doc__)
 359     
 360     # Set current working directory to the specified directory...
 361     WorkingDir = Options["--workingdir"]
 362     if WorkingDir:
 363         os.chdir(WorkingDir)
 364     
 365     # Handle examples option...
 366     if "--examples" in Options and Options["--examples"]:
 367         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 368         sys.exit(0)
 369 
 370 def ValidateOptions():
 371     """Validate option values"""
 372     
 373     MiscUtil.ValidateOptionTextValue("--calcRMSD", Options["--calcRMSD"], "RMSD BestRMSD")
 374     
 375     MiscUtil.ValidateOptionNumberValue("--maxIters", int(Options["--maxIters"]), {">": 0})
 376     MiscUtil.ValidateOptionTextValue("--mode", Options["--mode"], "OneToOne  AllToAll FirstToAll")
 377     
 378     MiscUtil.ValidateOptionFilePath("-r, --reffile", Options["--reffile"])
 379     MiscUtil.ValidateOptionFileExt("-r, --reffile", Options["--reffile"], "sdf sd mol")
 380     
 381     MiscUtil.ValidateOptionFilePath("-p, --probefile", Options["--probefile"])
 382     MiscUtil.ValidateOptionFileExt("-p, --probefile", Options["--probefile"], "sdf sd mol")
 383         
 384     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "csv tsv txt")
 385     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 386     MiscUtil.ValidateOptionsDistinctFileNames("-r, --reffile", Options["--reffile"], "-o, --outfile", Options["--outfile"])
 387     MiscUtil.ValidateOptionsDistinctFileNames("-p, --probefile", Options["--probefile"], "-o, --outfile", Options["--outfile"])
 388     
 389 if __name__ == "__main__":
 390     main()
