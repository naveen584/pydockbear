  29 """
  30 RDKitCompareMoleculeShapes.py - Compare shapes of molecules
  31 
  32 Usage:
  33     RDKitCompareMoleculeShapes.py [--alignment <Open3A, CrippenOpen3A>]
  34                                   [--distance <Tanimoto, Protrude, Both>]  [--infileParams <Name,Value,...>]
  35                                   [--maxIters <number>] [--mode <OneToOne, AllToAll, FirstToAll>]
  36                                   [--overwrite] [-w <dir>] -r <reffile> -p <probefile> -o <outfile> 
  37     RDKitCompareMoleculeShapes.py -h | --help | -e | --examples
  38 
  39 Description:
  40     Compare shapes of molecules between a set of molecules in reference and probe
  41     input files. The molecules are aligned using either Open 3DAlign or Crippen Open
  42     3DAlign before calculating shape Tanimoto and protrude distances.
  43 
  44     The supported input file formats are: Mol (.mol), SD (.sdf, .sd)
  45 
  46     The supported output file formats are: CSV (.csv), TSV (.tsv, .txt)
  47 
  48 Options:
  49     -a, --alignment <Open3A, CrippenOpen3A>  [default: Open3A]
  50         Alignment methodology to use for aligning molecules before calculating Tanimoto and
  51         protrude shape distances. Possible values: Open3A or CrippenOpen3A. Open 3DAlign
  52         (Open3A) [ Ref 132 ] overlays molecules based on MMFF atom types and charges.
  53         Crippen Open 3DAlign (CrippenOpen3A) uses Crippen logP contributions to overlay
  54         molecules.
  55     -d, --distance <Tanimoto, Protrude, Both>  [default: Both]
  56         Shape comparison distance to calculate for comparing shapes of molecules. Possible
  57         values: Tanimoto, Protrude, or Both. Shape Tanimoto distance takes the volume
  58         overlay into account during the calculation of distance. Shape protrude distance,
  59         however, focuses on the volume mismatch.
  60     --infileParams <Name,Value,...>  [default: auto]
  61         A comma delimited list of parameter name and value pairs for reading
  62         molecules from files. The supported parameter names for different file
  63         formats, along with their default values, are shown below:
  64             
  65             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  66             
  67     --maxIters <number>  [default: 50]
  68         Maximum number of iterations to perform for each molecule pair during alignment.
  69     -m, --mode <OneToOne, AllToAll, FirstToAll>  [default: OneToOne]
  70         Specify how molecules are handled in reference and probe input files during
  71         comparison of shapes between reference and probe molecules.  Possible values:
  72         OneToOne, AllToAll and AllToFirst. For OneToOne mode, the molecule shapes are
  73         calculated for each pair of molecules in the reference and probe file and the shape
  74         distances are written to the output file. For AllToAll mode, the shape distances are
  75         calculated for each reference molecule against all probe molecules. For FirstToAll mode,
  76         however, the shape distances are only calculated for the first reference molecule
  77         against all probe molecules.
  78     -e, --examples
  79         Print examples.
  80     -h, --help
  81         Print this help message.
  82     -p, --probefile <probefile>
  83         Probe input file name.
  84     -r, --reffile <reffile>
  85         Reference input file name.
  86     -o, --outfile <outfile>
  87         Output file name for writing out shape distances. Supported text file extensions: csv or tsv.
  88     --overwrite
  89         Overwrite existing files.
  90     -w, --workingdir <dir>
  91         Location of working directory which defaults to the current directory.
  92 
  93 Examples:
  94     To perform shape alignment using Open3A methodology between pair of molecules in
  95     reference and probe molecules in 3D SD files, calculate both Tanimoto and protrude
  96     distances, and write out a CSV file containing calculated distance values along with
  97     appropriate molecule IDs, type:
  98 
  99         % RDKitCompareMoleculeShapes.py  -r Sample3DRef.sdf -p Sample3DProb.sdf
 100           -o SampleOut.csv
 101 
 102     To perform shape alignment using Crippen Open3A methodology between all molecules in
 103     reference and probe molecules in 3D SD files, calculate only Tanimoto distance, and write
 104     out a TSV file containing calculated distance value along with appropriate molecule IDs, type:
 105 
 106         % RDKitCompareMoleculeShapes.py  -m AllToAll -a CrippenOpen3A -d Tanimoto
 107           -r Sample3DRef.sdf -p Sample3DProb.sdf -o SampleOut.csv
 108 
 109     To perform shape alignment using Open3A methodology between first reference molecule
 110     against all probe molecules in 3D SD files without removing hydrogens , calculate both
 111     Tanimoto and protrude distances, and write out a CSV file containing calculated distance values along with
 112     appropriate molecule IDs, type:
 113 
 114         % RDKitCompareMoleculeShapes.py -m FirstToAll -a Open3A -d Both 
 115           --infileParams "removeHydrogens,no" -r Sample3DRef.sdf
 116           -p Sample3DProb.sdf -o SampleOut.csv

 191 def CompareMoleculeShapes():
 192     """Compare shape of molecules."""
 193     
 194     if not re.match("^(OneToOne|AllToAll|FirstToAll)$", OptionsInfo["Mode"], re.I):
 195         MiscUtil.PrintError("Shape comparison couldn't be performed: Specified mode, %s, is not supported" % OptionsInfo["Mode"])
 196         
 197     if not re.match("^(Open3A|CrippenOpen3A)$", OptionsInfo["Alignment"], re.I):
 198         MiscUtil.PrintError("Shape couldn't be performed: Specified alignment mode, %s, is not supported" % OptionsInfo["Alignment"])
 199         
 200     RefFile = OptionsInfo["RefFile"]
 201     ProbeFile = OptionsInfo["ProbeFile"]
 202     
 203     Outfile = OptionsInfo["Outfile"]
 204     OutDelim = OptionsInfo["OutDelim"]
 205 
 206     # Read reference and probe molecules...
 207     OptionsInfo["InfileParams"]["AllowEmptyMols"] = False
 208     
 209     MiscUtil.PrintInfo("\nProcessing file %s..." % (RefFile))
 210     ValidRefMols, RefMolCount, ValidRefMolCount  = RDKitUtil.ReadAndValidateMolecules(RefFile, **OptionsInfo["InfileParams"])
 211     
 212     MiscUtil.PrintInfo("Processing file %s..." % (ProbeFile))
 213     ValidProbeMols, ProbeMolCount, ValidProbeMolCount  = RDKitUtil.ReadAndValidateMolecules(ProbeFile, **OptionsInfo["InfileParams"])
 214 
 215     # Set up output file...
 216     MiscUtil.PrintInfo("Generating file %s...\n" % Outfile)
 217     OutFH = open(Outfile, "w")
 218     if OutFH is None:
 219         MiscUtil.PrintError("Couldn't open output file: %s.\n" % (Outfile))
 220 
 221     if OptionsInfo["UseCrippenOpen3A"]:
 222         Line = "RefMolID%sProbeMolID%sCrippenOpen3AScore" % (OutDelim, OutDelim)
 223     else:
 224         Line = "RefMolID%sProbeMolID%sOpen3AScore" % (OutDelim, OutDelim)
 225         
 226     if OptionsInfo["CalcTanimotoDistance"]:
 227         Line = "%s%sTanimotoDistance" % (Line, OutDelim)
 228     if OptionsInfo["CalcProtrudeDistance"]:
 229         Line = "%s%sProtrudeDistance" % (Line, OutDelim)
 230     OutFH.write("%s\n" % Line)
 231         
 232     if re.match("^OneToOne$", OptionsInfo["Mode"], re.I):
 233         PerformOneToOneShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 234     elif re.match("^AllToAll$", OptionsInfo["Mode"], re.I):
 235         PerformAllToAllShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 236     elif re.match("^FirstToAll$", OptionsInfo["Mode"], re.I):
 237         PerformFirstToAllShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim)
 238     else:
 239         MiscUtil.PrintError("Shape comaprison couldn't be performed: Specified mode, %s, is not supported" % OptionsInfo["Mode"])
 240 
 241     OutFH.close()
 242     
 243     MiscUtil.PrintInfo("\nTotal number of molecules: Reference - %d; Probe - %d" % (RefMolCount, ProbeMolCount))
 244     MiscUtil.PrintInfo("Number of valid molecules: Reference - %d; Probe - %d" % (ValidRefMolCount, ValidProbeMolCount))
 245     MiscUtil.PrintInfo("Number of ignored molecules:  Reference - %d; Probe - %d" % ((RefMolCount - ValidRefMolCount), (ProbeMolCount - ValidProbeMolCount)))
 246     
 247 def PerformOneToOneShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 248     """Perform pairwise shape comparison"""
 249     
 250     ValidRefMolCount = len(ValidRefMols)
 251     ValidProbeMolCount = len(ValidProbeMols)
 252     
 253     MolCount = ValidRefMolCount
 254     if ValidRefMolCount > ValidProbeMolCount:
 255         MolCount = ValidProbeMolCount
 256     
 257     if ValidRefMolCount != ValidProbeMolCount:
 258         MiscUtil.PrintWarning("Number of valid reference molecules, %d,  is not equal to number of valid probe molecules, %d .\n" % (ValidRefMolCount, ValidProbeMolCount))
 259         MiscUtil.PrintWarning("Pairwise shape comparison will be performed only for first %s molecules.\n" % (MolCount))
 260 
 261     # Process molecules...
 262     for MolIndex in range(0, MolCount):
 263         RefMol = ValidRefMols[MolIndex]
 264         ProbeMol = ValidProbeMols[MolIndex]
 265 
 266         RefMolName = RDKitUtil.GetMolName(RefMol, (MolIndex + 1))
 267         ProbeMolName = RDKitUtil.GetMolName(ProbeMol, (MolIndex + 1))
 268 
 269         PerformShapeComparisonAndWrieOutput(RefMol, ProbeMol, RefMolName, ProbeMolName, OutFH, OutDelim)
 270         
 271 def PerformAllToAllShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 272     """Perform shape comparison between all pairs of molecules."""
 273     
 274     # Process molecules...
 275     RefMolCount = 0
 276     for RefMol in ValidRefMols:
 277         RefMolCount += 1
 278         RefMolName = RDKitUtil.GetMolName(RefMol, RefMolCount)
 279 
 280         ProbeMolCount = 0
 281         for ProbeMol in ValidProbeMols:
 282             ProbeMolCount += 1
 283             ProbeMolName = RDKitUtil.GetMolName(ProbeMol, ProbeMolCount)
 284             
 285             PerformShapeComparisonAndWrieOutput(RefMol, ProbeMol, RefMolName, ProbeMolName, OutFH, OutDelim)
 286         
 287 def PerformFirstToAllShapeComparison(ValidRefMols, ValidProbeMols, OutFH, OutDelim):
 288     """Perform shape comparison between first reference molecues and all probe molecules. """
 289     
 290     # Process molecules...
 291     RefMol = ValidRefMols[0]
 292     RefMolCount = 1
 293     RefMolName = RDKitUtil.GetMolName(RefMol, RefMolCount)
 294 
 295     ProbeMolCount = 0
 296     for ProbeMol in ValidProbeMols:
 297         ProbeMolCount += 1
 298         ProbeMolName = RDKitUtil.GetMolName(ProbeMol, ProbeMolCount)
 299         
 300         PerformShapeComparisonAndWrieOutput(RefMol, ProbeMol, RefMolName, ProbeMolName, OutFH, OutDelim)
 301 
 302 def PerformShapeComparisonAndWrieOutput(RefMol, ProbeMol, RefMolName, ProbeMolName, OutFH, OutDelim):
 303     """Perform shape comparison and write to output file."""
 304 
 305     AlignmentScore = PerformShapeAlignment(RefMol, ProbeMol)
 306     AlignmentScore = "%.2f" % AlignmentScore
 307     
 308     LineWords = []
 309     LineWords.extend([RefMolName, ProbeMolName, AlignmentScore])
 310     
 311     if OptionsInfo["CalcTanimotoDistance"]:
 312         TanimotoDistance = CalculateTanimotoShapeDistance(RefMol, ProbeMol)
 313         LineWords.append(TanimotoDistance)
 314     
 315     if OptionsInfo["CalcProtrudeDistance"]:
 316         ProtrudeDistance = CalculateProtrudeShapeDistance(RefMol, ProbeMol)
 317         LineWords.append(ProtrudeDistance)
 318     
 319     Line = OutDelim.join(LineWords)
 320     OutFH.write("%s\n" % Line)
 321     
 322 def PerformShapeAlignment(RefMol, ProbeMol):
 323     """Perform shape alignment and return alignment score."""
 324     
 325     if OptionsInfo["UseCrippenOpen3A"]:
 326         CrippenO3A = rdMolAlign.GetCrippenO3A(ProbeMol, RefMol)
 327         Score = CrippenO3A.Align()
 328     else:
 329         O3A = rdMolAlign.GetO3A(ProbeMol, RefMol)
 330         Score = O3A.Align()
 331 
 332     return Score
 333         
 334 def CalculateTanimotoShapeDistance(RefMol, ProbeMol):
 335     """Calculate Tanimoto shape for a pair of already aligned molecules and return it as a string"""
 336 
 337     Distance = rdShapeHelpers.ShapeTanimotoDist(ProbeMol, RefMol)
 338     Distance = "%.2f" % Distance
 339 
 340     return Distance
 341 
 342 def CalculateProtrudeShapeDistance(RefMol, ProbeMol):
 343     """Calculate protrude shape for a pair of already aligned molecules and return it as a string"""
 344 
 345     Distance = rdShapeHelpers.ShapeProtrudeDist(ProbeMol, RefMol)
 346     Distance = "%.2f" % Distance
 347 
 348     return Distance

 350 def ProcessOptions():
 351     """Process and validate command line arguments and options"""
 352     
 353     MiscUtil.PrintInfo("Processing options...")
 354     
 355     # Validate options...
 356     ValidateOptions()
 357     
 358     OptionsInfo["Alignment"] = Options["--alignment"]
 359     OptionsInfo["UseCrippenOpen3A"] = False
 360     if re.match("^CrippenOpen3A$", OptionsInfo["Alignment"], re.I):
 361         OptionsInfo["UseCrippenOpen3A"] = True
 362     
 363     OptionsInfo["Distance"] = Options["--distance"]
 364     
 365     OptionsInfo["CalcTanimotoDistance"] = False
 366     OptionsInfo["CalcProtrudeDistance"] = False
 367     if re.match("^Tanimoto$", OptionsInfo["Distance"], re.I):
 368         OptionsInfo["CalcTanimotoDistance"] = True
 369     elif re.match("^Protrude$", OptionsInfo["Distance"], re.I):
 370         OptionsInfo["CalcProtrudeDistance"] = True
 371     else:
 372         OptionsInfo["CalcTanimotoDistance"] = True
 373         OptionsInfo["CalcProtrudeDistance"] = True
 374     
 375     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 376     OptionsInfo["Mode"] = Options["--mode"]
 377     
 378     OptionsInfo["RefFile"] = Options["--reffile"]
 379     OptionsInfo["ProbeFile"] = Options["--probefile"]
 380     
 381     OptionsInfo["Outfile"] = Options["--outfile"]
 382     OptionsInfo["Overwrite"] = Options["--overwrite"]
 383     
 384     # No need for any RDKit specific --outfileParams....
 385     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"])
 386     
 387     OptionsInfo["OutDelim"] = " "
 388     if MiscUtil.CheckFileExt(OptionsInfo["Outfile"], "csv"):
 389         OptionsInfo["OutDelim"] = ","
 390     elif MiscUtil.CheckFileExt(OptionsInfo["Outfile"], "tsv txt"):
 391         OptionsInfo["OutDelim"] = "\t"
 392     else:
 393         MiscUtil.PrintError("The file name specified , %s, for option \"--outfile\" is not valid. Supported file formats: csv tsv txt\n" % (OptionsInfo["Outfile"]))
 394     
 395 def RetrieveOptions():
 396     """Retrieve command line arguments and options"""
 397     
 398     # Get options...
 399     global Options
 400     Options = docopt(__doc__)
 401 
 402     # Set current working directory to the specified directory...
 403     WorkingDir = Options["--workingdir"]
 404     if WorkingDir:
 405         os.chdir(WorkingDir)
 406     
 407     # Handle examples option...
 408     if "--examples" in Options and Options["--examples"]:
 409         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 410         sys.exit(0)
 411

 421     
 422     MiscUtil.ValidateOptionFilePath("-r, --reffile", Options["--reffile"])
 423     MiscUtil.ValidateOptionFileExt("-r, --reffile", Options["--reffile"], "sdf sd mol")
 424     
 425     MiscUtil.ValidateOptionFilePath("-p, --probefile", Options["--probefile"])
 426     MiscUtil.ValidateOptionFileExt("-p, --probefile", Options["--probefile"], "sdf sd mol")
 427     
 428     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "csv tsv txt")
 429     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 430     
 431     MiscUtil.ValidateOptionsDistinctFileNames("-r, --reffile", Options["--reffile"], "-o, --outfile", Options["--outfile"])
 432     MiscUtil.ValidateOptionsDistinctFileNames("-p, --probefile", Options["--probefile"], "-o, --outfile", Options["--outfile"])
 433 
 434 if __name__ == "__main__":
 435     main()
