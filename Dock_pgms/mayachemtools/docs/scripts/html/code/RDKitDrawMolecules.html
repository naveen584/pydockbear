  29 """
  30 RDKitDrawMolecules.py - Draw molecules and generate an image file
  31 
  32 Usage:
  33     RDKitDrawMolecules.py [--alignmentSMARTS <SMARTS>] [--atomLabelFontSize <number>]
  34                              [--bondLineWidth <number>] [--compute2DCoords <yes | no>] [--fontBold <yes or no>]
  35                              [--highlightSMARTS <SMARTS>] [--infileParams <Name,Value,...>] [--kekulize <yes or no>]
  36                              [--molImageSize <width,height>] [--numOfMolsPerRow <number>] [--overwrite]
  37                              [--showMolName <yes or no>] [-w <dir>] -i <infile> -o <outfile>
  38     RDKitDrawMolecules.py -h | --help | -e | --examples
  39 
  40 Description:
  41     Draw molecules in a grid and write them out as an image file. The SVG image file
  42     appears to be the best among all the available image file options, as rendered in a
  43     browser. The Python modules aggdraw/cairo are required to generate high quality
  44     PNG images.
  45 
  46     The options '--atomLabelFontSize' and '--bondLineWidth' don't appear
  47     to work during generation of a SVG image.
  48 
  49     The supported input file formats are: Mol (.mol), SD (.sdf, .sd), SMILES (.smi,
  50     .txt, .csv, .tsv)
  51 
  52     The output image file can be saved in any format supported by the Python Image
  53     Library (PIL). The image format is automatically detected from the output file extension.
  54 
  55     Some of the most common output image file formats are: GIF (.gif), JPEG (.jpg),
  56     PNG (.png), SVG (.svg), TIFF (.tif)
  57 
  58 Options:
  59     --alignmentSMARTS <SMARTS>  [default: none]
  60         SMARTS pattern for aligning molecules to a common template.
  61     --atomLabelFontSize <number>  [default: 12]
  62         Font size for drawing atom labels. This option is ignored during generation of a SVG
  63         output file.
  64     -b, --bondLineWidth <number>  [default: 1.2]
  65         Line width for drawing bonds. This option is ignored during generation of a SVG
  66         output file.
  67     -c, --compute2DCoords <yes | no>  [default: auto]
  68         Compute 2D coordinates of molecules before drawing. Default: yes for SMILES
  69         file; no for all other file types.
  70     -e, --examples
  71         Print examples.
  72     -f --fontBold <yes or no>  [default: yes]
  73         Make all text fonts bold during generation of  a SVG output file. This option is ignored for
  74         all other output files.
  75     -h, --help
  76         Print this help message.
  77     --highlightSMARTS <SMARTS>  [default: none]
  78         SMARTS pattern for highlighting atoms and bonds in molecules. All matched
  79         substructures are highlighted.
  80     -i, --infile <infile>
  81         Input file name.
  82     --infileParams <Name,Value,...>  [default: auto]
  83         A comma delimited list of parameter name and value pairs for reading
  84         molecules from files. The supported parameter names for different file
  85         formats, along with their default values, are shown below:
  86             
  87             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  88             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  89                 smilesTitleLine,auto,sanitize,yes
  90             
  91         Possible values for smilesDelimiter: space, comma or tab.
  92     -k, --kekulize <yes or no>  [default: yes]
  93         Perform kekulization on molecules. This option is ignored during generation of a SVG
  94         output file.
  95     -m, --molImageSize <width,height>  [default: 400,300]
  96         Image size of a molecule in pixels.
  97     -n, --numOfMolsPerRow <number>  [default: 2]
  98         Number of molecules to draw in a row.
  99     -o, --outfile <outfile>
 100         Output file name.
 101     --overwrite
 102         Overwrite existing files.
 103     -s, --showMolName <yes or no>  [default: yes]
 104         Show molecule names under the images.
 105     -w, --workingdir <dir>
 106         Location of working directory which defaults to the current directory.
 107 
 108 Examples:
 109     To automatically compute 2D coordinates for molecules in a SMILES file and
 110     generate a SVG image file containing 2 molecules per row in a grid with cell
 111     size of 400 x 300 pixels, type:
 112 
 113         % RDKitDrawMolecules.py -i Sample.smi -o SampleOut.svg
 114 
 115     To automatically compute 2D coordinates for molecules in a SMILES file and
 116     generate a SVG image file containing 4 molecules per row in a grid with cell
 117     size of 200 x 200 pixels and without any keulization along with highlighting
 118     a specific set of atoms and bonds indicated by a SMARTS pattern, type:
 119 
 120         % RDKitDrawMolecules.py -n 4 -m "200,200" -k no --fontBold no
 121           --highlightSMARTS  'c1ccccc1' -i Sample.smi -o SampleOut.svg
 122 
 123     To generate a PNG image file for molecules in a SD file using existing 2D
 124     coordinates, type
 125 
 126         % RDKitDrawMolecules.py --compute2DCoords no -i Sample.sdf
 127           -o SampleOut.png
 128 
 129     To automatically compute 2D coordinates for molecules in a CSV SMILES file
 130     with column headers, SMILES strings in column 1, and name in column 2 and
 131     generate a PDF image file, type:
 132 
 133         % RDKitDrawMolecules.py --infileParams "smilesDelimiter,comma,
 134           smilesTitleLine,yes,smilesColumn,1,smilesNameColumn,2"
 135           -i SampleSMILES.csv -o SampleOut.pdf

 190 Options = {}

 193 def main():
 194     """Start execution of the script"""
 195     
 196     MiscUtil.PrintInfo("\n%s (RDK v%s): Starting...\n" % (ScriptName, rdBase.rdkitVersion))
 197     
 198     (WallClockTime, ProcessorTime) = MiscUtil.GetWallClockAndProcessorTime()
 199     
 200     # Retrieve command line arguments and options...
 201     RetrieveOptions()
 202     
 203     # Process and validate command line arguments and options...
 204     ProcessOptions()
 205     
 206     # Perform actions required by the script...
 207     DrawMolecules()
 208     
 209     MiscUtil.PrintInfo("\n%s: Done...\n" % ScriptName)
 210     MiscUtil.PrintInfo("Total time: %s" % MiscUtil.GetFormattedElapsedTime(WallClockTime, ProcessorTime))

 212 def DrawMolecules():
 213     """Draw molecules"""
 214     
 215     Infile = OptionsInfo["Infile"]
 216     Outfile = OptionsInfo["Outfile"]
 217     
 218     # Read molecules...
 219     MiscUtil.PrintInfo("\nReading file %s..." % Infile)
 220 
 221     ValidMols, MolCount, ValidMolCount  = RDKitUtil.ReadAndValidateMolecules(Infile, **OptionsInfo["InfileParams"])
 222     
 223     MiscUtil.PrintInfo("Total number of molecules: %d" % MolCount)
 224     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 225     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 226 
 227     # Compute 2D coordinates...
 228     if OptionsInfo["Compute2DCoords"]:
 229         MiscUtil.PrintInfo("\nComputing 2D coordinates...")
 230         for Mol in ValidMols:
 231             AllChem.Compute2DCoords(Mol)
 232     
 233     MiscUtil.PrintInfo("Generating image grid...")
 234     
 235     # Setup atoms lists for highlighting atoms and bonds...
 236     AtomListsToHighlight = SetupAtomListsToHighlight(ValidMols)
 237     BondListsToHighlight = None
 238         
 239     # Set up legends...
 240     MolNames = None
 241     if OptionsInfo["ShowMolName"]:
 242         MolNames = []
 243         MolCount = 0
 244         for Mol in ValidMols:
 245             MolCount += 1
 246             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 247             MolNames.append(MolName)
 248     
 249     # Perform alignment to a common template...
 250     PerformAlignment(ValidMols)
 251     
 252     # Generate molecule images...
 253     NumOfMolsPerRow = OptionsInfo["NumOfMolsPerRow"]
 254     Width = OptionsInfo["MolImageWidth"]
 255     Height = OptionsInfo["MolImageHeight"]
 256 
 257     # Setup drawing options...
 258     UpdatedDrawingOptions = DrawingOptions()
 259     UpdatedDrawingOptions.atomLabelFontSize = int(OptionsInfo["AtomLabelFontSize"])
 260     UpdatedDrawingOptions.bondLineWidth = float(OptionsInfo["BondLineWidth"])
 261     
 262     # Setup generation of SVG image...
 263     UseSVG= False
 264     if MiscUtil.CheckFileExt(Outfile, "svg"):
 265         UseSVG = True
 266 
 267     if UseSVG:
 268         MolsImage = Draw.MolsToGridImage(ValidMols, molsPerRow = NumOfMolsPerRow, subImgSize = (Width,Height), legends = MolNames, highlightAtomLists = AtomListsToHighlight, highlightBondLists = BondListsToHighlight, useSVG = UseSVG)
 269 
 270         # It appears that the string 'xmlns:svg' needs to be replaced with 'xmlns' in the
 271         # SVG image string generated by RDKit. Otherwise, the image doesn't load in 
 272         # web browsers.
 273         #
 274         if re.search("xmlns:svg", MolsImage, re.I):
 275             MolsImage = re.sub("xmlns:svg", "xmlns", MolsImage)
 276         
 277         # Make mol names bold...
 278         if OptionsInfo["FontBold"]:
 279             MolsImage = re.sub("font-weight:normal;", "font-weight:bold;", MolsImage)
 280         
 281     else:
 282         MolsImage = Draw.MolsToGridImage(ValidMols, molsPerRow = NumOfMolsPerRow,  subImgSize = (Width,Height), legends = MolNames, highlightAtomLists = AtomListsToHighlight, highlightBondLists = BondListsToHighlight, useSVG = UseSVG, kekulize = OptionsInfo["Kekulize"],  options = UpdatedDrawingOptions)
 283 
 284     # Write out the image file...
 285     MiscUtil.PrintInfo("\nGenerating image file %s..." % Outfile)
 286     
 287     if UseSVG:
 288         OutFH = open(Outfile, "w")
 289         OutFH.write(MolsImage)
 290         OutFH.close()
 291     else:
 292         if MiscUtil.CheckFileExt(Outfile, "pdf"):
 293             if MolsImage.mode == 'RGBA':
 294                 MolsImage = MolsImage.convert('RGB')
 295 
 296         MolsImage.save(Outfile)
 297 
 298 def SetupAtomListsToHighlight(ValidMols):
 299     """Set up atom lists to highlight using specified SMARTS pattern."""
 300 
 301     AtomListsToHighlight = None
 302     if OptionsInfo["HighlightSMARTSPattern"] is None:
 303         return  AtomListsToHighlight
 304     
 305     PatternMol = Chem.MolFromSmarts(OptionsInfo["HighlightSMARTSPattern"])
 306     AtomListsToHighlight = []
 307     for ValidMol in ValidMols:
 308         # Get matched atom lists and flatten it...
 309         MatchedAtomsLists = ValidMol.GetSubstructMatches(PatternMol)
 310         MatchedAtoms = [ Atom for AtomsList in MatchedAtomsLists for Atom in AtomsList] 
 311         AtomListsToHighlight.append(MatchedAtoms)
 312     
 313     return  AtomListsToHighlight
 314 
 315 def PerformAlignment(ValidMols):
 316     """Perform alignment to a common template specified by a SMARTS pattern."""
 317     
 318     if OptionsInfo["AlignmentSMARTSPattern"] is None:
 319         return
 320     
 321     PatternMol = Chem.MolFromSmarts(OptionsInfo["AlignmentSMARTSPattern"])
 322     AllChem.Compute2DCoords(PatternMol)
 323         
 324     MatchedValidMols = [ValidMol for ValidMol in ValidMols if ValidMol.HasSubstructMatch(PatternMol)]
 325     for ValidMol in MatchedValidMols:
 326         AllChem.GenerateDepictionMatching2DStructure(ValidMol, PatternMol)
 327 
 328 def ProcessOptions():
 329     """Process and validate command line arguments and options"""
 330     
 331     MiscUtil.PrintInfo("Processing options...")
 332     
 333     # Validate options...
 334     ValidateOptions()
 335     
 336     OptionsInfo["Infile"] = Options["--infile"]
 337     OptionsInfo["Outfile"] = Options["--outfile"]
 338     OptionsInfo["Overwrite"] = Options["--overwrite"]
 339     
 340     # No need for any RDKit specific --outfileParams....
 341     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], OptionsInfo["Infile"])
 342 
 343     AlignmentSMARTSPattern = None
 344     if not re.match("^None$", Options["--alignmentSMARTS"], re.I):
 345         AlignmentSMARTSPattern = Options["--alignmentSMARTS"]
 346     OptionsInfo["AlignmentSMARTSPattern"]  = AlignmentSMARTSPattern
 347     
 348     OptionsInfo["AtomLabelFontSize"] = Options["--atomLabelFontSize"]
 349     OptionsInfo["BondLineWidth"] = Options["--bondLineWidth"]
 350     
 351     Compute2DCoords = False
 352     if re.match("^yes$", Options["--compute2DCoords"], re.I):
 353         Compute2DCoords = True
 354     elif re.match("^no$", Options["--compute2DCoords"], re.I):
 355         Compute2DCoords = False
 356     else:
 357         if MiscUtil.CheckFileExt(OptionsInfo["Infile"] ,"smi"):
 358             Compute2DCoords = True
 359     OptionsInfo["Compute2DCoords"]  = Compute2DCoords
 360 
 361     OptionsInfo["FontBold"] = True
 362     if re.match("^no$", Options["--fontBold"], re.I):
 363         OptionsInfo["FontBold"] = False
 364         
 365     HighlightSMARTSPattern = None
 366     if not re.match("^None$", Options["--highlightSMARTS"], re.I):
 367         HighlightSMARTSPattern = Options["--highlightSMARTS"]
 368     OptionsInfo["HighlightSMARTSPattern"]  = HighlightSMARTSPattern
 369     
 370     OptionsInfo["Kekulize"] = True
 371     if re.match("^no$", Options["--kekulize"], re.I):
 372         OptionsInfo["Kekulize"] = False
 373         
 374     SizeValues = Options["--molImageSize"].split(",")
 375     OptionsInfo["MolImageWidth"] = int(SizeValues[0])
 376     OptionsInfo["MolImageHeight"] = int(SizeValues[1])
 377 
 378     OptionsInfo["NumOfMolsPerRow"] = int(Options["--numOfMolsPerRow"])
 379 
 380     OptionsInfo["ShowMolName"] = True
 381     if re.match("^no$", Options["--showMolName"], re.I):
 382         OptionsInfo["ShowMolName"] = False
 383 
 384 def RetrieveOptions():
 385     """Retrieve command line arguments and options"""
 386     
 387     # Get options...
 388     global Options
 389     Options = docopt(__doc__)
 390     
 391     # Set current working directory to the specified directory...
 392     WorkingDir = Options["--workingdir"]
 393     if WorkingDir:
 394         os.chdir(WorkingDir)
 395     
 396     # Handle examples option...
 397     if "--examples" in Options and Options["--examples"]:
 398         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 399         sys.exit(0)
 400 
 401 def ValidateOptions():
 402     """Validate option values"""
 403     
 404     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 405     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi csv tsv txt")
 406 
 407     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 408     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 409     
 410     if not re.match("^None$", Options["--alignmentSMARTS"], re.I):
 411         PatternMol = Chem.MolFromSmarts(Options["--alignmentSMARTS"])
 412         if PatternMol is None:
 413             MiscUtil.PrintError("The value specified, %s, using option \"--alignmentSMARTS\" is not a valid SMARTS: Failed to create pattern molecule" % Options["--alignmentSMARTS"])
 414     
 415     MiscUtil.ValidateOptionNumberValue("--atomLabelFontSize", int(Options["--atomLabelFontSize"]), {">": 0})
 416     MiscUtil.ValidateOptionNumberValue("-b, --bondLineWidth", float(Options["--bondLineWidth"]), {">": 0.0})
 417     
 418     MiscUtil.ValidateOptionTextValue("--compute2DCoords", Options["--compute2DCoords"], "yes no auto")
 419     
 420     MiscUtil.ValidateOptionTextValue("--f, -fontBold", Options["--fontBold"], "yes no")
 421     
 422     if not re.match("^None$", Options["--highlightSMARTS"], re.I):
 423         PatternMol = Chem.MolFromSmarts(Options["--highlightSMARTS"])
 424         if PatternMol is None:
 425             MiscUtil.PrintError("The value specified, %s, using option \"--highlightSMARTS\" is not a valid SMARTS: Failed to create pattern molecule" % Options["--highlightSMARTS"])
 426     
 427     MiscUtil.ValidateOptionTextValue("--kekulize", Options["--kekulize"], "yes no")
 428     
 429     MiscUtil.ValidateOptionNumberValues("-m, --molImageSize", Options["--molImageSize"], 2, ",", "integer", {">": 0})
 430     MiscUtil.ValidateOptionNumberValue("--numOfMolsPerRow", int(Options["--numOfMolsPerRow"]), {">": 0})
 431     
 432     MiscUtil.ValidateOptionTextValue("--showMolName", Options["--showMolName"], "yes no")
 433 
 434 if __name__ == "__main__":
 435     main()
