  29 """
  30 RDKitEnumerateCompoundLibrary.py - Enumerate a virtual compound library
  31 
  32 Usage:
  33     RDKitEnumerateCompoundLibrary.py  [--compute2DCoords <yes or no>] [--infileParams <Name,Value,...>]
  34                                       [--mode <RxnByName or RxnBySMIRKS>] [--outfileParams <Name,Value,...>] [--overwrite]
  35                                       [--prodMolNames <UseReactants or Sequential>] [--rxnName <text>]
  36                                       [--rxnNamesFile <FileName or auto>] [--smirksRxn <text>] [--sanitize <yes or no>]
  37                                       [-w <dir>] -i  <ReactantFile1,...> -o <outfile>
  38     RDKitEnumerateCompoundLibrary.py [--rxnNamesFile <FileName or auto>] -l | --list
  39     RDKitEnumerateCompoundLibrary.py -h | --help | -e | --examples
  40 
  41 Description:
  42     Perform a combinatorial enumeration of a virtual library of molecules for a reaction specified
  43     using a reaction name or SMIRKS pattern and reactant input files.
  44 
  45     The SMIRKS patterns for supported reactions names [ Ref 134 ] are retrieved from file,
  46     ReactionNamesAndSMIRKS.csv, available in MayaChemTools data directory. The current
  47     list of supported reaction names is shown below:
  48 
  49     '1,2,4_triazole_acetohydrazide', '1,2,4_triazole_carboxylic_acid_ester', 3_nitrile_pyridine,
  50     Benzimidazole_derivatives_aldehyde, Benzimidazole_derivatives_carboxylic_acid_ester,
  51     Benzofuran, Benzothiazole, Benzothiophene, Benzoxazole_aromatic_aldehyde,
  52     Benzoxazole_carboxylic_acid, Buchwald_Hartwig, Decarboxylative_coupling, Fischer_indole,
  53     Friedlaender_chinoline, Grignard_alcohol, Grignard_carbonyl, Heck_non_terminal_vinyl,
  54     Heck_terminal_vinyl, Heteroaromatic_nuc_sub, Huisgen_Cu_catalyzed_1,4_subst,
  55     Huisgen_disubst_alkyne, Huisgen_Ru_catalyzed_1,5_subst, Imidazole, Indole, Mitsunobu_imide,
  56     Mitsunobu_phenole, Mitsunobu_sulfonamide, Mitsunobu_tetrazole_1, Mitsunobu_tetrazole_2,
  57     Mitsunobu_tetrazole_3, Mitsunobu_tetrazole_4, N_arylation_heterocycles, Negishi,
  58     Niementowski_quinazoline, Nucl_sub_aromatic_ortho_nitro, Nucl_sub_aromatic_para_nitro,
  59     Oxadiazole, Paal_Knorr_pyrrole, Phthalazinone, Pictet_Spengler, Piperidine_indole,
  60     Pyrazole, Reductive_amination, Schotten_Baumann_amide, Sonogashira, Spiro_chromanone,
  61     Stille, Sulfon_amide, Suzuki, Tetrazole_connect_regioisomer_1, Tetrazole_connect_regioisomer_2,
  62     Tetrazole_terminal, Thiazole, Thiourea, Triaryl_imidazole, Urea, Williamson_ether, Wittig 
  63 
  64     The supported input file formats are: SD (.sdf, .sd), SMILES (.smi, .csv, .tsv, .txt)
  65 
  66     The supported output file formats are:  SD (.sdf, .sd), SMILES (.smi)
  67 
  68 Options:
  69     -c, --compute2DCoords <yes or no>  [default: yes]
  70         Compute 2D coordinates of product molecules before writing them out.
  71     -i, --infiles <ReactantFile1, ReactantFile2...>
  72         Comma delimited list of reactant file names for enumerating a compound library
  73         using reaction SMIRKS. The number of reactant files must match number of
  74         reaction components in reaction SMIRKS. All reactant input files must have
  75         the same format.
  76     --infileParams <Name,Value,...>  [default: auto]
  77         A comma delimited list of parameter name and value pairs for reading
  78         molecules from files. The supported parameter names for different file
  79         formats, along with their default values, are shown below:
  80             
  81             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  82             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  83                 smilesTitleLine,auto,sanitize,yes
  84             
  85         Possible values for smilesDelimiter: space, comma or tab. These parameters apply
  86         to all reactant input files, which must have the same file format.
  87     -e, --examples
  88         Print examples.
  89     -h, --help
  90         Print this help message.
  91     -l, --list
  92         List available reaction names along with corresponding SMIRKS patterns without
  93         performing any enumeration.
  94     -m, --mode <RxnByName or RxnBySMIRKS>  [default: RxnByName]
  95         Indicate whether a reaction is specified by a reaction name or a SMIRKS pattern.
  96         Possible values: RxnByName or RxnBySMIRKS.
  97     -o, --outfile <outfile>
  98         Output file name.
  99     --outfileParams <Name,Value,...>  [default: auto]
 100         A comma delimited list of parameter name and value pairs for writing
 101         molecules to files. The supported parameter names for different file
 102         formats, along with their default values, are shown below:
 103             
 104             SD: kekulize,no
 105             SMILES: kekulize,no,smilesDelimiter,space, smilesIsomeric,yes,
 106                 smilesTitleLine,yes
 107             
 108     -p, --prodMolNames <UseReactants or Sequential>  [default: UseReactants]
 109         Generate names of product molecules using reactant names or assign names in
 110         a sequential order. Possible values: UseReactants or Sequential. Format of
 111         molecule names: UseReactants - <ReactName1>_<ReactName2>..._Prod<Num>;
 112         Sequential - Prod<Num>
 113     --overwrite
 114         Overwrite existing files.
 115     -r, --rxnName <text>
 116         Name of a reaction to use for enumerating a compound library. This option
 117         is only used during 'RxnByName' value of '-m, --mode' option.
 118     --rxnNamesFile <FileName or auto>  [default: auto]
 119         Specify a file name containing data for names of reactions and SMIRKS patterns or
 120         use default file, ReactionNamesAndSMIRKS.csv, available in MayaChemTools data
 121         directory.
 122         
 123         Reactions SMIRKS file format: RxnName,RxnSMIRKS.
 124         
 125         The format of data in local reaction names file must match format of the reaction
 126         SMIRKS file available in MayaChemTools data directory.
 127     -s, --smirksRxn <text>
 128         SMIRKS pattern of a reaction to use for enumerating a compound library. This
 129         option is only used during 'RxnBySMIRKS' value of '-m, --mode' option.
 130     --sanitize <yes or no>  [default: yes]
 131         Sanitize product molecules before writing them out.
 132     -w, --workingdir <dir>
 133         Location of working directory which defaults to the current directory.

 247 def PerformEnumeration():
 248     """Enumerate virutal compound library."""
 249 
 250     ReactantFilesList = OptionsInfo["ReactantFilesList"]
 251     Outfile = OptionsInfo["Outfile"]
 252 
 253     RxnByNameMode = OptionsInfo["RxnByNameMode"]
 254     if RxnByNameMode:
 255         RxnSMIRKSPattern = OptionsInfo["RxnNameSMIRKS"]
 256     else:
 257         RxnSMIRKSPattern = OptionsInfo["SpecifiedSMIRKS"]
 258 
 259     # Set up a reaction and match number of reactants in rxn SMIRKS against number of
 260     # reactant files...
 261     Rxn = AllChem.ReactionFromSmarts(RxnSMIRKSPattern)
 262     RxnReactantsCount = Rxn.GetNumReactantTemplates()
 263 
 264     ReactantFilesList = OptionsInfo["ReactantFilesList"]
 265     ReactantFilesCount = len(ReactantFilesList)
 266     if  ReactantFilesCount != RxnReactantsCount:
 267         MiscUtil.PrintError("The number of specified reactant files, %d, must match number of rectants, %d, in reaction SMIRKS" % (ReactantFilesCount, RxnReactantsCount))
 268         
 269     # Retrieve reactant molecules...
 270     ReactantsMolsList = RetrieveReactantsMolecules()
 271     
 272     # Set up  a molecule writer...
 273     Writer = None
 274     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 275     if Writer is None:
 276         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 277 
 278     MiscUtil.PrintInfo("\nGenerating file %s..." % Outfile)
 279 
 280     # Set up reaction...
 281     ReturnReactants = False
 282     if OptionsInfo["UseReactantNames"]:
 283         ReturnReactants = True
 284     RxnProducts = AllChem.EnumerateLibraryFromReaction(Rxn, ReactantsMolsList, ReturnReactants)
 285 
 286     # Generate product molecules and write them out...
 287     
 288     Compute2DCoords = OptionsInfo["Compute2DCoords"]
 289     Sanitize = OptionsInfo["Sanitize"]
 290     
 291     ProdMolCount = 0
 292     ValidProdMolCount = 0
 293     
 294     if ReturnReactants:
 295         for Products, Reactants in list(RxnProducts):
 296             for ProdMol in Products:
 297                 ProdMolCount += 1
 298 
 299                 # Set product name...
 300                 ReactantMolNames = [ReactantMol.GetProp("_Name") for ReactantMol in Reactants]
 301                 Delimiter = "_"
 302                 ProdMolName = Delimiter.join(ReactantMolNames) + "_Prod%d" % ProdMolCount
 303                 ProdMol.SetProp("_Name", ProdMolName)
 304 
 305                 Status = WriteProductMolecule(Writer, ProdMol, Sanitize, Compute2DCoords)
 306                 if Status:
 307                     ValidProdMolCount += 1
 308     else:
 309         for Products in list(RxnProducts):
 310             for ProdMol in Products:
 311                 ProdMolCount += 1
 312 
 313                 # Set product name...
 314                 ProdMolName = "Prod%d" % ProdMolCount
 315                 ProdMol.SetProp("_Name", ProdMolName)
 316                 
 317                 Status = WriteProductMolecule(Writer, ProdMol, Sanitize, Compute2DCoords)
 318                 if Status:
 319                     ValidProdMolCount += 1
 320 
 321     if Writer is not None:
 322         Writer.close()
 323     
 324     if ValidProdMolCount:
 325         MiscUtil.PrintInfo("\nTotal number of product molecules: %d" % ProdMolCount)
 326         MiscUtil.PrintInfo("Number of valid product molecules: %d" % ValidProdMolCount)
 327         MiscUtil.PrintInfo("Number of ignored product molecules: %d" % (ProdMolCount - ValidProdMolCount))
 328     else:
 329         MiscUtil.PrintInfo("\nThe compound library enumeration failed to generate any product molecules.\nCheck to make sure the reactants specified in input files match their corresponding specifications in reaction SMIRKS and try again.")
 330 
 331 def WriteProductMolecule(Writer, ProdMol, Sanitize, Compute2DCoords):
 332     """Prepare and write out product  molecule."""
 333 
 334     try:
 335         if Sanitize:
 336             Chem.SanitizeMol(ProdMol)
 337     except (RuntimeError, ValueError):
 338         MiscUtil.PrintWarning("Ignoring product molecule: Failed to sanitize...\n")
 339         return False
 340 
 341     try:
 342         if Compute2DCoords:
 343             AllChem.Compute2DCoords(ProdMol)
 344     except (RuntimeError, ValueError):
 345         MiscUtil.PrintWarning("Ignoring product molecule: Failed to compute 2D coordinates...\n")
 346         return False
 347 
 348     Writer.write(ProdMol)
 349 
 350     return True

 352 def RetrieveReactantsMolecules():
 353     """Retrieve reactant molecules from each reactant file and return a list containing lists of molecules
 354     for each reactant file."""
 355 
 356     MiscUtil.PrintInfo("\nProcessing reactant file(s)...")
 357     
 358     ReactantsMolsList = []
 359     ReactantFilesList = OptionsInfo["ReactantFilesList"]
 360     UseReactantNames = OptionsInfo["UseReactantNames"]
 361     ReactantCount = 0
 362     
 363     for FileIndex in range(0, len(ReactantFilesList)):
 364         ReactantCount += 1
 365         ReactantFile = ReactantFilesList[FileIndex]
 366         
 367         MiscUtil.PrintInfo("\nProcessing reactant file: %s..." % ReactantFile)
 368 
 369         Mols  = RDKitUtil.ReadMolecules(ReactantFile, **OptionsInfo["InfileParams"])
 370         
 371         ValidMols = []
 372         MolCount = 0
 373         ValidMolCount = 0
 374         
 375         for Mol in Mols:
 376             MolCount += 1
 377             if Mol is None:
 378                 continue
 379             
 380             if RDKitUtil.IsMolEmpty(Mol):
 381                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
 382                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 383                 continue
 384             
 385             ValidMolCount += 1
 386 
 387             # Check and set mol name...
 388             if UseReactantNames:
 389                 MolName = RDKitUtil.GetMolName(Mol)
 390                 if not len(MolName):
 391                     MolName = "React%dMol%d" % (ReactantCount, MolCount)
 392                     Mol.SetProp("_Name", MolName)
 393                 
 394             ValidMols.append(Mol)
 395 
 396         ReactantsMolsList.append(ValidMols)
 397         
 398         MiscUtil.PrintInfo("Total number of molecules: %d" % MolCount)
 399         MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 400         MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 401     
 402     return ReactantsMolsList

 413 def ProcessSpecifiedReactionName():
 414     """Process and validate specified reaction name."""
 415 
 416     OptionsInfo["RxnNameSMIRKS"] = None
 417     
 418     # Set up a map of valid group rxn names for checking specified rxn names...
 419     CanonicalRxnNameMap = {}
 420     for Name in RxnNamesMap['Names']:
 421         CanonicalRxnNameMap[Name.lower()] = Name
 422     
 423     CanonicalRxnName = OptionsInfo["RxnName"].lower()
 424     if CanonicalRxnName in CanonicalRxnNameMap:
 425         Name = CanonicalRxnNameMap[CanonicalRxnName]
 426         OptionsInfo["RxnNameSMIRKS"] = RxnNamesMap['SMIRKSPattern'][Name]
 427     else:
 428         MiscUtil.PrintError("The rxn name name, %s, specified using \"-r, --rxnName\" option is not a valid name." % (OptionsInfo["RxnName"]))

 442 def RetrieveReactionNamesInfo():
 443     """Retrieve reaction names information."""
 444 
 445     RxnNamesFilePath = OptionsInfo["RxnNamesFile"]
 446     if RxnNamesFilePath is None:
 447         MayaChemToolsDataDir = MiscUtil.GetMayaChemToolsLibDataPath()
 448         RxnNamesFilePath = os.path.join(MayaChemToolsDataDir, "ReactionNamesAndSMIRKS.csv")
 449         
 450     MiscUtil.PrintInfo("\nRetrieving reaction names and SMIRKS patterns from file %s" % (RxnNamesFilePath))
 451     
 452     if not os.path.exists(RxnNamesFilePath):
 453         MiscUtil.PrintError("The reaction names file, %s, doesn't exist.\n" % (RxnNamesFilePath))
 454 
 455     Delimiter = ','
 456     QuoteChar = '"'
 457     HeaderLinePresent = True
 458     RxnLinesWords = MiscUtil.GetTextLinesWords(RxnNamesFilePath, Delimiter, QuoteChar, HeaderLinePresent)
 459     
 460     RxnNamesMap['Names'] = []
 461     RxnNamesMap['SMIRKSPattern'] = {}
 462     
 463     for LineWords in RxnLinesWords:
 464         Name = LineWords[0]
 465         SMIRKSPattern = LineWords[1]
 466 
 467         if Name in RxnNamesMap['SMIRKSPattern']:
 468             MiscUtil.PrintWarning("Ignoring duplicate reaction name: %s..." % Name)
 469         else:
 470             RxnNamesMap['Names'].append(Name)
 471             RxnNamesMap['SMIRKSPattern'][Name] = SMIRKSPattern
 472         
 473     if not len(RxnNamesMap['Names']):
 474         MiscUtil.PrintError("Failed to retrieve any reaction names and SMIRKS patterns...")
 475         
 476     MiscUtil.PrintInfo("Total number of reactions present in reaction names and SMIRKS file: %d" % (len(RxnNamesMap['Names'])))
 477 
 478 def ListReactionNamesInfo():
 479     """List reaction names information"""
 480 
 481     MiscUtil.PrintInfo("\nListing available freaction names and SMIRKS patterns...")
 482     MiscUtil.PrintInfo("\nReactionName\tSMIRKSPattern")
 483     
 484     for Name in sorted(RxnNamesMap['Names']):
 485         SMIRKSPattern = RxnNamesMap['SMIRKSPattern'][Name]
 486         MiscUtil.PrintInfo("%s\t%s" % (Name, SMIRKSPattern))
 487 
 488     MiscUtil.PrintInfo("")

 490 def ProcessOptions():
 491     """Process and validate command line arguments and options"""
 492     
 493     MiscUtil.PrintInfo("Processing options...")
 494     
 495     # Validate options...
 496     ValidateOptions()
 497     
 498     Compute2DCoords = True
 499     if not re.match("^yes$", Options["--compute2DCoords"], re.I):
 500         Compute2DCoords = False
 501     OptionsInfo["Compute2DCoords"]  = Compute2DCoords
 502 
 503     OptionsInfo["Mode"] = Options["--mode"]
 504     RxnByNameMode = True
 505     if not re.match("^RxnByName$", Options["--mode"], re.I):
 506         RxnByNameMode = False
 507     OptionsInfo["RxnByNameMode"] = RxnByNameMode
 508 
 509     OptionsInfo["ProdMolNamesMode"] = Options["--prodMolNames"]
 510     UseReactantNames = False
 511     if re.match("^UseReactants$", Options["--prodMolNames"], re.I):
 512         UseReactantNames = True
 513     OptionsInfo["UseReactantNames"] = UseReactantNames
 514     
 515     OptionsInfo["RxnName"] = Options["--rxnName"]
 516     OptionsInfo["RxnNameSMIRKS"] = None
 517     if OptionsInfo["RxnByNameMode"]:
 518         if not Options["--rxnName"]:
 519             MiscUtil.PrintError("No rxn name specified using \"-r, --rxnName\" option during \"RxnByName\" value of \"-m, --mode\" option")
 520 
 521     OptionsInfo["RxnNamesFile"] = None
 522     if not re.match("^auto$", Options["--rxnNamesFile"], re.I):
 523         OptionsInfo["RxnNamesFile"] = Options["--rxnNamesFile"]
 524 
 525     ReactantFiles = re.sub(" ", "", Options["--infiles"])
 526     ReactantFilesList = []
 527     ReactantFilesList = ReactantFiles.split(",")
 528     OptionsInfo["ReactantFiles"] = ReactantFiles
 529     OptionsInfo["ReactantFilesList"] = ReactantFilesList
 530 
 531     OptionsInfo["SpecifiedSMIRKS"] = Options["--smirksRxn"]
 532     if not OptionsInfo["RxnByNameMode"]:
 533         if not Options["--smirksRxn"]:
 534             MiscUtil.PrintError("No rxn SMIRKS pattern specified using \"-r, --rxnName\" option during \"RxnByName\" value of \"-m, --mode\" option")
 535     
 536     OptionsInfo["Outfile"] = Options["--outfile"]
 537     OptionsInfo["Overwrite"] = Options["--overwrite"]
 538 
 539     # Use first reactant file as input file as all input files have the same format...
 540     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], ReactantFilesList[0])
 541 
 542     # No need to pass any input or output file name due to absence of any auto parameter...
 543     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 544     
 545     Sanitize = True
 546     if not re.match("^yes$", Options["--sanitize"], re.I):
 547         Sanitize = False
 548     OptionsInfo["Sanitize"]  = Sanitize

 558     WorkingDir = Options["--workingdir"]

 561

 572 def ValidateOptions():
 573     """Validate option values"""
 574     
 575     MiscUtil.ValidateOptionTextValue("--compute2DCoords", Options["--compute2DCoords"], "yes no")
 576     
 577     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "RxnByName RxnBySMIRKS")
 578     MiscUtil.ValidateOptionTextValue("-p, --prodMolNames", Options["--prodMolNames"], "UseReactants Sequential")
 579     
 580     if not re.match("^auto$", Options["--rxnNamesFile"], re.I):
 581         MiscUtil.ValidateOptionFilePath("--rxnNamesFile", Options["--rxnNamesFile"])
 582 
 583     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 584     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 585     
 586     ReactantFiles = re.sub(" ", "", Options["--infiles"])
 587     if not ReactantFiles:
 588         MiscUtil.PrintError("No reactant files specified for \"-i, --infiles\" option")
 589 
 590     # Validate file extensions...
 591     for ReactantFile in ReactantFiles.split(","):
 592         MiscUtil.ValidateOptionFilePath("-i, --infiles", ReactantFile)
 593         MiscUtil.ValidateOptionFileExt("-i, --infiles", ReactantFile, "sdf sd smi csv tsv txt")
 594         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infiles", ReactantFile, "-o, --outfile", Options["--outfile"])
 595         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infiles", ReactantFile, "-o, --outfile", Options["--outfile"])
 596         
 597     # Match file formats...
 598     FirstFile = True
 599     FirstFileFormat = ""
 600     for ReactantFile in ReactantFiles.split(","):
 601         FileFormat = ""
 602         if MiscUtil.CheckFileExt(ReactantFile, "sdf sd"):
 603             FileFormat = "SD"
 604         elif MiscUtil.CheckFileExt(ReactantFile, "smi csv tsv txt"):
 605             FileFormat = "SMILES"
 606         else:
 607             MiscUtil.PrintError("The file name specified , %s, for option \"-i, --infiles\" is not valid. Supported file formats: sdf sd smi csv tsv txt\n" % ReactantFile)
 608             
 609         if FirstFile:
 610             FirstFile = False
 611             FirstFileFormat = FileFormat
 612             continue
 613         
 614         if not re.match("^%s$" % FirstFileFormat, FileFormat, re.IGNORECASE):
 615             MiscUtil.PrintError("All reactant file names -  %s - specified using option \"-i, --infiles\" must have the same file format.\n" % ReactantFiles)
 616             
 617 
 618     MiscUtil.ValidateOptionTextValue("--sanitize", Options["--sanitize"], "yes no")
 619
