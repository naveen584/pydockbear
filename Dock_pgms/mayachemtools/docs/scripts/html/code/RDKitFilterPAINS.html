  29 """
  30 RDKitFilterPAINS.py - Filter PAINS molecules
  31 
  32 Usage:
  33     RDKitFilterPAINS.py  [--infileParams <Name,Value,...>] [--mode <filter or count>]
  34                          [ --outfileParams <Name,Value,...> ] [--painsMode <All, A, B or C>]
  35                          [--negate <yes or no>] [--overwrite] [-w <dir>] -i <infile> -o <outfile>
  36     RDKitFilterPAINS.py -h | --help | -e | --examples
  37 
  38 Description:
  39     Filter Pan-assay Interference molecules (PAINS) [ Ref 130 - 131 ] from an input
  40     file by performing a substructure search using SMARTS pattern specified in
  41     MAYACHEMTOOLS/lib/data/PAINSFilter.csv file and write out appropriate
  42     molecules to an output file or simply count the number of filtered molecules.
  43 
  44     The supported input file formats are: SD (.sdf, .sd), SMILES (.smi, .csv,
  45      .tsv, .txt)
  46 
  47     The supported output file formats are: SD (.sdf, .sd), SMILES (.smi)
  48 
  49 Options:
  50     -e, --examples
  51         Print examples.
  52     -h, --help
  53         Print this help message.
  54     -i, --infile <infile>
  55         Input file name.
  56     --infileParams <Name,Value,...>  [default: auto]
  57         A comma delimited list of parameter name and value pairs for reading
  58         molecules from files. The supported parameter names for different file
  59         formats, along with their default values, are shown below:
  60             
  61             SD: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  62             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  63                 smilesTitleLine,auto,sanitize,yes
  64             
  65         Possible values for smilesDelimiter: space, comma or tab.
  66     -m, --mode <filter or count>  [default: filter]
  67         Specify whether to filter the matched molecules and write out the rest of the 
  68         molecules to an outfile or simply count the number of matched molecules
  69         marked for filtering.
  70     -n, --negate <yes or no>  [default: no]
  71         Specify whether to filter molecules not matching the PAINS filters specified by
  72         SMARTS patterns.
  73     -o, --outfile <outfile>
  74         Output file name.
  75     --outfileParams <Name,Value,...>  [default: auto]
  76         A comma delimited list of parameter name and value pairs for writing
  77         molecules to files. The supported parameter names for different file
  78         formats, along with their default values, are shown below:
  79             
  80             SD: compute2DCoords,auto,kekulize,no
  81             SMILES: kekulize,no,smilesDelimiter,space, smilesIsomeric,yes,
  82                 smilesTitleLine,yes
  83             
  84         Default value for compute2DCoords: yes for SMILES input file; no for all other
  85         file types.
  86     --overwrite
  87         Overwrite existing files.
  88     -p, --painsMode <All, A, B, or C>  [default: All]
  89         Specify PAINS filter family type to used for filtering molecules. 
  90     -w, --workingdir <dir>
  91         Location of working directory which defaults to the current directory.
  92 
  93 Examples:
  94     To count the number of molecules not containing any substructure corresponding to
  95     PAINS SMARTS patterns and write out a SMILES file, type: 
  96 
  97         % RDKitFilterPAINS.py -i Sample.smi -o SampleOut.smi
  98 
  99     To only count the number of molecules not containing any substructure corresponding
 100     to PAINS SMARTS patterns without writing out any file file, type: 
 101 
 102         % RDKitFilterPAINS.py -m count -i Sample.sdf -o SampleOut.smi

 191 def PerformFiltering():
 192     """Filter molecules using SMARTS specified in PAINS filter file."""
 193     
 194     Infile = OptionsInfo["Infile"]
 195     Outfile = OptionsInfo["Outfile"]
 196     
 197     CountMode = OptionsInfo["CountMode"]
 198     NegateMatch = OptionsInfo["NegateMatch"]
 199     PAINSMode = OptionsInfo["PAINSMode"]
 200 
 201     # Setup PAINS patterns and pattern mols...
 202     PAINSPatterns = RetrievePAINSPatterns(PAINSMode)
 203     PAINSPatternMols = SetupPAINSPatternMols(PAINSPatterns)
 204     
 205     # Setup a molecule reader...
 206     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 207     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 208     
 209     # Set up a molecule writer...
 210     Writer = None
 211     if not CountMode:
 212         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 213         if Writer is None:
 214             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 215         MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 216 
 217     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 218     
 219     # Process molecules...
 220     MolCount = 0
 221     ValidMolCount = 0
 222     FilteredCount = 0
 223 
 224     for Mol in Mols:
 225         MolCount += 1
 226         
 227         if Mol is None:
 228             continue
 229         
 230         if RDKitUtil.IsMolEmpty(Mol):
 231             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 232             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 233             continue
 234         
 235         ValidMolCount += 1
 236         
 237         MolMatched = DoesMoleculeContainsPAINSPattern(Mol, PAINSPatternMols)
 238         if MolMatched == NegateMatch:
 239             FilteredCount += 1
 240             if not CountMode:
 241                 if Compute2DCoords:
 242                     AllChem.Compute2DCoords(Mol)
 243                 Writer.write(Mol)
 244     
 245     if Writer is not None:
 246         Writer.close()
 247     
 248     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 249     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 250     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 251 
 252     MiscUtil.PrintInfo("\nTotal number of filtered molecules: %d" % FilteredCount)
 253 
 254 def DoesMoleculeContainsPAINSPattern(Mol, PAINSPatternMols):
 255     """Check presence of PAINS pattern in the molecule"""
 256 
 257     MolMatched = False
 258     
 259     for PatternMol in PAINSPatternMols:
 260         if Mol.HasSubstructMatch(PatternMol, useChirality = True):
 261             MolMatched = True
 262             break
 263         
 264     return MolMatched
 265     
 266 def RetrievePAINSPatterns(PAINSFilterMode):
 267     """Retrieve PAINS patterns for specified PAINS mode"""
 268 
 269     MayaChemToolsDataDir = MiscUtil.GetMayaChemToolsLibDataPath()
 270     PAINSFiltersFilePath = os.path.join(MayaChemToolsDataDir, "PAINSFilters.csv")
 271     
 272     MiscUtil.PrintInfo("\nRetrieving PAINS SMARTS patterns for PAINS filter type, %s, from file %s" % (PAINSFilterMode, PAINSFiltersFilePath))
 273 
 274     if not os.path.exists(PAINSFiltersFilePath):
 275         MiscUtil.PrintError("The PAINS filters file, %s, doesn't exist.\n" % (PAINSFiltersFilePath))
 276         
 277     FilterFile = open(PAINSFiltersFilePath, "r")
 278     if FilterFile is None:
 279         MiscUtil.PrintError("Couldn't open PAINS filter file: %s.\n" % (PAINSFiltersFilePath))
 280 
 281     # Collect all PAINS filter lines...
 282     HeaderLine = True
 283     FiltersLines = []
 284     for Line in FilterFile:
 285         Line = Line.strip()
 286         # Ignore comments...
 287         if re.match("^#", Line, re.I):
 288             continue
 289         # Ignore header line...
 290         if HeaderLine:
 291             HeaderLine = False
 292             continue
 293         FiltersLines.append(Line)
 294         
 295     # Process PAINS filter lines using csv reader...
 296     SMARTSPatterns = []
 297     
 298     FiltersReader = csv.reader(FiltersLines, delimiter=',', quotechar='"')
 299     for LineWords in FiltersReader:
 300         FilterType = LineWords[0]
 301         ID = LineWords[1]
 302         SMARTS = LineWords[2]
 303 
 304         if re.match("^All$", PAINSFilterMode, re.I) or FilterType.lower() == PAINSFilterMode.lower():
 305             SMARTSPatterns.append(SMARTS)
 306         
 307     FilterFile.close()
 308 
 309     MiscUtil.PrintInfo("Total number of PAINS SMARTS patterns: %d" % (len(SMARTSPatterns)))
 310     
 311     return SMARTSPatterns
 312 
 313 def SetupPAINSPatternMols(PAINSPatterns):
 314     """Set up PAINS pattern mols for substructure search"""
 315 
 316     MiscUtil.PrintInfo("\nSetting up PAINS pattern molecules for performins substructure search...")
 317     PatternMols = []
 318     for Pattern in PAINSPatterns:
 319         PatternMol = Chem.MolFromSmarts(Pattern)
 320         PatternMols.append(PatternMol)
 321         
 322     return PatternMols

 324 def ProcessOptions():
 325     """Process and validate command line arguments and options"""
 326     
 327     MiscUtil.PrintInfo("Processing options...")
 328     
 329     # Validate options...
 330     ValidateOptions()
 331     
 332     OptionsInfo["Infile"] = Options["--infile"]
 333     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 334     
 335     OptionsInfo["Outfile"] = Options["--outfile"]
 336     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 337     
 338     OptionsInfo["Overwrite"] = Options["--overwrite"]
 339 
 340     OptionsInfo["CountMode"] = False
 341     if re.match("^count$", Options["--mode"], re.I):
 342         OptionsInfo["CountMode"] = True
 343     
 344     OptionsInfo["NegateMatch"] = False
 345     if re.match("^yes$", Options["--negate"], re.I):
 346         OptionsInfo["NegateMatch"] = True
 347         
 348     OptionsInfo["PAINSMode"] = Options["--painsMode"]
 349     
 350 def RetrieveOptions():
 351     """Retrieve command line arguments and options"""
 352     
 353     # Get options...
 354     global Options
 355     Options = docopt(__doc__)
 356     
 357     # Set current working directory to the specified directory...
 358     WorkingDir = Options["--workingdir"]
 359     if WorkingDir:
 360         os.chdir(WorkingDir)
 361     
 362     # Handle examples option...
 363     if "--examples" in Options and Options["--examples"]:
 364         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 365         sys.exit(0)
 366 
 367 def ValidateOptions():
 368     """Validate option values"""
 369     
 370     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 371     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 372     
 373     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 374     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 375     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 376 
 377     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "filter count")
 378     if re.match("^filter$", Options["--mode"], re.I):
 379         if not Options["--outfile"]:
 380             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"filter\" value of \"-m, --mode\" option")
 381         
 382     MiscUtil.ValidateOptionTextValue("-n, --negate", Options["--negate"], "yes no")
 383     
 384     MiscUtil.ValidateOptionTextValue("-p, --painsMode", Options["--painsMode"], "All A B C")
 385 
 386 if __name__ == "__main__":
 387     main()
