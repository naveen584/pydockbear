  29 """
  30 RDKitGenerateConformers.py - Generate molecular conformations
  31 
  32 Usage:
  33     RDKitGenerateConformers.py [--alignConformers <yes or no>] [--addHydrogens <yes or no>]
  34                                [--conformerGenerator <SDG, ETDG, KDG, ETKDG>] [--embedRMSDCutoff <number>]
  35                                [--enforceChirality <yes or no>] [--energyRMSDCutoff <number>]
  36                                [--energyWindow <number> ]  [--forceField <UFF, MMFF, None>]
  37                                [--infileParams <Name,Value,...>] [--maxConfs <number>]
  38                                [--maxIters <number>]  [ --outfileParams <Name,Value,...> ]  [--overwrite]
  39                                [ --removeHydrogens <yes or no>] [--randomSeed <number>] [-w <dir>] -i <infile> -o <outfile> 
  40     RDKitGenerateConformers.py -h | --help | -e | --examples
  41 
  42 Description:
  43     Generate molecular conformations using a combination of distance geometry and
  44     forcefield minimization. The forcefield minimization may be skipped to only generate
  45     conformations by available distance geometry based methodologies.
  46 
  47     The supported input file formats are: Mol (.mol), SD (.sdf, .sd), SMILES (.smi,
  48     .csv, .tcsv .txt)
  49 
  50     The supported output file format are: SD (.sdf, .sd)
  51 
  52 Options:
  53     -a, --addHydrogens <yes or no>  [default: yes]
  54         Add hydrogens before minimization.
  55     --alignConformers <yes or no>  [default: yes]
  56         Align conformers for each molecule.
  57     -c, --conformerGenerator <SDG, ETDG, KDG, ETKDG>  [default: ETKDG]
  58         Conformation generation methodology for generating initial 3D coordinates of a
  59         molecule. Possible values: Standard Distance Geometry, (SDG), Experimental
  60         Torsion-angle preference with Distance Geometry (ETDG)  [Ref 129] , basic Knowledge-terms
  61         with Distance Geometry (KDG), and Experimental Torsion-angle preference along
  62         with basic Knowledge-terms and Distance Geometry (ETKDG). 
  63     --embedRMSDCutoff <number>  [default: none]
  64         RMSD cutoff for retaining conformations after embedding and before energy minimization.
  65         All embedded conformations are kept by default. Otherwise, only those conformations
  66         which are different from each other by the specified RMSD cutoff are kept. The first
  67         embedded conformation is always retained.
  68     --enforceChirality <yes or no>  [default: Yes]
  69         Enforce chirality for defined chiral centers.
  70     --energyRMSDCutoff <number>  [default: none]
  71         RMSD cutoff for retaining conformations after energy minimization. By default,
  72         all minimized conformations with in the specified energy window from the lowest energy
  73         conformation are kept. Otherwise, only those conformations which are different from
  74         the lowest energy conformation by the specified RMSD cutoff and are with in the
  75         specified energy window are kept. The lowest energy conformation is always retained.
  76     --energyWindow <number>  [default: 20]
  77         Energy window in kcal/mol for selecting conformers. This option is ignored during
  78         'None' value of '-f, --forcefield' option.
  79     -e, --examples
  80         Print examples.
  81     -f, --forceField <UFF, MMFF, None>  [default: MMFF]
  82         Forcefield method to use for energy minimization. Possible values: Universal Force
  83         Field (UFF) [Ref 81],  Merck Molecular Mechanics Force Field (MMFF) [Ref 83-87] or
  84         None.
  85     -h, --help
  86         Print this help message.
  87     -i, --infile <infile>
  88         Input file name.
  89     --infileParams <Name,Value,...>  [default: auto]
  90         A comma delimited list of parameter name and value pairs for reading
  91         molecules from files. The supported parameter names for different file
  92         formats, along with their default values, are shown below:
  93             
  94             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  95             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  96                 smilesTitleLine,auto,sanitize,yes
  97             
  98         Possible values for smilesDelimiter: space, comma or tab.
  99     --maxConfs <number>  [default: auto]
 100         Maximum number of conformations to generate for each molecule by conformation
 101         generation methodology. The conformations are minimized using the specified
 102         forcefield as needed and written to the output file. The default value for maximum
 103         number of conformations is dependent on the number of rotatable bonds in molecules:
 104         RotBonds <= 5, maxConfs = 100; RotBonds >=6 and <= 10, MaxConfs = 200;
 105         RotBonds >= 11, maxConfs = 300
 106     --maxIters <number>  [default: 250]
 107         Maximum number of iterations to perform for each molecule during forcefield
 108         minimization.
 109     -o, --outfile <outfile>
 110         Output file name.
 111     --outfileParams <Name,Value,...>  [default: auto]
 112         A comma delimited list of parameter name and value pairs for writing
 113         molecules to files. The supported parameter names for different file
 114         formats, along with their default values, are shown below:
 115             
 116             SD: kekulize,no
 117             
 118     --overwrite
 119         Overwrite existing files.
 120     -r, --removeHydrogens <yes or no>  [default: Yes]
 121         Remove hydrogens after minimization.
 122     --randomSeed <number>  [default: auto]
 123         Seed for the random number generator for reproducing 3D coordinates.
 124         Default is to use a random seed.
 125     -w, --workingdir <dir>
 126         Location of working directory which defaults to the current directory.
 127 
 128 Examples:
 129     To generate conformers using Experimental Torsion-angle preference along
 130     with basic Knowledge-terms and Distance Geometry (ETKDG) followed by
 131     MMFF minimization with automatic determination of maximum number of
 132     conformers for each molecule and write out a SD file, type:
 133 
 134         % RDKitGenerateConformers.py  -i Sample.smi -o SampleOut.sdf
 135 
 136     To generate up to 150 conformers for each molecule using ETKDG and UFF forcefield
 137     minimization along with conformers within 25 kcal/mol energy window and write out a
 138     SD file, type:
 139 
 140         % RDKitGenerateConformers.py  --energyWindow 25 -f UFF --maxConfs 150
 141           -i Sample.smi -o SampleOut.sdf
 142 
 143     To generate up to 50 conformers for each molecule using KDG without any forcefield
 144     minimization and alignment of conformers and write out a SD file, type:
 145 
 146         % RDKitGenerateConformers.py  -f none --maxConfs 50 --alignConformers no
 147           -i Sample.sdf -o SampleOut.sdf
 148 
 149     To generate up to 50 conformers using SDG without any forcefield minimization
 150     and alignment of conformers for molecules in a  CSV SMILES file, SMILES strings
 151     in column 1, name in column 2, and write out a SD file, type:
 152 
 153         % RDKitGenerateConformers.py  --maxConfs 50  --maxIters 50 -c SDG
 154           --alignConformers no -f none --infileParams "smilesDelimiter,comma,
 155           smilesTitleLine,yes, smilesColumn,1,smilesNameColumn,2"
 156           -i SampleSMILES.csv -o SampleOut.sdf

 232 def GenerateConformers():
 233     """Generate conformers."""
 234     
 235     Infile = OptionsInfo["Infile"]
 236     Outfile = OptionsInfo["Outfile"]
 237     
 238     # Setup a molecule reader...
 239     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 240     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 241     
 242     # Set up a molecule writer...
 243     Writer = None
 244     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 245     if Writer is None:
 246         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 247     MiscUtil.PrintInfo("Generating file %s...\n" % Outfile)
 248 
 249     # Process molecules...
 250     MolCount = 0
 251     ValidMolCount = 0
 252     ConfGenFailedCount = 0
 253 
 254     SkipMinimization = OptionsInfo["SkipForceFieldMinimization"]
 255     
 256     if SkipMinimization:
 257         MiscUtil.PrintInfo("Generating conformers without performing energy minimization...\n")
 258     else:
 259         MiscUtil.PrintInfo("Generating conformers and performing energy minimization...\n")
 260         
 261     for Mol in Mols:
 262         MolCount += 1
 263         
 264         if Mol is None:
 265             continue
 266         
 267         if RDKitUtil.IsMolEmpty(Mol):
 268             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 269             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 270             continue
 271         
 272         ValidMolCount += 1
 273 
 274         if SkipMinimization:
 275             Status = GenerateMolConformers(Mol, MolCount, Writer)
 276         else:
 277             Status = GenerateAndMinimizeMolConformers(Mol, MolCount, Writer)
 278         
 279         if not Status:
 280             ConfGenFailedCount += 1
 281     
 282     if Writer is not None:
 283         Writer.close()
 284     
 285     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 286     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 287     MiscUtil.PrintInfo("Number of molecules failed during conformation generation or minimization: %d" % ConfGenFailedCount)
 288     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + ConfGenFailedCount))
 289 
 290 def GenerateMolConformers(Mol, MolCount, Writer):
 291     "Generate conformers for a molecule and write them out."
 292     
 293     MolName = RDKitUtil.GetMolName(Mol, MolCount)
 294     
 295     ConfIDs = EmbedMolecule(Mol)
 296     if not len(ConfIDs):
 297         MiscUtil.PrintWarning("Conformation generation couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 298         return False
 299     
 300     if OptionsInfo["AlignConformers"]:
 301         AllChem.AlignMolConformers(Mol)
 302     
 303     
 304     # Write out the conformers...
 305     for ConfID in ConfIDs:
 306         SetConfMolName(Mol, MolName, ConfID)
 307         Writer.write(Mol, confId = ConfID)
 308     
 309     MiscUtil.PrintInfo("\nNumber of conformations written for %s: %d" % (MolName, len(ConfIDs)))
 310     
 311     return True
 312     
 313 def GenerateAndMinimizeMolConformers(Mol, MolCount, Writer):
 314     "Generate and mininize conformers for a molecule and write them out."
 315     
 316     ConfIDs = EmbedMolecule(Mol)
 317     if not len(ConfIDs):
 318         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 319         MiscUtil.PrintWarning("Conformation generation couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 320         return False
 321 
 322     if  OptionsInfo["AddHydrogens"]:
 323         Mol = Chem.AddHs(Mol)
 324 
 325     CalcEnergyMap = {}
 326     for ConfID in ConfIDs:
 327         try:
 328             if OptionsInfo["UseUFF"]:
 329                 Status = AllChem.UFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 330             elif OptionsInfo["UseMMFF"]:
 331                 Status = AllChem.MMFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 332             else:
 333                 MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 334         except RuntimeError as ErrMsg:
 335             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 336             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 337             return False
 338         
 339         Energy = GetConformerEnergy(Mol, ConfID)
 340 
 341         if Status != 0:
 342             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 343             MiscUtil.PrintWarning("Minimization failed to converge for conformation number %d of molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (ConfID, MolName, OptionsInfo["MaxIters"]))
 344             
 345         CalcEnergyMap[ConfID] = Energy
 346 
 347     if  OptionsInfo["RemoveHydrogens"]:
 348         Mol = Chem.RemoveHs(Mol)
 349 
 350     # Align molecules after minimization...
 351     if OptionsInfo["AlignConformers"]:
 352         AllChem.AlignMolConformers(Mol)
 353     
 354     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
 355     
 356     MinEnergyConfID = SortedConfIDs[0]
 357     MinConfEnergy = CalcEnergyMap[MinEnergyConfID]
 358 
 359     EnergyWindow = OptionsInfo["EnergyWindow"]
 360     MolName = RDKitUtil.GetMolName(Mol, MolCount)
 361 
 362     EnergyRMSDCutoff = OptionsInfo["EnergyRMSDCutoff"]
 363     ApplyEnergyRMSDCutoff = False
 364     if EnergyRMSDCutoff > 0:
 365         ApplyEnergyRMSDCutoff = True
 366 
 367     # Calculate RMSD values for conformers...
 368     PreAligned = False
 369     if OptionsInfo["AlignConformers"]:
 370         PreAligned = True
 371     
 372     CalcRMSDMap = {}
 373     if ApplyEnergyRMSDCutoff:
 374         for ConfID in SortedConfIDs:
 375             RMSD = AllChem.GetConformerRMS(Mol, MinEnergyConfID, ConfID, prealigned=PreAligned)
 376             CalcRMSDMap[ConfID] = RMSD
 377     
 378     # Write out the conformers with in the specified energy window  from the lowest
 379     # energy conformation along with applying RMSD cutoff as needed...
 380     #
 381     ConfCount = 0
 382     IgnoredByEnergyConfCount = 0
 383     IgnoredByRMSDConfCount = 0
 384     
 385     FirstConf = True
 386     
 387     for ConfID in SortedConfIDs:
 388         if FirstConf:
 389             FirstConf = False
 390             # Write it out...
 391             ConfCount += 1
 392             SetConfMolName(Mol, MolName, ConfID)
 393             Writer.write(Mol, confId = ConfID)
 394             continue
 395             
 396         ConfEnergyDiff = abs(CalcEnergyMap[ConfID] - MinConfEnergy )
 397         if  ConfEnergyDiff > EnergyWindow:
 398             IgnoredByEnergyConfCount += 1
 399             continue
 400         
 401         if ApplyEnergyRMSDCutoff:
 402             if CalcRMSDMap[ConfID] < EnergyRMSDCutoff:
 403                 IgnoredByRMSDConfCount += 1
 404                 continue
 405 
 406         # Write it out...
 407         ConfCount += 1
 408         SetConfMolName(Mol, MolName, ConfID)
 409         Writer.write(Mol, confId = ConfID)
 410     
 411     MiscUtil.PrintInfo("\nTotal Number of conformations written for %s: %d" % (MolName, ConfCount))
 412     MiscUtil.PrintInfo("Number of conformations ignored due to energy window cutoff: %d" % (IgnoredByEnergyConfCount))
 413     if ApplyEnergyRMSDCutoff:
 414         MiscUtil.PrintInfo("Number of conformations ignored due to energy RMSD cutoff:  %d" % (IgnoredByRMSDConfCount))
 415     
 416     return True
 417     
 418 def GetConformerEnergy(Mol, ConfID):
 419     "Calculate conformer energy"
 420 
 421     Energy = 0.0
 422     if OptionsInfo["UseUFF"]:
 423         UFFMoleculeForcefield = AllChem.UFFGetMoleculeForceField(Mol, confId = ConfID)
 424         Energy = UFFMoleculeForcefield.CalcEnergy()
 425     elif OptionsInfo["UseMMFF"]:
 426         MMFFMoleculeProperties = AllChem.MMFFGetMoleculeProperties(Mol)
 427         MMFFMoleculeForcefield = AllChem.MMFFGetMoleculeForceField(Mol, MMFFMoleculeProperties, confId = ConfID)
 428         Energy = MMFFMoleculeForcefield.CalcEnergy()
 429     else:
 430         MiscUtil.PrintError("Couldn't retrieve conformer energy: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 431     
 432     return Energy
 433     
 434 def EmbedMolecule(Mol):
 435     "Embed conformations"
 436     
 437     ConfIDs = []
 438     
 439     # Figure out the number of conformations to embded...
 440     if re.match("^Auto$", OptionsInfo["MaxConfs"], re.I):
 441         NumOfRotBonds = Descriptors.NumRotatableBonds(Mol)
 442         if NumOfRotBonds <= 5:
 443             MaxConfs = 100
 444         elif NumOfRotBonds >=6 and NumOfRotBonds <= 10:
 445             MaxConfs = 200
 446         else:
 447             MaxConfs = 300
 448     else:
 449         MaxConfs = int(OptionsInfo["MaxConfs"])
 450         
 451     RandomSeed = OptionsInfo["RandomSeed"]
 452     EnforceChirality = OptionsInfo["EnforceChirality"]
 453     UseExpTorsionAnglePrefs = OptionsInfo["UseExpTorsionAnglePrefs"]
 454     UseBasicKnowledge = OptionsInfo["UseBasicKnowledge"]
 455     EmbedRMSDCutoff = OptionsInfo["EmbedRMSDCutoff"]
 456     
 457     ConfIDs = AllChem.EmbedMultipleConfs(Mol, numConfs = MaxConfs, randomSeed = RandomSeed, pruneRmsThresh = EmbedRMSDCutoff, enforceChirality = EnforceChirality, useExpTorsionAnglePrefs = UseExpTorsionAnglePrefs, useBasicKnowledge = UseBasicKnowledge)
 458 
 459     return ConfIDs

 461 def SetConfMolName(Mol, MolName, ConfCount):
 462     """Set conf mol name"""
 463 
 464     ConfName = "%s_Conf%d" % (MolName, ConfCount)
 465     Mol.SetProp("_Name", ConfName)

 467 def ProcessOptions():
 468     """Process and validate command line arguments and options"""
 469     
 470     MiscUtil.PrintInfo("Processing options...")
 471     
 472     # Validate options...
 473     ValidateOptions()
 474     
 475     OptionsInfo["Infile"] = Options["--infile"]
 476     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 477     
 478     OptionsInfo["Outfile"] = Options["--outfile"]
 479     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 480     
 481     OptionsInfo["Overwrite"] = Options["--overwrite"]
 482 
 483     OptionsInfo["AddHydrogens"] = True
 484     if re.match("^no$", Options["--addHydrogens"], re.I):
 485         OptionsInfo["AddHydrogens"] = False
 486 
 487     OptionsInfo["AlignConformers"] = True
 488     if re.match("^no$", Options["--alignConformers"], re.I):
 489         OptionsInfo["AlignConformers"] = False
 490 
 491     if re.match("^ETDG$", Options["--conformerGenerator"], re.I):
 492         ConformerGenerator = "ETDG"
 493         UseExpTorsionAnglePrefs = True
 494         UseBasicKnowledge = False
 495     elif re.match("^KDG$", Options["--conformerGenerator"], re.I):
 496         ConformerGenerator = "KDG"
 497         UseExpTorsionAnglePrefs = False
 498         UseBasicKnowledge = True
 499     elif re.match("^ETKDG$", Options["--conformerGenerator"], re.I):
 500         ConformerGenerator = "ETKDG"
 501         UseExpTorsionAnglePrefs = True
 502         UseBasicKnowledge = True
 503     elif re.match("^SDG$", Options["--conformerGenerator"], re.I):
 504         ConformerGenerator = "SDG"
 505         UseExpTorsionAnglePrefs = False
 506         UseBasicKnowledge = False
 507     
 508     OptionsInfo["ConformerGenerator"] = ConformerGenerator
 509     OptionsInfo["UseExpTorsionAnglePrefs"] = UseExpTorsionAnglePrefs
 510     OptionsInfo["UseBasicKnowledge"] = UseBasicKnowledge
 511 
 512     if re.match("^UFF$", Options["--forceField"], re.I):
 513         ForceField = "UFF"
 514         UseUFF = True
 515         UseMMFF = False
 516         SkipForceFieldMinimization = False
 517     elif re.match("^MMFF$", Options["--forceField"], re.I):
 518         ForceField = "MMFF"
 519         UseUFF = False
 520         UseMMFF = True
 521         SkipForceFieldMinimization = False
 522     else:
 523         ForceField = "None"
 524         UseUFF = False
 525         UseMMFF = False
 526         SkipForceFieldMinimization = True
 527         
 528     OptionsInfo["SkipForceFieldMinimization"] = SkipForceFieldMinimization
 529     OptionsInfo["ForceField"] = ForceField
 530     OptionsInfo["UseMMFF"] = UseMMFF
 531     OptionsInfo["UseUFF"] = UseUFF
 532         
 533     OptionsInfo["EnforceChirality"] = True
 534     if re.match("^no$", Options["--enforceChirality"], re.I):
 535         OptionsInfo["EnforceChirality"] = False
 536     
 537     OptionsInfo["EnergyWindow"] = float(Options["--energyWindow"])
 538     
 539     EmbedRMSDCutoff = -1.0
 540     if not re.match("^none$", Options["--embedRMSDCutoff"], re.I):
 541         EmbedRMSDCutoff = float(Options["--embedRMSDCutoff"])
 542     OptionsInfo["EmbedRMSDCutoff"] = EmbedRMSDCutoff
 543     
 544     EnergyRMSDCutoff = -1.0
 545     if not re.match("^none$", Options["--energyRMSDCutoff"], re.I):
 546         EnergyRMSDCutoff = float(Options["--energyRMSDCutoff"])
 547     OptionsInfo["EnergyRMSDCutoff"] = EnergyRMSDCutoff
 548     
 549     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 550     OptionsInfo["MaxConfs"] = Options["--maxConfs"]
 551     
 552     RandomSeed = -1
 553     if not re.match("^auto$", Options["--randomSeed"], re.I):
 554         RandomSeed = int(Options["--randomSeed"])
 555     OptionsInfo["RandomSeed"] = RandomSeed
 556     
 557     OptionsInfo["RemoveHydrogens"] = True
 558     if re.match("^no$", Options["--removeHydrogens"], re.I):
 559         OptionsInfo["RemoveHydrogens"] = False
 560     
 561 def RetrieveOptions():
 562     """Retrieve command line arguments and options"""
 563     
 564     # Get options...
 565     global Options
 566     Options = docopt(__doc__)
 567 
 568     # Set current working directory to the specified directory...
 569     WorkingDir = Options["--workingdir"]
 570     if WorkingDir:
 571         os.chdir(WorkingDir)
 572     
 573     # Handle examples option...
 574     if "--examples" in Options and Options["--examples"]:
 575         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 576         sys.exit(0)

 578 def ValidateOptions():
 579     """Validate option values"""
 580     
 581     MiscUtil.ValidateOptionTextValue("-a, --addHydrogens", Options["--addHydrogens"], "yes no")
 582     MiscUtil.ValidateOptionTextValue("--alignConformers", Options["--alignConformers"], "yes no")
 583     MiscUtil.ValidateOptionTextValue("-c, --conformerGenerator", Options["--conformerGenerator"], "SDG ETDG KDG ETKDG")
 584     MiscUtil.ValidateOptionTextValue("-f, --forceField", Options["--forceField"], "UFF MMFF None")
