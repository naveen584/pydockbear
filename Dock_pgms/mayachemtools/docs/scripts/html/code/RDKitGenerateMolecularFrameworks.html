  29 """
  30 RDKitGenerateMolecularFrameworks.py - Generate Bemis Murcko molecular frameworks
  31 
  32 Usage:
  33     RDKitGenerateMolecularFrameworks.py [--infileParams <Name,Value,...>]
  34                                          [--mode <GraphFrameworks or AtomicFrameworks> ]
  35                                          [ --outfileParams <Name,Value,...> ]  [--overwrite] [--removeDuplicates <yes or no>]
  36                                          [--sort <yes or no>] [--sortOrder <ascending or descending>]
  37                                          [--useChirality <yes or no>] [-w <dir>] -i <infile> -o <outfile>
  38     RDKitGenerateMolecularFrameworks.py -h | --help | -e | --examples
  39 
  40 Description:
  41     Generate Bemis Murcko [ Ref 133 ] molecular frameworks for molecules. Two types of molecular
  42     frameworks can be generated: Graph or atomic frameworks. The graph molecular framework
  43     is a generic framework. The atom type, hybridization, and bond order is ignore during its
  44     generation. All atoms are set to carbon atoms and all bonds are single bonds. The atom type,
  45     hybridization, and bond order is preserved during generation of atomic molecular frameworks.
  46 
  47     The supported input file formats are: SD (.sdf, .sd), SMILES (.smi, .csv, .tsv, .txt)
  48 
  49     The supported output file formats are: SD (.sdf, .sd), SMILES (.smi)
  50 
  51 Options:
  52     -e, --examples
  53         Print examples.
  54     -h, --help
  55         Print this help message.
  56     -i, --infile <infile>
  57         Input file name.
  58     --infileParams <Name,Value,...>  [default: auto]
  59         A comma delimited list of parameter name and value pairs for reading
  60         molecules from files. The supported parameter names for different file
  61         formats, along with their default values, are shown below:
  62             
  63             SD: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  64             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  65                 smilesTitleLine,auto,sanitize,yes
  66             
  67         Possible values for smilesDelimiter: space, comma or tab.
  68     -m, --mode <GraphFrameworks or AtomicFrameworks>  [default: GraphFrameworks]
  69         Type of molecular frameworks to generate for molecules. Possible values: GraphFrameworks
  70          or AtomicFrameworks. The graph molecular framework is a generic framework. The atom type,
  71          hybridization, and bond order is ignore during its generation. All atoms are set to carbon atoms
  72          and all bonds are single bonds. The atom type, hybridization, and bond order is preserved
  73          during the generation of atomic molecular frameworks.
  74     -o, --outfile <outfile>
  75         Output file name.
  76     --outfileParams <Name,Value,...>  [default: auto]
  77         A comma delimited list of parameter name and value pairs for writing
  78         molecules to files. The supported parameter names for different file
  79         formats, along with their default values, are shown below:
  80             
  81             SD: compute2DCoords,auto
  82             SMILES: smilesDelimiter,space,smilesTitleLine,yes
  83             
  84         Default value for compute2DCoords: yes for SMILES input file; no for all other
  85         file types.
  86     --overwrite
  87         Overwrite existing files.
  88     -r, --removeDuplicates <yes or no>  [default: no]
  89         Remove duplicate molecular frameworks. Possible values: yes or no. The duplicate
  90         molecular franworks are identified using canonical SMILES. The removed frameworks
  91         are written to a separate output file.
  92     -s, --sort <yes or no>  [default: no]
  93         Sort molecular frameworks by heavy atom count. Possible values: yes or no.
  94     --sortOrder <ascending or descending>  [default: ascending]
  95         Sorting order for molecular frameworks. Possible values: ascending or descending.
  96     -u, --useChirality <yes or no>  [default: yes]
  97         Use stereochemistry for generation of canonical SMILES strings to identify
  98         duplicate molecular frameworks.
  99     -w, --workingdir <dir>
 100         Location of working directory which defaults to the current directory.
 101 
 102 Examples:
 103     To generate graph molecular framworks for molecules and write out a SMILES file,
 104     type:
 105 
 106         % RDKitGenerateMolecularFrameworks.py -i Sample.smi -o SampleOut.smi
 107 
 108     To generate graph molecular framworks, remove duplicate frameworks for molecules
 109     and write out SD files for unique and duplicate frameworks, type:
 110 
 111         % RDKitGenerateMolecularFrameworks.py -m GraphFrameworks -r yes
 112           -i Sample.sdf -o SampleOut.sdf
 113 
 114     To generate atomic molecular framworks, remove duplicate frameworks, sort

 117 
 118         % RDKitGenerateMolecularFrameworks.py -m AtomicFrameworks -r yes
 119           -s yes -i Sample.smi -o SampleOut.smi

 121     To generate graph molecular framworks for molecules in a CSV SMILES file,
 122     SMILES strings in column 1, name in olumn 2, emove duplicate frameworks,

 124     file, type:
 125 
 126         % RDKitGenerateMolecularFrameworks.py -m AtomicFrameworks
 127           --removeDuplicates yes -s yes --sortOrder descending --infileParams

 129           smilesNameColumn,2" --outfileParams "compute2DCoords,yes"

 207 def GenerateMolecularFrameworks():
 208     """Generate Bemis Murcko molecular framworks."""
 209     
 210     Infile = OptionsInfo["Infile"]
 211     Outfile = OptionsInfo["Outfile"]
 212 
 213     UseChirality = OptionsInfo["UseChirality"]
 214 
 215     RemoveDuplicateFrameworks = OptionsInfo["RemoveDuplicateFrameworks"]
 216     UseGraphFrameworks = OptionsInfo["UseGraphFrameworks"]
 217     
 218     SortFrameworks = OptionsInfo["SortFrameworks"]
 219     if SortFrameworks:
 220         FrameworkMolIDs = []
 221         FrameworkMolIDToMolMap = {}
 222         FrameworkMolIDToAtomCountMap = {}
 223         
 224         DuplicateFrameworkMolIDs = []
 225         DuplicateFrameworkMolIDToMolMap = {}
 226         DuplicateFrameworkMolIDToAtomCountMap = {}
 227         
 228     DuplicatesOutfile = ""
 229     if RemoveDuplicateFrameworks:
 230         DuplicatesOutfile = OptionsInfo["DuplicatesOutfile"]
 231 
 232     # Setup a molecule reader...
 233     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 234     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 235     
 236     # Set up a molecular framework  writer...
 237     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 238     if Writer is None:
 239         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 240     
 241     # Set up a duplicate molecular framework writer...    
 242     if RemoveDuplicateFrameworks:
 243         DuplicatesWriter = RDKitUtil.MoleculesWriter(DuplicatesOutfile, **OptionsInfo["OutfileParams"])
 244         if Writer is None:
 245             MiscUtil.PrintError("Failed to setup a writer for duplicates output fie %s " % DuplicatesOutfile)
 246         
 247     if RemoveDuplicateFrameworks:
 248         MiscUtil.PrintInfo("Generating files: %s and %s..." % (Outfile, DuplicatesOutfile))
 249     else:
 250         MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 251 
 252     # Process molecules...
 253     MolCount = 0
 254     ValidMolCount = 0
 255     
 256     FrameworksCount = 0
 257     UniqueFrameworksCount = 0
 258     DuplicateFrameworksCount = 0
 259     
 260     CanonicalSMILESMap = {}
 261     
 262     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 263     
 264     for Mol in Mols:
 265         MolCount += 1
 266         
 267         if Mol is None:
 268             continue
 269         
 270         if RDKitUtil.IsMolEmpty(Mol):
 271             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 272             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 273             continue
 274         
 275         ValidMolCount += 1
 276 
 277         if UseGraphFrameworks:
 278             FrameworksMol = MurckoScaffold.MakeScaffoldGeneric(Mol)
 279         else:
 280             FrameworksMol = MurckoScaffold.GetScaffoldForMol(Mol)
 281 
 282         if Compute2DCoords:
 283             AllChem.Compute2DCoords(FrameworksMol)
 284             
 285         if SortFrameworks:
 286             HeavyAtomCount = FrameworksMol.GetNumHeavyAtoms()
 287 
 288         FrameworksCount += 1
 289         
 290         if RemoveDuplicateFrameworks:
 291             CanonicalSMILES = Chem.MolToSmiles(FrameworksMol, isomericSmiles = UseChirality, canonical = True)
 292             if CanonicalSMILES in CanonicalSMILESMap:
 293                 DuplicateFrameworksCount += 1
 294                 if SortFrameworks:
 295                     # Track duplicate frameworks...
 296                     DuplicateFrameworkMolIDs.append(DuplicateFrameworksCount)
 297                     DuplicateFrameworkMolIDToMolMap[DuplicateFrameworksCount] = FrameworksMol
 298                     DuplicateFrameworkMolIDToAtomCountMap[DuplicateFrameworksCount] = HeavyAtomCount
 299                 else:
 300                     # Write it out...
 301                     DuplicatesWriter.write(FrameworksMol)
 302             else:
 303                 UniqueFrameworksCount += 1
 304                 CanonicalSMILESMap[CanonicalSMILES] = CanonicalSMILES
 305                 if SortFrameworks:
 306                     # Track unique frameworks...
 307                     FrameworkMolIDs.append(UniqueFrameworksCount)
 308                     FrameworkMolIDToMolMap[UniqueFrameworksCount] = FrameworksMol
 309                     FrameworkMolIDToAtomCountMap[UniqueFrameworksCount] = HeavyAtomCount
 310                 else:
 311                     # Write it out...
 312                     Writer.write(FrameworksMol)
 313         elif SortFrameworks:
 314             # Track for sorting...
 315             FrameworkMolIDs.append(FrameworksCount)
 316             FrameworkMolIDToMolMap[FrameworksCount] = FrameworksMol
 317             FrameworkMolIDToAtomCountMap[FrameworksCount] = HeavyAtomCount
 318         else:
 319             # Write it out...
 320             Writer.write(FrameworksMol)
 321             
 322     if SortFrameworks:
 323         ReverseOrder = OptionsInfo["DescendingSortOrder"]
 324         SortAndWriteFrameworks(Writer, FrameworkMolIDs, FrameworkMolIDToMolMap, FrameworkMolIDToAtomCountMap, ReverseOrder)
 325         if RemoveDuplicateFrameworks:
 326             SortAndWriteFrameworks(DuplicatesWriter, DuplicateFrameworkMolIDs, DuplicateFrameworkMolIDToMolMap, DuplicateFrameworkMolIDToAtomCountMap, ReverseOrder)
 327     
 328     Writer.close()
 329     if RemoveDuplicateFrameworks:
 330         DuplicatesWriter.close()
 331 
 332     MiscUtil.PrintInfo("\nTotal number of molecular frameworks: %d" % FrameworksCount)
 333     if RemoveDuplicateFrameworks:
 334         MiscUtil.PrintInfo("Number of unique molecular frameworks: %d" % UniqueFrameworksCount)
 335         MiscUtil.PrintInfo("Number of duplicate molecular frameworks: %d" % DuplicateFrameworksCount)
 336         
 337     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 338     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 339     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 340 
 341 def SortAndWriteFrameworks(MolWriter, MolIDs, MolIDToMolMap, MolIDToAtomCountMap, ReverseOrder):
 342     """Sort frameworks and write them out."""
 343     SortedMolIDs = sorted(MolIDs, key = lambda MolID: MolIDToAtomCountMap[MolID], reverse = ReverseOrder)
 344     for MolID in SortedMolIDs:
 345         FrameworksMol = MolIDToMolMap[MolID]
 346         MolWriter.write(FrameworksMol)
 347 
 348 def ProcessOptions():
 349     """Process and validate command line arguments and options"""
 350     
 351     MiscUtil.PrintInfo("Processing options...")
 352     
 353     # Validate options...
 354     ValidateOptions()
 355     
 356     OptionsInfo["Infile"] = Options["--infile"]
 357     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 358     
 359     OptionsInfo["Outfile"] = Options["--outfile"]
 360     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 361     
 362     OptionsInfo["Overwrite"] = Options["--overwrite"]
 363 
 364     OptionsInfo["Mode"] = Options["--mode"]
 365     OptionsInfo["UseGraphFrameworks"] = False
 366     if re.match("^GraphFrameworks$", OptionsInfo["Mode"], re.I):
 367         OptionsInfo["UseGraphFrameworks"] = True
 368     
 369     OptionsInfo["RemoveDuplicates"] = Options["--removeDuplicates"]
 370     OptionsInfo["RemoveDuplicateFrameworks"] = False
 371     if re.match("^Yes$", OptionsInfo["RemoveDuplicates"], re.I):
 372         OptionsInfo["RemoveDuplicateFrameworks"] = True
 373         
 374     # Setup outfile for writing out duplicates...
 375     OptionsInfo["DuplicatesOutfile"] = ""
 376     if OptionsInfo["RemoveDuplicateFrameworks"]:
 377         FileDir, FileName, FileExt = MiscUtil.ParseFileName(OptionsInfo["Outfile"])
 378         OptionsInfo["DuplicatesOutfile"] = "%sDuplicates.%s" % (FileName, FileExt)
 379 
 380     OptionsInfo["Sort"] = Options["--sort"]
 381     OptionsInfo["SortFrameworks"] = False
 382     if re.match("^Yes$", OptionsInfo["Sort"], re.I):
 383         OptionsInfo["SortFrameworks"] = True
 384     
 385     OptionsInfo["SortOrder"] = Options["--sortOrder"]
 386     OptionsInfo["DescendingSortOrder"] = False
 387     if re.match("^Descending$", OptionsInfo["SortOrder"], re.I):
 388         OptionsInfo["DescendingSortOrder"] = True
 389     
 390     OptionsInfo["UseChirality"] = False
 391     if re.match("^yes$", Options["--useChirality"], re.I):
 392         OptionsInfo["UseChirality"] = True
 393 
 394 def RetrieveOptions():
 395     """Retrieve command line arguments and options"""
 396     
 397     # Get options...
 398     global Options
 399     Options = docopt(__doc__)
 400     
 401     # Set current working directory to the specified directory...
 402     WorkingDir = Options["--workingdir"]
 403     if WorkingDir:
 404         os.chdir(WorkingDir)
 405     
 406     # Handle examples option...
 407     if "--examples" in Options and Options["--examples"]:
 408         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 409         sys.exit(0)
 410 
 411 def ValidateOptions():
 412     """Validate option values"""
 413     
 414     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 415     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 416     
 417     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 418     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 419     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 420     
 421     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "GraphFrameworks AtomicFrameworks")
 422     
 423     MiscUtil.ValidateOptionTextValue("-r, --removeDuplicates", Options["--removeDuplicates"], "yes no")
 424     MiscUtil.ValidateOptionTextValue("-s, --sort", Options["--sort"], "yes no")
 425     MiscUtil.ValidateOptionTextValue("--sortOrder", Options["--sortOrder"], "ascending descending")
 426 
 427     MiscUtil.ValidateOptionTextValue("--useChirality", Options["--useChirality"], "yes no")
 428     
 429 if __name__ == "__main__":
 430     main()
