  29 """
  30 RDKitPerformMinimization.py - Perform structure minimization
  31 
  32 Usage:
  33     RDKitPerformMinimization.py [--addHydrogens <yes or no>] [--conformerGenerator <SDG, ETDG, KDG, ETKDG, None> ]
  34                                 [--forceField <UFF or MMFF>] [--enforceChirality <yes or no>] [--infileParams <Name,Value,...>]
  35                                 [--maxConfs <number>] [--maxIters <number>] [ --outfileParams <Name,Value,...> ]
  36                                 [--overwrite] [ --removeHydrogens <yes or no>] [--randomSeed <number>] [-w <dir>] -i <infile> -o <outfile> 
  37     RDKitPerformMinimization.py -h | --help | -e | --examples
  38 
  39 Description:
  40     Generate 3D structures for molecules using combination of distance geometry
  41     and forcefield minimization or minimize existing 3D structures using a specified
  42     forcefield.
  43 
  44     The supported input file formats are: Mol (.mol), SD (.sdf, .sd), SMILES (.smi)
  45     .csv, .tcsv .txt)
  46 
  47     The supported output file formats are: SD (.sdf, .sd)
  48 
  49 Options:
  50     -a, --addHydrogens <yes or no>  [default: yes]
  51         Add hydrogens before minimization.
  52     -c, --conformerGenerator <SDG, ETDG, KDG, ETKDG, None>  [default: ETKDG]
  53         Conformation generation methodology for generating initial 3D coordinates.
  54         Possible values: Standard Distance Geometry, (SDG), Experimental Torsion-angle
  55         preference with Distance Geometry (ETDG), basic Knowledge-terms with Distance
  56         Geometry (KDG),  and Experimental Torsion-angle preference along with basic
  57         Knowledge-terms with Distance Geometry (ETKDG) [Ref 129] .
  58         
  59         The conformation generation step may be skipped by specifying 'None' value to
  60         perform only forcefield minimization of molecules with 3D structures in input
  61         file.  This doesn't work for molecules in SMILES file or molecules in SD/MOL files
  62         containing 2D structures.
  63     -f, --forceField <UFF or MMFF>  [default: MMFF]
  64         Forcefield method to use for energy minimization. Possible values: Universal Force
  65         Field (UFF) [Ref 81] or Merck Molecular Mechanics Force Field (MMFF) [Ref 83-87] .
  66     --enforceChirality <yes or no>  [default: Yes]
  67         Enforce chirality for defined chiral centers.
  68     -e, --examples
  69         Print examples.
  70     -h, --help
  71         Print this help message.
  72     -i, --infile <infile>
  73         Input file name.
  74     --infileParams <Name,Value,...>  [default: auto]
  75         A comma delimited list of parameter name and value pairs for reading
  76         molecules from files. The supported parameter names for different file
  77         formats, along with their default values, are shown below:
  78             
  79             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  80             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  81                 smilesTitleLine,auto,sanitize,yes
  82             
  83         Possible values for smilesDelimiter: space, comma or tab.
  84     --maxConfs <number>  [default: 250]
  85         Maximum number of conformations to generate for each molecule by conformation
  86         generation methodology for initial 3D coordinates. The conformations are minimized
  87         using the specified forcefield and the lowest energy conformation is written to the
  88         output file. This option is ignored during 'None' value of '-c --conformerGenerator'
  89         option.
  90     --maxIters <number>  [default: 500]
  91         Maximum number of iterations to perform for each molecule during forcefield
  92         minimization.
  93     -o, --outfile <outfile>
  94         Output file name.
  95     --outfileParams <Name,Value,...>  [default: auto]
  96         A comma delimited list of parameter name and value pairs for writing
  97         molecules to files. The supported parameter names for different file
  98         formats, along with their default values, are shown below:
  99             
 100             SD: kekulize,no
 101             
 102     --overwrite
 103         Overwrite existing files.
 104     -r, --removeHydrogens <yes or no>  [default: Yes]
 105         Remove hydrogens after minimization.
 106     --randomSeed <number>  [default: auto]
 107         Seed for the random number generator for reproducing 3D coordinates.
 108         Default is to use a random seed.
 109     -w, --workingdir <dir>
 110         Location of working directory which defaults to the current directory.
 111 
 112 Examples:
 113     To generate up to 250 conformations using ETKDG methodology followed by MMFF
 114     forcefield minimization for a maximum of 500 iterations for molecules in a SMILES file
 115     and write out a SD file containing minimum energy structurse corresponding to each
 116     molecule, type
 117 
 118         % RDKitPerformMinimization.py  -i Sample.smi -o SampleOut.sdf
 119 
 120     To generate up to 150 conformations using ETKDG methodology followed by MMFF
 121     forcefield minimization for a maximum of 250 iterations along with a specified random
 122     seed  for molecules in a SMILES file and write out a SD file containing minimum energy
 123     structures corresponding to each molecule, type
 124 
 125         % RDKitPerformMinimization.py  --maxConfs 150  --randomSeed 201780117 
 126           --maxIters 250  -i Sample.smi -o SampleOut.sdf
 127 
 128     To minimize structures in a 3D SD file using UFF forcefield for a maximum of 150
 129     iterations without generating any conformations and write out a SD file containing
 130     minimum energy structures corresponding to each molecule, type
 131 
 132         % RDKitPerformMinimization.py  -c None -f UFF --maxIters 150
 133           -i Sample3D.sdf -o SampleOut.sdf
 134 
 135     To generate up to 50 conformations using SDG methodology followed
 136     by UFF forcefield minimization for a maximum of 50 iterations for 
 137     molecules in a CSV SMILES file, SMILES strings in column 1, name in
 138     column 2, and write out a SD file, type:
 139 
 140         % RDKitPerformMinimization.py  --maxConfs 50  --maxIters 50 -c SDG
 141           -f UFF --infileParams "smilesDelimiter,comma,smilesTitleLine,yes,
 142           smilesColumn,1,smilesNameColumn,2"  -i SampleSMILES.csv
 143           -o SampleOut.sdf

 219 def PerformMinimization():
 220     """Perform minimization."""
 221     
 222     Infile = OptionsInfo["Infile"]
 223     Outfile = OptionsInfo["Outfile"]
 224     
 225     # Setup a molecule reader...
 226     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 227     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 228     
 229     # Set up a molecule writer...
 230     Writer = None
 231     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 232     if Writer is None:
 233         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 234     MiscUtil.PrintInfo("Generating file %s...\n" % Outfile)
 235 
 236     # Process molecules...
 237     MolCount = 0
 238     ValidMolCount = 0
 239     MinimizationFailedCount = 0
 240 
 241     if OptionsInfo["SkipConformerGeneration"]:
 242         MiscUtil.PrintInfo("Performing minimization without generation of conformers...\n")
 243     else:
 244         MiscUtil.PrintInfo("Performing minimization with generation of conformers...\n")
 245         
 246     for Mol in Mols:
 247         MolCount += 1
 248         
 249         if Mol is None:
 250             continue
 251         
 252         if RDKitUtil.IsMolEmpty(Mol):
 253             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 254             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 255             continue
 256         
 257         ValidMolCount += 1
 258 
 259         if OptionsInfo["SkipConformerGeneration"]:
 260             Status = MinimizeMolecule(Mol, MolCount, Writer)
 261         else:
 262             Status = GenerateAndMinimizeConformers(Mol, MolCount, Writer)
 263         
 264         if not Status:
 265             MinimizationFailedCount += 1
 266             
 267     if Writer is not None:
 268         Writer.close()
 269     
 270     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 271     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 272     MiscUtil.PrintInfo("Number of molecules failed during conformation generation or minimization: %d" % MinimizationFailedCount)
 273     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + MinimizationFailedCount))
 274 
 275 def MinimizeMolecule(Mol, MolCount, Writer):
 276     "Minimize moleculer and write it out"
 277     
 278     if  OptionsInfo["AddHydrogens"]:
 279         Mol = Chem.AddHs(Mol)
 280 
 281     Status = 0
 282     try:
 283         if OptionsInfo["UseUFF"]:
 284             Status = AllChem.UFFOptimizeMolecule(Mol, maxIters = OptionsInfo["MaxIters"])
 285         elif OptionsInfo["UseMMFF"]:
 286             Status = AllChem.MMFFOptimizeMolecule(Mol,  maxIters = OptionsInfo["MaxIters"])
 287         else:
 288             MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 289     except RuntimeError as ErrMsg:
 290         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 291         MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 292         return False
 293     
 294     if Status != 0:
 295         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 296         MiscUtil.PrintWarning("Minimization failed to converge for molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (MolName, OptionsInfo["MaxIters"]))
 297         
 298     if  OptionsInfo["RemoveHydrogens"]:
 299         Mol = Chem.RemoveHs(Mol)
 300     
 301     Writer.write(Mol)
 302     
 303     return True
 304 
 305 def GenerateAndMinimizeConformers(Mol, MolCount, Writer):
 306     "Generate and mininize conformers for a molecule and write out the lowest energy conformer."
 307     
 308     ConfIDs = EmbedMolecule(Mol)
 309     if not len(ConfIDs):
 310         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 311         MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 312         return False
 313 
 314     CalcEnergyMap = {}
 315     for ConfID in ConfIDs:
 316         try:
 317             if OptionsInfo["UseUFF"]:
 318                 Status = AllChem.UFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 319             elif OptionsInfo["UseMMFF"]:
 320                 Status = AllChem.MMFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 321             else:
 322                 MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 323         except RuntimeError as ErrMsg:
 324             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 325             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 326             return False
 327         
 328         Energy = GetConformerEnergy(Mol, ConfID)
 329 
 330         if Status != 0:
 331             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 332             MiscUtil.PrintWarning("Minimization failed to converge for conformation number %d of molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (ConfID, MolName, OptionsInfo["MaxIters"]))
 333             
 334         CalcEnergyMap[ConfID] = Energy
 335 
 336     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
 337     MinEnergyConfID = SortedConfIDs[0]
 338         
 339     if  OptionsInfo["RemoveHydrogens"]:
 340         Mol = Chem.RemoveHs(Mol)
 341     
 342     Writer.write(Mol, confId = MinEnergyConfID)
 343     
 344     return True
 345     
 346 def GetConformerEnergy(Mol, ConfID):
 347     "Calculate conformer energy"
 348 
 349     Energy = 0.0
 350     if OptionsInfo["UseUFF"]:
 351         UFFMoleculeForcefield = AllChem.UFFGetMoleculeForceField(Mol, confId = ConfID)
 352         Energy = UFFMoleculeForcefield.CalcEnergy()
 353     elif OptionsInfo["UseMMFF"]:
 354         MMFFMoleculeProperties = AllChem.MMFFGetMoleculeProperties(Mol)
 355         MMFFMoleculeForcefield = AllChem.MMFFGetMoleculeForceField(Mol, MMFFMoleculeProperties, confId = ConfID)
 356         Energy = MMFFMoleculeForcefield.CalcEnergy()
 357     else:
 358         MiscUtil.PrintError("Couldn't retrieve conformer energy: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 359     
 360     return Energy
 361     
 362 def EmbedMolecule(Mol):
 363     "Embed conformations"
 364     
 365     ConfIDs = []
 366     
 367     MaxConfs = OptionsInfo["MaxConfs"]
 368     RandomSeed = OptionsInfo["RandomSeed"]
 369     EnforceChirality = OptionsInfo["EnforceChirality"]
 370     UseExpTorsionAnglePrefs = OptionsInfo["UseExpTorsionAnglePrefs"]
 371     UseBasicKnowledge = OptionsInfo["UseBasicKnowledge"]
 372 
 373     ConfIDs = AllChem.EmbedMultipleConfs(Mol, numConfs = MaxConfs, randomSeed = RandomSeed, enforceChirality = EnforceChirality, useExpTorsionAnglePrefs = UseExpTorsionAnglePrefs, useBasicKnowledge = UseBasicKnowledge)
 374     
 375     return ConfIDs

 377 def ProcessOptions():
 378     """Process and validate command line arguments and options"""
 379     
 380     MiscUtil.PrintInfo("Processing options...")
 381     
 382     # Validate options...
 383     ValidateOptions()
 384     
 385     OptionsInfo["Infile"] = Options["--infile"]
 386     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 387     
 388     OptionsInfo["Outfile"] = Options["--outfile"]
 389     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 390     
 391     OptionsInfo["Overwrite"] = Options["--overwrite"]
 392 
 393     OptionsInfo["AddHydrogens"] = True
 394     if re.match("^no$", Options["--addHydrogens"], re.I):
 395         OptionsInfo["AddHydrogens"] = False
 396 
 397     if re.match("^ETDG$", Options["--conformerGenerator"], re.I):
 398         ConformerGenerator = "ETDG"
 399         UseExpTorsionAnglePrefs = True
 400         UseBasicKnowledge = False
 401         SkipConformerGeneration = False
 402     elif re.match("^KDG$", Options["--conformerGenerator"], re.I):
 403         ConformerGenerator = "KDG"
 404         UseExpTorsionAnglePrefs = False
 405         UseBasicKnowledge = True
 406         SkipConformerGeneration = False
 407     elif re.match("^ETKDG$", Options["--conformerGenerator"], re.I):
 408         ConformerGenerator = "ETKDG"
 409         UseExpTorsionAnglePrefs = True
 410         UseBasicKnowledge = True
 411         SkipConformerGeneration = False
 412     elif re.match("^SDG$", Options["--conformerGenerator"], re.I):
 413         ConformerGenerator = "SDG"
 414         UseExpTorsionAnglePrefs = False
 415         UseBasicKnowledge = False
 416         SkipConformerGeneration = False
 417     else:
 418         ConformerGenerator = "None"
 419         UseExpTorsionAnglePrefs = False
 420         UseBasicKnowledge = False
 421         SkipConformerGeneration = True
 422     
 423     OptionsInfo["SkipConformerGeneration"] = SkipConformerGeneration
 424     OptionsInfo["ConformerGenerator"] = ConformerGenerator
 425     OptionsInfo["UseExpTorsionAnglePrefs"] = UseExpTorsionAnglePrefs
 426     OptionsInfo["UseBasicKnowledge"] = UseBasicKnowledge
 427 
 428     if re.match("^UFF$", Options["--forceField"], re.I):
 429         ForceField = "UFF"
 430         UseUFF = True
 431         UseMMFF = False
 432     elif re.match("^MMFF$", Options["--forceField"], re.I):
 433         ForceField = "MMFF"
 434         UseUFF = False
 435         UseMMFF = True
 436     
 437     OptionsInfo["ForceField"] = ForceField
 438     OptionsInfo["UseMMFF"] = UseMMFF
 439     OptionsInfo["UseUFF"] = UseUFF
 440         
 441     OptionsInfo["EnforceChirality"] = True
 442     if re.match("^no$", Options["--enforceChirality"], re.I):
 443         OptionsInfo["EnforceChirality"] = False
 444     
 445     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 446     OptionsInfo["MaxConfs"] = int(Options["--maxConfs"])
 447     
 448     RandomSeed = -1
 449     if not re.match("^auto$", Options["--randomSeed"], re.I):
 450         RandomSeed = int(Options["--randomSeed"])
 451     OptionsInfo["RandomSeed"] = RandomSeed
 452     
 453     OptionsInfo["RemoveHydrogens"] = True
 454     if re.match("^no$", Options["--removeHydrogens"], re.I):
 455         OptionsInfo["RemoveHydrogens"] = False
 456 
 457 def RetrieveOptions():
 458     """Retrieve command line arguments and options"""
 459     
 460     # Get options...
 461     global Options
 462     Options = docopt(__doc__)
 463     
 464     # Set current working directory to the specified directory...
 465     WorkingDir = Options["--workingdir"]
 466     if WorkingDir:
 467         os.chdir(WorkingDir)
 468     
 469     # Handle examples option...
 470     if "--examples" in Options and Options["--examples"]:
 471         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 472         sys.exit(0)

 478     MiscUtil.ValidateOptionTextValue("-c, --conformerGenerator", Options["--conformerGenerator"], "SDG ETDG KDG ETKDG None")
 479     MiscUtil.ValidateOptionTextValue("-f, --forceField", Options["--forceField"], "UFF MMFF")
 480     
 481     MiscUtil.ValidateOptionTextValue("--enforceChirality ", Options["--enforceChirality"], "yes no")
 482     
 483     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 484     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi txt csv tsv")
 485     
 486     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd")
 487     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 488     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 489         
 490     MiscUtil.ValidateOptionNumberValue("--maxConfs", int(Options["--maxConfs"]), {">": 0})
 491     MiscUtil.ValidateOptionNumberValue("--maxIters", int(Options["--maxIters"]), {">": 0})
 492     
 493     if not re.match("^auto$", Options["--randomSeed"], re.I):
 494         RandomSeed = int(Options["--randomSeed"])
 495     
 496     MiscUtil.ValidateOptionTextValue("-r, --removeHydrogens", Options["--removeHydrogens"], "yes no")
 497 
 498 if __name__ == "__main__":
