  29 """
  30 RDKitRemoveDuplicateMolecules.py - Remove duplicate molecules
  31 
  32 Usage:
  33     RDKitRemoveDuplicateMolecules.py  [--infileParams <Name,Value,...>]
  34                               [--mode <remove or count>] [ --outfileParams <Name,Value,...> ] 
  35                               [--overwrite] [--useChirality <yes or no>] [-w <dir>] [-o <outfile>]  -i <infile>
  36     RDKitRemoveDuplicateMolecules.py -h | --help | -e | --examples
  37 
  38 Description:
  39     Identify and remove duplicate molecules based on canonical SMILES strings or
  40     simply count the number of duplicate molecules.
  41 
  42     The supported input file formats are: SD (.sdf, .sd), SMILES (.smi., csv, .tsv, .txt)
  43 
  44     The supported output file formats are: SD (.sdf, .sd), SMILES (.smi)
  45 
  46 Options:
  47     -e, --examples
  48         Print examples.
  49     -h, --help
  50         Print this help message.
  51     -i, --infile <infile>
  52         Input file name.
  53     --infileParams <Name,Value,...>  [default: auto]
  54         A comma delimited list of parameter name and value pairs for reading
  55         molecules from files. The supported parameter names for different file
  56         formats, along with their default values, are shown below:
  57             
  58             SD: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  59             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  60                 smilesTitleLine,auto,sanitize,yes
  61             
  62         Possible values for smilesDelimiter: space, comma or tab.
  63     -m, --mode <remove or count>  [default: remove]
  64         Specify whether to remove duplicate molecules and write out filtered molecules
  65         to output files or or simply count the number of duplicate molecules.
  66     -o, --outfile <outfile>
  67         Output file name.
  68     --outfileParams <Name,Value,...>  [default: auto]
  69         A comma delimited list of parameter name and value pairs for writing
  70         molecules to files. The supported parameter names for different file
  71         formats, along with their default values, are shown below:
  72             
  73             SD: compute2DCoords,auto,kekulize,no
  74             SMILES: kekulize,no,smilesDelimiter,space, smilesIsomeric,yes,
  75                 smilesTitleLine,yes
  76             
  77         Default value for compute2DCoords: yes for SMILES input file; no for all other
  78         file types.
  79     --overwrite
  80         Overwrite existing files.
  81     -u, --useChirality <yes or no>  [default: yes]
  82         Use stereochemistry information for generation of canonical SMILES strings
  83         to identify duplicate molecules.
  84     -w, --workingdir <dir>
  85         Location of working directory which defaults to the current directory.
  86 
  87 Examples:
  88     To remove duplicate molecules and generate output files containing unique and
  89     duplicate SMILES strings, type:
  90 
  91         % RDKitRemoveDuplicateMolecules.py -i Sample.smi -o SampleOut.smi
  92 
  93     To remove duplicate molecules without using stereochemistry information for
  94     generation of canonical SMILES and generate output files containing unique and
  95     duplicate SMILES strings, type:
  96 
  97         % RDKitRemoveDuplicateMolecules.py -u no -i Sample.sdf -o SampleOut.sdf
  98 
  99     To count number of unique and duplicate molecules without generating any

 142 # RDKit imports...
 143 try:
 144     from rdkit import rdBase
 145     from rdkit import Chem
 146     from rdkit.Chem import AllChem

 148     sys.stderr.write("\nFailed to import RDKit module/package: %s\n" % ErrMsg)

 152 # MayaChemTools imports...
 153 try:
 154     from docopt import docopt
 155     import MiscUtil
 156     import RDKitUtil
 157 except ImportError as ErrMsg:
 158     sys.stderr.write("\nFailed to import MayaChemTools module/package: %s\n" % ErrMsg)
 159     sys.stderr.write("Check/update your MayaChemTools environment and try again.\n\n")
 160     sys.exit(1)

 162 ScriptName = os.path.basename(sys.argv[0])
 163 Options = {}
 164 OptionsInfo = {}
 165 
 166 def main():
 167     """Start execution of the script"""
 168     
 169     MiscUtil.PrintInfo("\n%s (RDK v%s): Starting...\n" % (ScriptName, rdBase.rdkitVersion))
 170     
 171     (WallClockTime, ProcessorTime) = MiscUtil.GetWallClockAndProcessorTime()
 172     
 173     # Retrieve command line arguments and options...
 174     RetrieveOptions()
 175     
 176     # Process and validate command line arguments and options...
 177     ProcessOptions()
 178     
 179     # Perform actions required by the script...
 180     RemoveDuplicates()
 181     
 182     MiscUtil.PrintInfo("\n%s: Done...\n" % ScriptName)
 183     MiscUtil.PrintInfo("Total time: %s" % MiscUtil.GetFormattedElapsedTime(WallClockTime, ProcessorTime))
 184 
 185 def RemoveDuplicates():
 186     """Identify and remove duplicate molecules based on canonical SMILES"""
 187     
 188     Infile = OptionsInfo["Infile"]
 189     Outfile = OptionsInfo["Outfile"]
 190     DuplicatesOutfile = OptionsInfo["DuplicatesOutfile"]
 191     
 192     CountMode = OptionsInfo["CountMode"]
 193     UseChirality = OptionsInfo["UseChirality"]
 194     
 195     # Setup a molecule reader...
 196     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 197     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 198     
 199     # Set up a molecule writer...
 200     Writer = None
 201     DuplicatesWriter = None
 202     if not CountMode:
 203         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 204         if Writer is None:
 205             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 206         DuplicatesWriter = RDKitUtil.MoleculesWriter(DuplicatesOutfile, **OptionsInfo["OutfileParams"])
 207         if DuplicatesWriter is None:
 208             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % DuplicatesOutfile)
 209         
 210         MiscUtil.PrintInfo("Generating files %s and %s..." % (Outfile, DuplicatesOutfile))
 211 
 212     # Process molecules...
 213     MolCount = 0
 214     ValidMolCount = 0
 215     
 216     UniqueMolCount = 0
 217     DuplicateMolCount = 0
 218     
 219     CanonicalSMILESMap = {}
 220     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 221 
 222     for Mol in Mols:
 223         MolCount += 1
 224         
 225         if Mol is None:
 226             continue
 227         
 228         if RDKitUtil.IsMolEmpty(Mol):
 229             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 230             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 231             continue
 232         
 233         ValidMolCount += 1
 234         
 235         CanonicalSMILES = Chem.MolToSmiles(Mol, isomericSmiles = UseChirality, canonical = True)
 236         
 237         if Compute2DCoords:
 238             if not CountMode:
 239                 AllChem.Compute2DCoords(Mol)
 240         
 241         if CanonicalSMILES in CanonicalSMILESMap:
 242             DuplicateMolCount += 1
 243             if not CountMode:
 244                 DuplicatesWriter.write(Mol)
 245         else:
 246             UniqueMolCount += 1
 247             CanonicalSMILESMap[CanonicalSMILES] = CanonicalSMILES
 248             if not CountMode:
 249                 Writer.write(Mol)
 250     
 251     if Writer is not None:
 252         Writer.close()
 253     
 254     if DuplicatesWriter is not None:
 255         DuplicatesWriter.close()
 256     
 257     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 258     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 259     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 260 
 261     MiscUtil.PrintInfo("\nTotal number of unique molecules: %d" % UniqueMolCount)
 262     MiscUtil.PrintInfo("Total number of duplicate molecules: %d" % DuplicateMolCount)
 263 
 264 def ProcessOptions():
 265     """Process and validate command line arguments and options"""
 266     
 267     MiscUtil.PrintInfo("Processing options...")
 268     
 269     # Validate options...
 270     ValidateOptions()
 271     
 272     OptionsInfo["Infile"] = Options["--infile"]
 273     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 274     
 275     OptionsInfo["Outfile"] = Options["--outfile"]
 276     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 277     
 278     OptionsInfo["Overwrite"] = Options["--overwrite"]
 279 
 280     OptionsInfo["CountMode"] = False
 281     if re.match("^count$", Options["--mode"], re.I):
 282         OptionsInfo["CountMode"] = True
 283     
 284     # Setup outfile for writing out duplicates...
 285     OptionsInfo["DuplicatesOutfile"] = ""
 286     if not OptionsInfo["CountMode"] :
 287         FileDir, FileName, FileExt = MiscUtil.ParseFileName(OptionsInfo["Outfile"])
 288         OptionsInfo["DuplicatesOutfile"] = "%sDuplicates.%s" % (FileName, FileExt)
 289 
 290     OptionsInfo["UseChirality"] = False
 291     if re.match("^yes$", Options["--useChirality"], re.I):
 292         OptionsInfo["UseChirality"] = True
 293     
 294 def RetrieveOptions():
 295     """Retrieve command line arguments and options"""
 296     
 297     # Get options...
 298     global Options
 299     Options = docopt(__doc__)
 300     
 301     # Set current working directory to the specified directory...
 302     WorkingDir = Options["--workingdir"]
 303     if WorkingDir:
 304         os.chdir(WorkingDir)
 305     
 306     # Handle examples option...
 307     if "--examples" in Options and Options["--examples"]:
 308         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 309         sys.exit(0)
 310 
 311 def ValidateOptions():
 312     """Validate option values"""
 313     
 314     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 315     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 316     
 317     if Options["--outfile"]:
 318         MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 319         MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 320         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 321 
 322     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "remove count")
 323     if re.match("^remove$", Options["--mode"], re.I):
 324         if not Options["--outfile"]:
 325             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"remove\" value of \"-m, --mode\" option")
 326         
 327     MiscUtil.ValidateOptionTextValue("--useChirality", Options["--useChirality"], "yes no")
 328     
 329 if __name__ == "__main__":
 330     main()
