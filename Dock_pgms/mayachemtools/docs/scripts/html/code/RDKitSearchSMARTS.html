  29 """
  30 RDKitSearchSMARTS.py - Perform a substructure search using SMARTS pattern
  31 
  32 Usage:
  33     RDKitSearchSMARTS.py  [--infileParams <Name,Value,...>] [--mode <retrieve or count>]
  34                           [--negate <yes or no>] [--outfileParams <Name,Value,...>] [--overwrite]
  35                           [--useChirality <yes or no>] [-w <dir>] [-o <outfile>] -p <SMARTS> -i <infile>
  36     RDKitSearchSMARTS.py -h | --help | -e | --examples
  37 
  38 Description:
  39     Perform a substructure search in an input file using specified SMARTS pattern and
  40     write out the matched molecules to an output file or simply count the number
  41     of matches.
  42 
  43     The supported input file formats are: SD (.sdf, .sd), SMILES (.smi., csv, .tsv, .txt)
  44 
  45     The supported output file formats are: SD (.sdf, .sd), SMILES (.smi)
  46 
  47 Options:
  48     -e, --examples
  49         Print examples.
  50     -h, --help
  51         Print this help message.
  52     -i, --infile <infile>
  53         Input file name.
  54     --infileParams <Name,Value,...>  [default: auto]
  55         A comma delimited list of parameter name and value pairs for reading
  56         molecules from files. The supported parameter names for different file
  57         formats, along with their default values, are shown below:
  58             
  59             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
  60             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
  61                 smilesTitleLine,auto,sanitize,yes
  62             
  63         Possible values for smilesDelimiter: space, comma or tab.
  64     -m, --mode <retrieve or count>  [default: retrieve]
  65         Specify whether to retrieve and write out matched molecules to an output
  66         file or simply count the number of matches.
  67     -n, --negate <yes or no>  [default: no]
  68         Specify whether to find molecules not matching the specified SMARTS pattern.
  69     -o, --outfile <outfile>
  70         Output file name.
  71     --outfileParams <Name,Value,...>  [default: auto]
  72         A comma delimited list of parameter name and value pairs for writing
  73         molecules to files. The supported parameter names for different file
  74         formats, along with their default values, are shown below:
  75             
  76             SD: compute2DCoords,auto,kekulize,no
  77             SMILES: kekulize,no,smilesDelimiter,space, smilesIsomeric,yes,
  78                 smilesTitleLine,yes
  79             
  80         Default value for compute2DCoords: yes for SMILES input file; no for all other
  81         file types.
  82     --overwrite
  83         Overwrite existing files.
  84     -p, --pattern <SMARTS>  [default: none]
  85         SMARTS pattern for performing search.
  86     -u, --useChirality <yes or no>  [default: no]
  87         Use stereochemistry information for SMARTS search.
  88     -w, --workingdir <dir>
  89         Location of working directory which defaults to the current directory.
  90 
  91 Examples:
  92     To retrieve molecules containing the substructure corresponding to a specified
  93     SMARTS pattern and write out a SMILES file, type: 
  94 
  95         % RDKitSearchSMARTS.py -p 'c1ccccc1' -i Sample.smi -o SampleOut.smi
  96 
  97     To only count the number of molecules containing the substructure corresponding
  98     to a specified SMARTS pattern without writing out any file, type: 
  99 
 100         % RDKitSearchSMARTS.py -m count -p 'c1ccccc1' -i Sample.smi
 101 
 102     To count the number of molecules in a SD file not containing the substructure
 103     corresponding to a specified SMARTS pattern and write out a SD file, type: 
 104 
 105         % RDKitSearchSMARTS.py -n yes -p 'c1ccccc1' -i Sample.sdf -o SampleOut.sdf
 106 
 107     To retrieve molecules containing the substructure corresponding to a specified
 108     SMARTS pattern from a CSV SMILES file, SMILES strings in column 1, name in
 109     and write out a SD file, type: 
 110 
 111         % RDKitSearchSMARTS.py -p 'c1ccccc1' --infileParams
 112           "smilesDelimiter,comma,smilesTitleLine,yes,smilesColumn,1,
 113           smilesNameColumn,2" --outfileParams "compute2DCoords,yes"
 114           -i SampleSMILES.csv -o SampleOut.sdf

 137 from __future__ import print_function

 139 # Add local python path to the global path and import appropriate modules...
 140 import os
 141 import sys;  sys.path.insert(0, os.path.join(os.path.dirname(sys.argv[0]), "..", "lib", "Python"))
 142 import time
 143 import re
 144 
 145 # RDKit imports...
 146 try:
 147     from rdkit import rdBase
 148     from rdkit import Chem
 149     from rdkit.Chem import AllChem
 150 except ImportError as ErrMsg:
 151     sys.stderr.write("\nFailed to import RDKit module/package: %s\n" % ErrMsg)
 152     sys.stderr.write("Check/update your RDKit environment and try again.\n\n")
 153     sys.exit(1)
 154 
 155 # MayaChemTools imports...
 156 try:
 157     from docopt import docopt
 158     import MiscUtil
 159     import RDKitUtil
 160 except ImportError as ErrMsg:
 161     sys.stderr.write("\nFailed to import MayaChemTools module/package: %s\n" % ErrMsg)
 162     sys.stderr.write("Check/update your MayaChemTools environment and try again.\n\n")
 163     sys.exit(1)
 164 
 165 ScriptName = os.path.basename(sys.argv[0])
 166 Options = {}
 167 OptionsInfo = {}
 168 
 169 def main():
 170     """Start execution of the script"""
 171     
 172     MiscUtil.PrintInfo("\n%s (RDK v%s): Starting...\n" % (ScriptName, rdBase.rdkitVersion))
 173     
 174     (WallClockTime, ProcessorTime) = MiscUtil.GetWallClockAndProcessorTime()
 175     
 176     # Retrieve command line arguments and options...
 177     RetrieveOptions()
 178     
 179     # Process and validate command line arguments and options...
 180     ProcessOptions()
 181     
 182     # Perform actions required by the script...
 183     PerformSearch()
 184     
 185     MiscUtil.PrintInfo("\n%s: Done...\n" % ScriptName)
 186     MiscUtil.PrintInfo("Total time: %s" % MiscUtil.GetFormattedElapsedTime(WallClockTime, ProcessorTime))
 187 
 188 def PerformSearch():
 189     """Perform search using specified SMARTS pattern."""
 190     
 191     Infile = OptionsInfo["Infile"]
 192     Outfile = OptionsInfo["Outfile"]
 193     
 194     CountMode = OptionsInfo["CountMode"]
 195     NegateMatch = OptionsInfo["NegateMatch"]
 196     UseChirality = OptionsInfo["UseChirality"]
 197 
 198     # Set up a pattern molecule...
 199     PatternMol = Chem.MolFromSmarts(OptionsInfo["Pattern"])
 200     
 201     # Setup a molecule reader...
 202     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 203     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 204     
 205     # Set up a molecule writer...
 206     Writer = None
 207     if not CountMode:
 208         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 209         if Writer is None:
 210             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 211         MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 212 
 213     # Process molecules...
 214     MolCount = 0
 215     ValidMolCount = 0
 216     MatchCount = 0
 217 
 218     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 219     
 220     for Mol in Mols:
 221         MolCount += 1
 222         
 223         if Mol is None:
 224             continue
 225         
 226         if RDKitUtil.IsMolEmpty(Mol):
 227             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 228             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 229             continue
 230         
 231         ValidMolCount += 1
 232 
 233         MolMatched = Mol.HasSubstructMatch(PatternMol, useChirality = UseChirality)
 234         if MolMatched != NegateMatch:
 235             MatchCount += 1
 236             if not CountMode:
 237                 if Compute2DCoords:
 238                     AllChem.Compute2DCoords(Mol)
 239                 Writer.write(Mol)
 240     
 241     if Writer is not None:
 242         Writer.close()
 243     
 244     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 245     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 246     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 247 
 248     if NegateMatch:
 249         MiscUtil.PrintInfo("\nTotal number of not matched molecules: %d" % MatchCount)
 250     else:
 251         MiscUtil.PrintInfo("\nTotal number of matched molecules: %d" % MatchCount)
 252 
 253 def ProcessOptions():
 254     """Process and validate command line arguments and options"""
 255     
 256     MiscUtil.PrintInfo("Processing options...")
 257     
 258     # Validate options...
 259     ValidateOptions()
 260     
 261     OptionsInfo["Infile"] = Options["--infile"]
 262     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 263     
 264     OptionsInfo["Outfile"] = Options["--outfile"]
 265     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 266     
 267     OptionsInfo["Overwrite"] = Options["--overwrite"]
 268 
 269     OptionsInfo["CountMode"] = False
 270     if re.match("^count$", Options["--mode"], re.I):
 271         OptionsInfo["CountMode"] = True
 272     
 273     OptionsInfo["NegateMatch"] = False
 274     if re.match("^yes$", Options["--negate"], re.I):
 275         OptionsInfo["NegateMatch"] = True
 276     
 277     OptionsInfo["Pattern"] = Options["--pattern"]
 278     
 279     OptionsInfo["UseChirality"] = False
 280     if re.match("^yes$", Options["--useChirality"], re.I):
 281         OptionsInfo["UseChirality"] = True
 282 
 283 def RetrieveOptions():
 284     """Retrieve command line arguments and options"""
 285     
 286     # Get options...
 287     global Options
 288     Options = docopt(__doc__)
 289     
 290     # Set current working directory to the specified directory...
 291     WorkingDir = Options["--workingdir"]
 292     if WorkingDir:
 293         os.chdir(WorkingDir)
 294     
 295     # Handle examples option...
 296     if "--examples" in Options and Options["--examples"]:
 297         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(__doc__))
 298         sys.exit(0)
 299 
 300 def ValidateOptions():
 301     """Validate option values"""
 302     
 303     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 304     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 305     if Options["--outfile"]:
 306         MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 307         MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 308         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 309 
 310     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "retrieve count")
 311     if re.match("^retrieve$", Options["--mode"], re.I):
 312         if not Options["--outfile"]:
 313             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"retrieve\" value of \"-m, --mode\" option")
 314         
 315     MiscUtil.ValidateOptionTextValue("-n, --negate", Options["--negate"], "yes no")
 316     
 317     PatternMol = Chem.MolFromSmarts(Options["--pattern"])
 318     if PatternMol is None:
 319         MiscUtil.PrintError("The value specified, %s, using option \"-p, --pattern\" is not a valid SMARTS: Failed to create pattern molecule" % Options["--pattern"])
 320     
 321     MiscUtil.ValidateOptionTextValue("--useChirality", Options["--useChirality"], "yes no")
 322 
 323 if __name__ == "__main__":
 324     main()
